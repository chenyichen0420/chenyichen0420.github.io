思路分析

非常无脑的 log2 解法。

一个简单的观察是答案为 ∑i=0n−1​minj=0i​aj​，正确性比较显然：能利用上的都利用上了。

我们考虑使用“积木”证明上述结论并得到做法：

我们在第 i 列竖直放置 ai​ 个积木，如图所示：

显然，按照上述观察，答案就是每一行从第 0 列开始向右的最远连续段的长度之和。而像这样放置显然最优：如果我只将一块积木向上移动，那么一定不如在最底层放着优秀。

然后，加入一个积木是有贡献的，当且仅当它所在行的左边都有积木。形式化的，添加之后的 bi​≤bj​(∀j<i)。

那么贡献具体有多少呢？他将他自己与右边的一部分积木连接到了左边界上，产生了一定的贡献。形式化的，这个贡献为 k−i+1(k=maxj,∀p∈[i,j],ai​≤ap​)。

显然，后面这个东西可以使用线段树与二分直接 O(log2) 维护出来。删除是类似的。

代码如下：

#include<bits/stdc++.h>
using namespace std;
#define int long long
int n, a[200005], ans, m, x, y;
struct seg_tree {
	struct node { int l, r, v; }re[200005 << 2];
	inline void build(int l, int r, int p) {
		re[p].l = l; re[p].r = r; int mid = l + r >> 1;
		if (l == r) return void(re[p].v = a[l]);
		build(mid + 1, r, p << 1 | 1);
		build(l, mid, p << 1);
		re[p].v = min(re[p << 1].v, re[p << 1 | 1].v);
	}
	inline void ins(int cp, int cv, int p) {
		if (re[p].l == re[p].r) return void(re[p].v = cv);
		ins(cp, cv, p << 1 | (cp > re[p << 1].r));
		re[p].v = min(re[p << 1].v, re[p << 1 | 1].v);
	}
	inline int que(int l, int r, int p) {
		if (re[p].l >= l && re[p].r <= r) return re[p].v;
		int ret = 1e9;
		if (l <= re[p << 1].r) ret = min(ret, que(l, r, p << 1));
		if (r > re[p << 1].r) ret = min(ret, que(l, r, p << 1 | 1));
		return ret;
	}
}sgt;
signed main() {
	ios::sync_with_stdio(0); cin >> n;
	for (int i = 0, ma = 1e9; i != n; ++i)
		cin >> a[i], ma = min(ma, a[i]), ans += ma;
	sgt.build(0, n - 1, 1); cout << ans << endl;
	for (cin >> m; m; m--)
		if (cin >> x >> y, x & 1) {
			a[y]++; sgt.ins(y, a[y], 1);
			if (!y || sgt.que(0, y - 1, 1) >= a[y]) {
				int l = y, r = n - 1, mid;
				while (l != r) {
					mid = l + r + 1 >> 1;
					if (sgt.que(y, mid, 1) >= a[y]) l = mid;
					else r = mid - 1;
				}
				ans += l - y + 1;
			}
			cout << ans << endl;
		}
		else {
			if (!y || sgt.que(0, y - 1, 1) >= a[y]) {
				int l = y, r = n - 1, mid;
				while (l != r) {
					mid = l + r + 1 >> 1;
					if (sgt.que(y, mid, 1) >= a[y]) l = mid;
					else r = mid - 1;
				}
				ans -= l - y + 1;
			}
			a[y]--; sgt.ins(y, a[y], 1);
			cout << ans << endl;
		}
}
