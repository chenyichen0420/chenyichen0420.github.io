::::success[正解]
正如提示所说，我们考虑一下并查集和倍增可以玩出什么花来。

我们思考，倍增的本质是给一个长度为 2i 的区间打上一个区间共有的标记，然后对于一般的修改或查询，使用 log 个区间拼起来，得到结果。

比如说在这道题中，两个长度都为 2i 的区间对应位置相等时，我们可以考虑一下将这一层的两个区间左端点合并一下。

下传标记呢？我们注意到 [i,i+2k)=[j,j+2k)⟺[i,i+2k−1)=[j,j+2k−1)&[i+2k−1,i+2k)=[j+2k−1,j+2k)。也就是说标记可以轻松的下传到下辖的两个小区间。

至此，此题可解。

::::success[实现]

#include<bits/stdc++.h>
using namespace std;
#define int long long
int n,m,l1,l2,ln,ans,c;
struct mss{
    int f[500005];
    mss(){ iota(f,f+500001,0); }
    inline int find(int p){
        return f[p]==p?p:f[p]=find(f[p]);
    }
    inline void merge(int l,int r){
        f[find(l)]=find(r);
    }
}ms[20];
signed main() {
	ios::sync_with_stdio(0);
    cin>>n>>m;
    for(int i=1;i<=m;++i){
        cin>>l1>>l2>>ln;
        for(int j=19;j>=0;j--)
            if(ln>>j&1)
                ms[j].merge(l1,l2),
                l1+=1ll<<j,l2+=1ll<<j;
    }
    for(int j=19;j;j--)
        for(int i=1;i+(1ll<<j)-1<=n;++i)
            ms[j-1].merge(ms[j].find(i),i),
            ms[j-1].merge(ms[j].find(i)+(1ll<<j-1),i+(1ll<<j-1));
    for(int i=1;i<=n;++i) c+=ms[0].find(i)==i;
    cout<<c<<endl;
}
