思路分析

这题相对特别一点的地方就是把求贡献转化成了求期望。

其实，点到这里就已经差不多了。我们将每次操作的执行概率设为 12\frac{1}{2}21​，最后再将符合要求的状态的期望乘上 2m2^m2m，求一下和就做完了。

初始状态：dpi,j=[ai>aj]dp_{i,j}=[a_i>a_j]dpi,j​=[ai​>aj​]。

状态转移：

dpx,y=dpy,x=(dpx,y+dpy,x)/2dp_{x,y}=dp_{y,x}=(dp_{x,y}+dp_{y,x})/2dpx,y​=dpy,x​=(dpx,y​+dpy,x​)/2；

dpj,l=dpj,r=(dpj,l+dpj,r)/2dp_{j,l}=dp_{j,r}=(dp_{j,l}+dp_{j,r})/2dpj,l​=dpj,r​=(dpj,l​+dpj,r​)/2；

dpl,j=dpr,j=(dpl,j+dpr,j)/2dp_{l,j}=dp_{r,j}=(dp_{l,j}+dp_{r,j})/2dpl,j​=dpr,j​=(dpl,j​+dpr,j​)/2；

其实说白了就是这两个数一换就把和他们相关的概率给摇匀了。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
constexpr int mod = 1e9 + 7;
int n, m, a[3005], dp[3005][3005], ny, ti, ans;
signed main() {
	ios::sync_with_stdio(0);
	cin >> n >> m; ny = mod + 1 >> 1; ti = 1;
	for (int i = 1; i <= n; ++i) cin >> a[i];
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= n; ++j)
			dp[i][j] = (a[i] > a[j]);
	for (int i = 1, l, r; i <= m; ++i) {
		cin >> l >> r; ti = ti * 2 % mod;
		dp[l][r] = dp[r][l] = (dp[l][r] + dp[r][l]) * ny % mod;
		for (int j = 1; j <= n; ++j)
			if (j != l && j != r)
				dp[j][l] = dp[j][r] = (dp[j][l] + dp[j][r]) * ny % mod,
				dp[l][j] = dp[r][j] = (dp[l][j] + dp[r][j]) * ny % mod;
	}
	for (int i = 1; i <= n; ++i)
		for (int j = i + 1; j <= n; ++j)
			ans = (ans + dp[i][j] * ti) % mod;
	cout << ans << endl;
	return 0;
}
