思路分析

简单题。

首先，注意到变换次数较多，于是考虑一下能不能压缩变换的次数。

变换一次，数最大为 9000，两次最大为 35，三次最大为 11，四次就变成一位数了，此后肯定不再变化。

由此得出结论：m>9000 时答案为 0，剩下的情况可以 min(4,k)→k。

接下来考虑求解答案。不难想到计算出操作 1 次之后各种和的出现的次数，进而快速的计算出操作 2,3,4 次时的答案。

由于有上限，于是非常套路的考虑数位 DP。注意到紧贴上界的情况是简单的。当与上界分离的时候，本质上就是多次长度为 x 的数（可以有前导零）的数位和是 y 的方案数。

显然后者可以先进行一次 DP 预处理出来，然后再进行求解。

时间复杂度为 O(n2)，其中 n=log10​N。代码如下：

#include<bits/stdc++.h>
using namespace std;
#define int long long
constexpr int mod = 1e9 + 7;
int t, m, k, ans[5][9005], dp[1005][9005]; string n;
inline void gen(int p, int s) {
	if (p == n.size()) {
		ans[1][s]++; return;
	}
	for (int i = 0; i != (n[p] ^ 48); ++i) {
		int ts = s + i, ws = n.size() - p - 1;
		for (int j = 0; j <= 9 * ws; ++j)
			ans[1][ts + j] = (ans[1][ts + j] + dp[ws][j]) % mod;
	}
	gen(p + 1, s + (n[p] ^ 48));
}
inline int calc(int v) {
	int ret = 0;
	while (v) ret += v % 10, v /= 10;
	return ret;
}
signed main() {
	ios::sync_with_stdio(0); dp[0][0] = 1;
	for (int i = 0; i != 1000; ++i)
		for (int j = 0; j <= 9 * i; ++j)
			for (int k = 0; k != 10; ++k)
				dp[i + 1][j + k] = (dp[i + 1][j + k] + dp[i][j]) % mod;
	for (cin >> t; t; t--) {
		cin >> n >> m >> k;
		/*
		1: 9000
		2: 35
		3: 11
		4: 9
		5: 9
		...
		*/
		m = min(m, 4ll);
		if (k > 9000) {
			cout << 0 << endl;
			continue;
		}
		memset(ans, 0, sizeof ans);
		gen(0, 0);
		for (int i = 2; i <= m; ++i)
			for (int j = 0; j <= 9000; ++j)
				ans[i][calc(j)] = (ans[i][calc(j)] + ans[i - 1][j]) % mod;
		cout << ans[m][k] << endl;
	}
}
