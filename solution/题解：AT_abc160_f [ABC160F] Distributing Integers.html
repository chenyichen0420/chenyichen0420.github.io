思路分析

首先，按照题目的描述，我们很容易知道他求的就是树上拓扑序的数量。同时，我们知道其公式 n!∏i=1nsizi\frac{n!}{\prod_{i=1}^nsiz_i}∏i=1n​sizi​n!​。

那么，这道题就是一道很明显的换根 DP 了。

具体怎么换呢？我们考虑两个点 a,ba,ba,b，他们之间有边连接。假设我们已经知道以 aaa 为根的答案 vav_ava​，分析二者的不同。

我们很容易发现，二者不同的地方有且仅有 sizasiz_asiza​ 和 sizbsiz_bsizb​。换根前，siza=nsiz_a=nsiza​=n。换根后，sizasiz_asiza​ 变为 n−sizbn-siz_bn−sizb​，sizbsiz_bsizb​ 变为 nnn。

这时候我们再次分析答案中的变化。可以发现，因为变化后的 sizbsiz_bsizb​ 就等于变化前的 sizasiz_asiza​，所以以 bbb 为根的答案 vb=va×sizbn−sizbv_b=v_a\times\frac{siz_b}{n-siz_b}vb​=va​×n−sizb​sizb​​。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
constexpr int mod = 1e9 + 7, szv = 1e6 + 5;
vector<int>son[szv];
int n, siz[szv], ans[szv], jc, ny[szv];
inline void dfsz(const int& p, const int& f) {
	siz[p] = 1;
	for (int sp : son[p])
		if (sp != f) 
			dfsz(sp, p), 
			siz[p] += siz[sp];
	ans[1] = ans[1] * ny[siz[p]] % mod;
}
inline void dfsa(const int& p, const int& f) {
	if (f) ans[p] = ans[f] * ny[n - siz[p]] % mod * siz[p] % mod;
	for (int sp : son[p]) if (sp != f) dfsa(sp, p);
}
inline int qpow(int a) {
	int tmp = 1, b = mod - 2;
	while (b) {
		if (b & 1) tmp *= a, tmp %= mod;
		a *= a; a %= mod; b >>= 1;
	}
	return tmp;
}
signed main() {
	ios::sync_with_stdio(0);
	cin >> n; jc = 1;
	for (int i = 1, l, r; i < n; ++i)
		cin >> l >> r,
		son[l].emplace_back(r),
		son[r].emplace_back(l);
	for (int i = 2; i <= n; ++i) jc = jc * i % mod;
	for (int i = n; i >= 0; i--) ny[i] = qpow(i);
	ans[1] = jc; dfsz(1, 0); dfsa(1, 0);
	for (int i = 1; i <= n; ++i) cout << ans[i] << endl;
	return 0;
}
