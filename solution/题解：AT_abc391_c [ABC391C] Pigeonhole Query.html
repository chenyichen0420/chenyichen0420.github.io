思路分析

数据范围太小了，直接用桶就可以玩。开两个数组，一个记录鸽子的位置，一个记录位置的鸽子数，动态更新就完了。

代码如下：

#include<bits/stdc++.h>
using namespace std;
int n,m,o,a,b,p[1000005],t[1000005],ans;
inline void move(int p,int v){
	if(--t[::p[p]]==1) ans--;
	if(t[::p[p]=v]++==1) ans++;
}
signed main(){
	ios::sync_with_stdio(0);
	cin>>n>>m; t[0]=n; ans=1;
	for(int i=1;i<=n;++i) move(i,i);
	for(int i=1;i<=m;++i)
		if(cin>>o,o&1) cin>>a>>b,move(a,b);
		else cout<<ans<<endl;
	return 0; 
}
