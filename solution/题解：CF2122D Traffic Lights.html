思路分析

因为至多在 1 单位时间的等待后，就可以扫到一个之前完全无法到达的点。换言之，至多经过 2×n 的时间后一定是全图可达的。因此不难，也一定能够构造出来一组解，满足消耗的时间不超过 2×n。

我们在综合一下数据范围，猜测时间复杂度是 O(n2) 的。

我们考虑一个动态规划，dpi,j​ 表示在时间 j 的时候来到 i 的最少等待时间。不难想到有两种可行的转移：

  1. dpi,j​→dpsoni,jmodsoncounti​+1​,j​。
  2. dpi,j​+1→dpi,j+1​。

发现时间复杂度真的是正确的！注意一下无效状态的标记就行了。代码如下：

#include<bits/stdc++.h>
using namespace std;
int t, n, m, dp[5005][10005]; vector<int>son[5005];
inline void tmin(int& l, const int r) { (l > r) && (l = r); }
signed main() {
	ios::sync_with_stdio(0);
	for (cin >> t; t; t--) {
		cin >> n >> m;
		for (int i = 1; i <= n; ++i) son[i].clear();
		for (int i = 1, l, r; i <= m; ++i)
			cin >> l >> r,
			son[l].emplace_back(r),
			son[r].emplace_back(l);
		for (int i = 1; i <= n; ++i)
			for (int j = 0; j <= n * 2; ++j)
				dp[i][j] = 1e9;
		dp[1][0] = 0;
		for (int j = 0; j <= n * 2; ++j) {
			if (dp[n][j] <= j) {
				cout << j << " " << dp[n][j] << endl;
				break;
			}
			for (int i = 1; i <= n; ++i)
				tmin(dp[son[i][j % son[i].size()]][j + 1], dp[i][j]),
				tmin(dp[i][j + 1], dp[i][j] + 1);
		}
	}
	return 0;
}
