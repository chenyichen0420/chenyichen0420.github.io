思路分析

正如题目背景所说，这道题在最开始出出来的时候就是奔着板子题去的，所以基本上没有什么好讲的。

我们考虑，如果是一棵树，那么这个问题是简单的，因为只有一条路径，查询出来的值就是答案。

但是现在是边仙人掌。我们考虑边仙人掌意味着什么：在经过一个环的时候，我可以选择从哪一边走过去。

这样的描述是不好优化的，我们换一个角度：我先随便选一个方向，然后如果觉得不优，可以通过异或上整个环的方式来选择另一边。

于是问题被转化成了两点之间有一个路径的基值，对于路径上的每一个环，你可以选择异或上他的权值，问最终权值的最小值，多次询问。

这是不困难的，我们直接建出圆方树，把环长挂到方点上，然后就变成了查询两点之间的异或值以及其路径上方点的元素所构成的线性基。

第一个是简单的，第二个怎么办呢？

我们会前缀线性基，于是我们直接把这个东西上树。具体来说，$p$ 节点继承 $fa_p$ 的线性基，然后时间戳标记为一个祖先严格大于后代的值就行了，比如说深度，欧拉序等都可以。

然后类似前缀线性基，我们可以 $O(\log)$ 的提取出某一段向上的链中的元素所构成的线性基。于是我们把查询的部分拆成两段，求出两个线性基然后 $\log^2$ 合并起来就行了。

显然瓶颈在于合并线性基，总时间复杂度是 $O(n\log V+k\log^2 V)$ 的。注意卡时间和空间的常数（没办法各种歪解太狂妄了）。参考代码如下：

CPP 
#include <bits/stdc++.h>
using namespace std;
struct IO {
#define mxsz (1 << 21)
    char buf[mxsz], *p1, *p2;
    char pbuf[mxsz], *pp;
    IO() : p1(buf), p2(buf), pp(pbuf) {}
    ~IO() { fwrite(pbuf, 1, pp - pbuf, stdout); }
    inline char gc() {
        if (p1 == p2) p2 = (p1 = buf) + fread(buf, 1, mxsz, stdin);
        return p1 == p2 ? ' ' : *p1++;
    }
    inline int read() {
        int r = 0; char c = gc();
        while (c < '0' || c > '9') c = gc();
        while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
        return r;
    }
    inline void push(const char& c) {
        if (pp - pbuf == mxsz) fwrite(pbuf, 1, mxsz, stdout), pp = pbuf;
        *pp++ = c;
    }
    inline void write(int x) {
        static char sta[22]; int top = 0;
        do sta[top++] = x % 10, x /= 10; while (x);
        while (top) push(sta[--top] ^ 48);
    }
    inline void write(int x, char opc) {
        write(x); push(opc);
    }
} io;
struct edge { int p, v, nt; } e1[2000005], e2[1000005];
int h1[500005], h2[1000005], ec1 = 1, ec2 = 1;
inline void adde1(int l, int r, int v) {
    e1[++ec1] = {r, v, h1[l]}; h1[l] = ec1;
    e1[++ec1] = {l, v, h1[r]}; h1[r] = ec1;
}
inline void adde2(int l, int r, int v) {
    e2[++ec2] = {r, v, h2[l]}; h2[l] = ec2;
}
int n, m, k, cnt, dfn[1000005], cln[1000005];
int dep[1000005], fa[1000005], sz[1000005], son[1000005], top[1000005];
int pfa[500005], v[500005], fe[500005], dis[1000005], vcn;
inline void tmin(int& l, int r) { if (l > r) l = r; }
inline void mkcir(int ep, int sp, int sv) {
    for (int p = sp; p != ep; p = pfa[p]) dis[p] = sv, sv ^= v[p];
    dis[ep] = cln[ep] = cln[++vcn] = sv;
    adde2(ep, vcn, 0);
    for (int p = sp; p != ep; p = pfa[p])
        cln[p] = sv, adde2(vcn, p, dis[p]);
}
inline void tarjan(int p, int id) {
    dfn[p] = dep[p] = ++cnt;
    for (int i = h1[p], sp; i; i = e1[i].nt)
        if (!dfn[sp = e1[i].p]) {
            pfa[sp] = p; v[sp] = e1[i].v; fe[sp] = i;
            tarjan(sp, i); tmin(dep[p], dep[sp]);
            if (dfn[p] < dep[sp]) adde2(p, sp, e1[i].v);
        } else if (i ^ id ^ 1) tmin(dep[p], dfn[sp]);
    for (int i = h1[p], sp; i; i = e1[i].nt)
        if (dfn[p] < dfn[sp = e1[i].p] && (fe[sp] ^ i))
            mkcir(p, sp, e1[i].v);
}
int *lb_v[1000005], *lb_p[1000005];
int pool_v[500005 * 30], pool_p[500005 * 30];
int pool_top = 0;
int zero_v[30] = {0}, zero_p[30] = {0};
inline void ins_basis(int cp, int lp, int cv) {
    lb_v[cp] = pool_v + pool_top;
    lb_p[cp] = pool_p + pool_top;
    pool_top += 30;
    memcpy(lb_v[cp], lb_v[lp], 30 * sizeof(int));
    memcpy(lb_p[cp], lb_p[lp], 30 * sizeof(int));
    if (!cv) return;
    int np = dfn[cp];
    int i = 31 - __builtin_clz(cv);
    while (true) {
        if (lb_v[cp][i]) {
            if (lb_p[cp][i] < np) {
                swap(lb_p[cp][i], np);
                swap(lb_v[cp][i], cv);
            }
            cv ^= lb_v[cp][i];
            if (!cv) return;
            i = 31 - __builtin_clz(cv);
        } else {
            lb_v[cp][i] = cv;
            lb_p[cp][i] = np;
            return;
        }
    }
}
void dfs1(int p) {
    sz[p] = 1;
    for (int i = h2[p]; i; i = e2[i].nt) {
        int sp = e2[i].p;
        fa[sp] = p;
        dep[sp] = dep[p] + 1;
        dis[sp] = dis[p] ^ e2[i].v;
        dfs1(sp);
        sz[p] += sz[sp];
        if (sz[sp] > sz[son[p]]) son[p] = sp;
    }
}
void dfs2(int p, int tp) {
    dfn[p] = ++cnt; top[p] = tp;
    if (p > n) ins_basis(p, fa[p], cln[p]);
    else {
        lb_v[p] = lb_v[fa[p]];
        lb_p[p] = lb_p[fa[p]];
    }
    if (son[p]) dfs2(son[p], tp);
    for (int i = h2[p]; i; i = e2[i].nt) {
        int sp = e2[i].p;
        if (sp != son[p]) dfs2(sp, sp);
    }
}
struct blin_base {
    int v[30];
    blin_base() { memset(v, 0, sizeof v); }
    inline void merge(int cv) {
        if (!cv) return;
        int i = 31 - __builtin_clz(cv);
        do {
            if (v[i]) {
                cv ^= v[i];
                if (!cv) return;
                i = 31 - __builtin_clz(cv);
            } else { v[i] = cv; return; }
        } while (true);
    }
    inline int qmin(int t) const {
        for (int i = 29; i >= 0; --i)
            if ((t ^ v[i]) < t) t ^= v[i];
        return t;
    }
};
inline void add_path(blin_base& bs, int la, int x) {
    int r = dfn[x], l = dfn[la];
    int *v_x = lb_v[x], *p_x = lb_p[x];
    for (int i = 0; i < 30; ++i) {
        int val = v_x[i];
        if (val && p_x[i] >= l) bs.merge(val);
    }
}
inline int lca(int u, int v) {
    while (top[u] != top[v]) {
        if (dep[top[u]] < dep[top[v]]) swap(u, v);
        u = fa[top[u]];
    }
    return dep[u] < dep[v] ? u : v;
}
int main() {
    ios::sync_with_stdio(0);
    lb_v[0] = zero_v; lb_p[0] = zero_p;
    vcn = n = io.read(); m = io.read(); k = io.read();
    for (int i = 1, l, r, v; i <= m; ++i) {
        l = io.read(); r = io.read(); v = io.read(); 
        adde1(l, r, v);
    }
    tarjan(1, 0);
    memset(dep, 0, (vcn + 1) * sizeof(int));
    memset(dis, 0, (vcn + 1) * sizeof(int));
    cnt = 0; dfs1(1);
    cnt = 0; dfs2(1, 1);
    for (int i = 1, l, r, la; i <= k; ++i) {
        l = io.read(); r = io.read(); la = lca(l, r);
        blin_base bs;
        if (la > n) bs.merge(cln[la]);
        add_path(bs, la, l);
        add_path(bs, la, r);
        io.write(bs.qmin(dis[l] ^ dis[r]), '\n');
    }
}
