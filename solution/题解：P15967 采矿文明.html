思路分析

首先，如果你学的比较广，那么你可能会想到 WQS 二分。这应该是一个正确的思路，不过配不上他黄的难度。

我们考虑枚举最后停在哪里。那么显然，贡献的最大值就是 min(m−∑j=1i−1​cj​,∑maxk of j=1i​aj​)。

显然，我们从前向后扫，用一个堆维护一下前 k 大就行了。代码如下：

#include<iostream>
#include<vector>
#include<queue>
#include<cstring>
#include<string>
#include<algorithm>
using namespace std;
#define int long long
constexpr int mod = 998244353;
int n, m, k, c[100005], a[100005], ans, sum;
priority_queue<int, vector<int>, greater<int>>pq;
signed main() {
	ios::sync_with_stdio(0);
	cin >> n >> m >> k;
	for (int i = 1;i != n;++i) cin >> c[i];
	for (int i = 1;i <= n;++i) cin >> a[i];
	for (int i = 1;i <= n;++i) {
		pq.emplace(a[i]); sum += a[i];
		if (pq.size() > k) sum -= pq.top(), pq.pop();
		ans = max(ans, min(m, sum));
		m -= c[i];
	}
	cout << ans << endl;
}
