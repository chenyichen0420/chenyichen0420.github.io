思路分析

很重要的有三点性质：

  1. 调和级数 O(nlog⁡n)O(n\log n)O(nlogn)；
  2. 如果一个数被删，就不可能去删掉别的更多的数；
  3. 一个质数只能被自己删掉。

所以说，删除的复杂度是严格小于 Vlog⁡VV\log VVlogV，大概有 150\frac{1}{50}501​ 常数。你可以暴力的修改他，只要他是 O(1)O(1)O(1) 的。查询也不需要特别快，O(log⁡n)O(\log n)O(logn) 或者 O(n)O(\sqrt n)O(n​) 都能接受。

333 性质其实限制了答案最大为第 2×1052\times 10^52×105 个质数。

所以，手打一个分块不就行了？代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
int t, n, m, sz = ceil(sqrt(2750160)); 
bool ext[2750160]; int cnt[1660];
#define blk(i) ((i - 1) / sz + 1)
inline void del(int v) {
	for (int i = v;i < 2750160;i += v)
		if (ext[i]) ext[i] = 0, cnt[blk(i)]--;
}
inline int que(int v) {
	int bk = 0;
	while (v > cnt[bk + 1])
		bk++, v -= cnt[bk];
	bk = sz * bk + 1;
	while (1)
		if (ext[bk] && !--v)
			return bk;
		else bk++;
}
signed main() {
	ios::sync_with_stdio(0);
	memset(ext, 1, sizeof ext);
	for (int i = 1;i != 2750160;++i)
		cnt[blk(i)]++;
	for (cin >> t;t;t--) {
		cin >> n >> m;
		if (n < 2750160 && ext[n]) del(n);
		cout << que(m) << endl;
	}
}
