思路分析

我们稍微发扬一下人类智慧，每一步都贪心的选择使得模长最短的那个方向。

这样子做是有一定的正确率的，但是不能保证一定通过，因为有的向量可能偏移量过大。

于是我们选择随机的打乱一下，不停的跑，直到发现一组可行解。

这非常的快，即使在 n=105 的数据范围下也可以在大约 0.1s 内创过去。

代码如下：

#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
#define int long long
struct nope {
	int x, y, i;
}p[100005];
int n, ans[100005], px, py;
inline int dis(int x, int y) {
	return x * x + y * y;
}
signed main() {
	ios::sync_with_stdio(0);
	cin >> n; srand((unsigned)time(0));
	for (int i = 1; i <= n; ++i) cin >> p[i].x >> p[i].y, p[i].i = i;
	do {
		random_shuffle(p + 1, p + n + 1); px = 0; py = 0;
		for (int i = 1; i <= n; ++i) {
			if (dis(p[i].x + px, p[i].y + py) < dis(px - p[i].x, py - p[i].y))
				ans[p[i].i] = 1, px += p[i].x, py += p[i].y;
			else ans[p[i].i] = -1, px -= p[i].x, py -= p[i].y;
		}
	} while (sqrt(dis(px, py)) > 1.5e6);
	for (int i = 1; i <= n; ++i) cout << ans[i] << " ";
}
