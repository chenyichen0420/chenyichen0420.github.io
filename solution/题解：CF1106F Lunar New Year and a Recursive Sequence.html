思路分析

一眼就是原根什么的，题面的提示性一定程度上给这道题降低了难度。

最大的提示就是那个幂的乘积，这确实太过于显眼了。我们考虑将整个式子重写成这样：

$g_i=\sum_{j=1}^kb_jg_{i-j}$，显然 $g_i=\log f_i$。

我们知道 $\log 1=0$，再求出来 $\log m$ 就可以列方程解 $g_k$ 了。具体来说，我们将所有的数值用 $g_k$ 的几倍来表示，比如 $g_1=0\times g_k$，$g_k=1\times g_k$。

随后我们直接使用矩阵快速幂就可以列出等式 $v\times g_k=\log m$。现在我们想办法让整个式子有意义。

显然 $998244353$ 有一个原根是 $3$，于是我们让上面的所有 $\log$ 都以 $3$ 为底进行计算即可。

于是，唯一一步可能出现无解的地方就是 $v\times g_k=\log m$ 无解，这部分当中 $g_k$ 显然可以通过 EXGCD 直接算即可。

最后 $f_k=3^{g_k}$ 还原回去就好了。参考代码如下：

CPP 
#include <bits/stdc++.h>
using namespace std;
#define int long long
constexpr int mod=998244353,g=3;
int k,n,m,b[105],lgm;
unordered_map<int,int>mp;
inline int calg(int v){
	mp.clear();
	int ret=-1;
	int cv=1,sv=sqrt(mod);
	for(int i=0;i!=sv;++i)
		mp[cv*v%mod]=i,cv=cv*g%mod;
	if(m==1) ret=0;
	else{
		int cur=1;
		for(int i=1;i<=sv+2;++i){
			cur=cur*cv%mod;
			if(mp.count(cur)){
				ret=i*sv-mp[cur];
				break;
			}
		}
	}
	if(ret==-1) exit(puts("-1")&0);
	return ret;
}
struct mat{
	int v[101][101];
	inline mat operator*(const mat&r)const{
		static mat ret;
		memset(ret.v,0,sizeof ret.v);
		for(int i=0;i!=k;++i)
			for(int j=0;j!=k;++j)
				for(int l=0;l!=k;++l)
					ret.v[i][l]=(ret.v[i][l]+v[i][j]*r.v[j][l])%(mod-1);
		return ret;
	}
}ans,tmp;
inline void qpow(int k){
	for(;k;k>>=1,tmp=tmp*tmp)
		if(k&1) ans=ans*tmp;
}
inline int qpow(int a,int b){
	int ret=1;
	for(;b;b>>=1,a=a*a%mod)	
		(b&1)&&(ret=ret*a%mod);
	return ret;
}
inline void exgcd(int a,int b,int &x,int &y){
	if(b==0) return x=1,void(y=0);
	exgcd(b,a%b,y,x); y-=a/b*x;
}
signed main() {
	ios::sync_with_stdio(0); cin>>k;
	/*
	g_i=log f_i 那么显然有:
	g_i=\sum_{j=1}^k b_j f_{i-j}
	有 g_{1}^{k-1}=0
	这里的 log 我们可以定义为 log3，因为 3 是 998244353 的原根
	此题可解！
	*/
	for(int i=1;i<=k;++i) cin>>b[i];
	cin>>n>>m; lgm=calg(m);
	if(lgm==0) exit(puts("1")&0);
	ans.v[0][k-1]=1;
	for(int i=0;i!=k-1;++i)
		tmp.v[i+1][i]=1;
	for(int i=0;i!=k;++i)
		tmp.v[i][k-1]=b[k-i];
	qpow(n-k);
	int a=ans.v[0][k-1];
	int b=lgm;
	int gc=__gcd(a,b);
	a/=gc; b/=gc;
	if(__gcd(a,mod-1)!=1) exit(puts("-1")&0);
	cerr<<"3^("<<b<<"/"<<a<<")\n";
	int inv,tmp;
	exgcd(a,mod-1,inv,tmp);
	inv=(inv%(mod-1)+mod-1)%(mod-1);
	b=b*inv%(mod-1);
	cout<<qpow(g,b)<<endl;
}
