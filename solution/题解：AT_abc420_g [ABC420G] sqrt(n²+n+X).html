算法流程：

  1. 计算 D=4x−1D = 4x - 1D=4x−1。
  2. 枚举所有包含负数的满足 a×b=Da\times b=Da×b=D 的整数对 (a,b)(a,b)(a,b)。
  3. 对于每个因子对 (a,b)(a,b)(a,b)：
    • 检查 a+ba+ba+b 是否能被 4 整除。
    • 计算 k=(a+b)/4,m=(b−a)/2k=(a+b)/4,m=(b-a)/2k=(a+b)/4,m=(b−a)/2。
    • 检查 mmm 是否为奇数。

  4. 去重排序后输出。

#include<bits/stdc++.h>
using namespace std;
#define int long long
int n, d, k, m, tn; vector<int>v;
signed main() {
	ios::sync_with_stdio(0);
	cin >> n; tn = abs(n = 4 * n - 1);
	for (int i = 1; i * i <= tn; ++i)
		if (tn % i == 0) {
			for (int k = 1; k <= 1; ++k) {
				int a = i, b = n / i;
				if ((a + b) % 4) break;
				int m = (b - a) / 2;
				if (m % 2 == 0) break;
				v.emplace_back((-1 + m) / 2);
				v.emplace_back((-1 - m) / 2);
			}
			for (int k = 1; k <= 1; ++k) {
				int a = -i, b = -n / i;
				if ((a + b) % 4) break;
				int m = (b - a) / 2;
				if (m % 2 == 0) break;
				v.emplace_back((-1 + m) / 2);
				v.emplace_back((-1 - m) / 2);
			}
		}
	sort(v.begin(), v.end());
	v.erase(unique(v.begin(), v.end()), v.end());
	cout << v.size() << endl;
	for (int i : v) cout << i << ' ';
}
