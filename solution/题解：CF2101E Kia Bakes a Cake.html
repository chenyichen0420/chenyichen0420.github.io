思路分析

下面的参考代码常数非常劣，不保证可以通过。

我们很容易想到一个 $dp_{i,j,k}$ 表示 $i$ 到 $j$ 的长度为 $k$ 的链的最后一条边的最小长度。

发现起点信息多余，于是我们倒过来考虑这个过程，$dp_{i,j}$ 表示到 $i$ 的长度为 $j$ 的链的最后一条边的最大长度，发现直接做单轮转移时间复杂度是 $O(n^2)$ 的，无法通过。

首先不难发现总的轮数是 $O(\log n)$ 的，因为每一轮要求长度砍半，而树上的距离不超过 $n-1$。

我们注意到这是一个树上路径类的转移问题，于是我们使用点分治优化这个过程，具体来说，我们对于重心 $p$，要求转移路径经过 $p$，于是之前一个 $dp_{i,j}$ 经过 $p$ 之后可以覆盖到和 $p$ 的距离不超过 $\frac{dp_{i,j}}{2}-dis(i,p)$ 的所有点 $j$，贡献是 $dis(i,p)+dis(p,j)$，不过自己子树的不行。

于是我们直接使用树状数组大力优化这个过程即可，显然时间复杂度是 $\log$ 轮 $\log$ 层的树状数组 $\log$，也就是 $O(n\log^3 n)$。参考代码如下：

CPP 
#include<bits/stdc++.h>
#include<unistd.h>
using namespace std;
#pragma GCC optimize(3,"Ofast")
#pragma GCC optimize("unroll-loops")
#pragma GCC optimize("omit-frame-pointer")
#pragma GCC optimize("no-stack-protector")
int t,n,v[70005],ans[70005],vn[70005],lim; char c[70005];
vector<int>son[70005]; bool vis[70005];
inline __attribute__((always_inline)) void tmax(int&l,const int r){(l<r)&&(l=r);}
struct tree_array{
	int c[70005];
	inline void ins(int p,int v){
		for(p=lim-p;p<=lim;p+=p&-p) tmax(c[p],v);
	}
	inline void rst(int p){
		for(p=lim-p;p<=lim;p+=p&-p) c[p]=-1e8;
	}
	inline int que(int p){
		int ret=-1e8;
		for(p=lim-p;p;p&=p-1) tmax(ret,c[p]);
		return ret;
	} 
}ta;
int sz[70005],hs,pup; vector<int>hss;
inline void dfs(int p,int f,int ts){
	sz[p]=1; int ms=0;
	for(int sp:son[p]) if(sp!=f&&!vis[sp])
		dfs(sp,p,ts),sz[p]+=sz[sp],
		(sz[sp]>ms)&&(ms=sz[sp]);
	tmax(ms,ts-sz[p]);
	if(ms<=ts/2) hs=p;
	tmax(lim,v[p]/2+1);
}
inline void ins(int p,int f,int d){
	if(v[p]/2>=d) ta.ins(v[p]/2-d,d),tmax(pup,d);
	for(int sp:son[p])
		if(!vis[sp]&&sp!=f) ins(sp,p,d+1);
}
inline void que(int p,int f,int d){
	if(d<lim)tmax(vn[p],ta.que(d)+d);
	for(int sp:son[p])
		if(!vis[sp]&&sp!=f) que(sp,p,d+1);
}
inline void rst(int p,int f,int d){
	if(v[p]/2>=d) ta.rst(v[p]/2-d);
	for(int sp:son[p])
		if(!vis[sp]&&sp!=f) rst(sp,p,d+1);
}
inline void solve(int p,int ts){
	dfs(p,0,ts); hss.emplace_back(hs);
	vis[p=hs]=1;
	lim=0; dfs(p,0,ts); pup=0;
	ta.ins(v[p]/2,0);
	for(int sp:son[p]) if(!vis[sp])
		que(sp,0,1),ins(sp,0,1);
	for(int sp:son[p]) if(!vis[sp])
		rst(sp,0,1);
	ta.rst(v[p]/2);
	reverse(son[p].begin(),son[p].end());
	ta.ins(v[p]/2,0);
	for(int sp:son[p]) if(!vis[sp])
		que(sp,0,1),ins(sp,0,1);
	for(int sp:son[p]) if(!vis[sp])
		rst(sp,0,1);
	ta.rst(v[p]/2);
	tmax(vn[p],pup);
	reverse(son[p].begin(),son[p].end());
	for(int sp:son[p]) if(!vis[sp])
		solve(sp,sz[sp]);
}
inline void fsolve(int p,int ts,int& id){
	hs=hss[id]; vis[p=hs]=1;
	lim=0; dfs(p,0,ts); pup=0;
	ta.ins(v[p]/2,0);
	for(int sp:son[p]) if(!vis[sp])
		que(sp,0,1),ins(sp,0,1);
	for(int sp:son[p]) if(!vis[sp])
		rst(sp,0,1);
	ta.rst(v[p]/2);
	reverse(son[p].begin(),son[p].end());
	ta.ins(v[p]/2,0);
	for(int sp:son[p]) if(!vis[sp])
		que(sp,0,1),ins(sp,0,1);
	for(int sp:son[p]) if(!vis[sp])
		rst(sp,0,1);
	ta.rst(v[p]/2);
	tmax(vn[p],pup);
	reverse(son[p].begin(),son[p].end());
	for(int sp:son[p]) if(!vis[sp])
		++id,fsolve(sp,sz[sp],id);
}
signed main(){
	ios::sync_with_stdio(0);
	cin.tie(0); cout.tie(0);
	for(cin>>t;t;t--){
		for(int i=1;i<=n;++i) son[i].clear();
		cin>>n>>c; memmove(c+1,c,n); bool hs=0;
		for(int i=1,l,r;i!=n;++i)
			cin>>l>>r,
			son[l].emplace_back(r),
			son[r].emplace_back(l);
		for(int i=1;i<=n;++i)
			if(c[i]^48) v[i]=2*(n-1),ans[i]=0,hs=1;
			else ans[i]=-1,v[i]=0;
		memset(ta.c,0xcf,sizeof ta.c); hss.clear();
		for(int i=1;hs;++i){
			memset(vis,0,sizeof vis);
			memset(vn,0,sizeof vn);
			hs=0; int cid=0;
			if(hss.empty())solve(1,n);
			else fsolve(1,n,cid);
			for(int j=1;j<=n;++j)
				if(c[j]=='0') continue;
				else if(v[j]=vn[j]) hs=1;
				else if(!ans[j]) ans[j]=i;
		}
		for(int i=1;i<=n;++i) cout<<ans[i]<<" "; cout<<endl;
	}
}
