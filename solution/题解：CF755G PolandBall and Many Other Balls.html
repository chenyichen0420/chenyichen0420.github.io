思路分析

不动脑子（其实就是不会推通项）的解法。

考虑 $dp_{i,j}$ 表示到了 $i$，共计产生 $j$ 组的方案数。显然有转移方程 $dp_{i,j}=dp_{i-1,j}+dp_{i-1,j-1}+dp_{i-2,j-1}$。

于是我们把这个东西写成普通生成函数的形式：$f_i=(1+x)f_{i+1}+xf_{i-2}$。

显然一项只和前两项相关。于是我们直接把矩阵乘法搬上来，尝试使用矩阵快速幂解决这个问题。

具体来说，直接把转移系数填到 $2\times2$ 的多项式矩形里面，然后进行多项式矩阵的快速幂即可。

显然时间复杂度是 $O(k\log k\log n)$ 带有非常大的常数的，但是不进行任何卡常就能过。参考代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
constexpr int mod=998244353,g=3;
int n,m,cp[1ll<<17],a[1ll<<17],b[1ll<<17];
inline int qpow(int a,int b){
	int ret=1;
	for(;b;b>>=1,a=a*a%mod)
		(b&1)&&(ret=ret*a%mod);
	return ret;
}
inline void fft(int*v,int lim,int l){
	for(int i=1;i!=lim;++i)
		cp[i]=(cp[i>>1]>>1)|((i&1)<<(l-1));
	for(int i=0;i!=lim;++i)
		if(cp[i]>i) swap(v[i],v[cp[i]]);
	for(int i=1;i!=lim;i<<=1){
		int tv=qpow(g,(mod-1)/i/2);
		for(int j=0;j!=lim;j+=i<<1){
			int vt=1,t;
			for(int k=0;k!=i;++k)
				t=v[j+k+i]*vt%mod,
				v[j+k+i]=(v[j+k]-t+mod)%mod,
				v[j+k]=(v[j+k]+t)%mod,
				vt=vt*tv%mod;
		}
	}
}
using poly=vector<int>;
inline void add(poly&l,const poly&r){
	l.resize(max(l.size(),r.size()));
	for(int i=0;i!=l.size();++i)
		(l[i]+=r[i])>=mod&&(l[i]-=mod);
}
inline poly time(const poly&l,const poly&r){
	int lim=1,ln=0,s=l.size()+r.size();
	while(lim<s) lim<<=1,ln++;
	int ny=qpow(lim,mod-2);
	memset(a,0,lim*sizeof(int));
	memset(b,0,lim*sizeof(int));
	memcpy(a,l.data(),l.size()*sizeof(int));
	memcpy(b,r.data(),r.size()*sizeof(int));
	fft(a,lim,ln); fft(b,lim,ln);
	for(int i=0;i!=lim;++i) a[i]=a[i]*b[i]%mod;
	fft(a,lim,ln); reverse(a+1,a+lim); 
	for(int i=0;i!=lim;++i) a[i]=a[i]*ny%mod;
	static poly ret; ret.resize(s-1);
	memcpy(ret.data(),a,sizeof(int)*(s-1));
	if(ret.size()>m+1) ret.resize(m+1);
	return ret;
}
struct mat{
	poly v[2][2];
	inline mat operator*(const mat&r)const{
		mat ret;
		for(int i=0;i!=2;++i)
			for(int j=0;j!=2;++j)
				for(int k=0;k!=2;++k)
					add(ret.v[i][k],time(v[i][j],r.v[j][k]));
					//可以优化，这里的乘法有大量重复，可以不用 NTT 这么多次
					//暂时先放着，我相信这个常数应该还好
		return ret;
	}
}ans,tmp;
signed main() {
	ios::sync_with_stdio(0); cin>>n>>m;
	/*
	我们考虑 dp_{i,j} 表示到了 i，共计产生了 j 组的方案数
	于是有转移方程：
	dp_{i,j}=dp_{i-1,j}+dp_{i-1,j-1}+dp_{i-2,j-1}
	组合意义是显然的
	我们考虑把这个东西写成生成函数的形式：
	f_i=(1+x)f_{i-1}+xf_{i-2}
	于是考虑把这个东西写成矩阵乘法的形式：
	[f_{i-2},f_{i-1}] *	[0,x  ]	= [f_{i-1},f_i]
						[1,x+1]
	初始状态：f_0=1, f_1=1+x
	或者更加简单的，f_{-1}=0,f_0=1
	我们要求的直接变成了[0,x  ]的n次方
						[1,x+1]
	*/
	ans.v[0][0]=ans.v[1][1]={1};
	ans.v[0][1]=ans.v[1][0]={0};
	tmp.v[0][0]={0}; tmp.v[1][0]={1};
	tmp.v[0][1]={0,1}; tmp.v[1][1]={1,1};
	for(;n;n>>=1,tmp=tmp*tmp)
		if(n&1) ans=ans*tmp;
	for(int i=1;i!=ans.v[1][1].size();++i)
		cout<<ans.v[1][1][i]<<" ";
	for(int i=ans.v[1][1].size();i<=m;++i)
		cout<<"0 ";
	cout<<endl;
}
