思路分析

数据范围足够小，可以枚举这两个骰子。

枚举之后我们可以统计出两个骰子中每一个数出现的次数，然后枚举其中一个骰子上出现的数，相同的概率是这个数在两个骰子中出现次数之积除以两个骰子的面数之积。

求个和就是二者数字一样的概率。代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
#define double long double
int n; vector<int>vt[105]; double ans;
map<int, int>cnt1, cnt2;
inline double get(int l, int r) {
	double get = 0;
	cnt1.clear(); cnt2.clear();
	for (int v : vt[l]) cnt1[v]++;
	for (int v : vt[r]) cnt2[v]++;
	for (auto v : cnt1)
		get += (double)(v.second) * 1.0 / vt[l].size() * cnt2[v.first] * 1.0 / vt[r].size();
	return get;
}
signed main() {
	ios::sync_with_stdio(0);
	cin >> n;
	for (int i = 1, a, v; i <= n; ++i) {
		for (cin >> a; a; a--) cin >> v, vt[i].emplace_back(v);
		sort(vt[i].begin(), vt[i].end());
	}
	for (int i = 1; i <= n; ++i)
		for (int j = i + 1; j <= n; ++j)
			ans = max(ans, get(i, j));
	printf("%.15Lf\n", ans);
}
