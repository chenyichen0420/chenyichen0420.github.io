题目大意

给你一张原图 G′=(V,E1)G^\prime=(V,E_1)G′=(V,E1​)，再给你一些边 E2E_2E2​，其中 E1E_1E1​ 的边边权确定，E2E_2E2​ 中的边的边权由你自己定。这时我们令 E=E1∪E2E=E_1\cup E_2E=E1​∪E2​，则 EEE 中每一个边的起始点均已确定。问你，如果在 G=(V,E)G=(V,E)G=(V,E) 中生成出一个指定的最小生成树（毕竟最小生成树可以有多个），那么这棵树上属于 E2E_2E2​ 的边的边权之和最大为多少。

思路分析

当你对一道题目无从下手时，我们就要观察数据范围。观察到： k≤20k\le20k≤20。这是一个突破口，这么小的数字多半与 2k2^k2k 挂钩。

顺势出现一个思路：枚举每一条边有没有出现在最小生成树中。虽然复杂度不对（等会儿分析），但是讲一下：枚举 2k2^k2k 种情况，对于每一种，先将所有已经选择的边加入最小生成树中（特判此时已经有环的情况，直接舍弃），然后对整张图跑一遍最小生成树。最后用上最小生成树的性质对被选择的边进行约束，即可求出答案。

性质：最小生成树再加上任意一条边都会形成一个环。假定这条边的边权为 aaa，那么所有环上的边的边权都小于等于 aaa。

分析一下复杂度：最外层 2k2^k2k 枚举，中间最小生成树，如果先排好了序的话每次是 mmm，总的时间复杂度大于 O(2km)O(2^km)O(2km)，大概是 101110^{11}1011，显然不可行。

考虑优化。首先给出一个结论：当我们先将 E2E_2E2​ 所有的边加入到最小生成树中（有环就不管，反正对 kruskal 没有影响），再跑最小生成树的话，加的那些 E1E_1E1​ 的边一定在所有情况下的最小生成树中。

证明一下：我们先将点分为两类：被选中的 E1E_1E1​ 连同的点 V1V_1V1​ 和其余的点 V2V_2V2​。在 E2E_2E2​ 的限制下，V2V_2V2​ 中的点必然会形成若干个连通块。

那么显然 E2E_2E2​ 在 V2V_2V2​ 中产生的限制不会影响到 V1V_1V1​，也就是说在 V1V_1V1​ 当中 E1E_1E1​ 所形成的最小生成树必然是最优的。

其次，因为 E2E_2E2​ 在 V2V_2V2​ 中的限制，所有从 V2V_2V2​ 的点连向 V2V_2V2​ 的点且不连通两个连通块的 E1E_1E1​ 中的边都会被舍弃，但是这些边当且仅当 E2E_2E2​ 的限制改变时才可能被加入最小生成树，所以不一定出现在所有情况下的最小生成树中。

最后是从 V1V_1V1​ 的点连向 V2V_2V2​ 的点或连通两个连通块的 V2V_2V2​ 的点的 E1E_1E1​ 中的边。这些边以最优的方式联通了这些东西，那么在 E2E_2E2​ 的限制变少的情况下，这些东西也不会变得更劣。毕竟 E2E_2E2​ 的限制改变只会影响连通块内的内容。

当然了，以上的证明只是本人的感性理解的证明，并不严谨，不过对于绝大部分的情况都是没有问题的。

这时候，我们可以干一件事：把所有被选中的 E1E_1E1​ 连通起来的点缩成一个大点。因为这些边一定要选，而且不包含 E2E_2E2​ 中的边，所以不会对答案造成任何影响。

此时，图中就只会剩下 k+1k+1k+1 个点。

最后，我们再用开始的思路进行求解就行了。因为只剩下了 k+1k+1k+1 个点，因此我们不需要对最小生成树的限制与求解这一步进行任何优化，直接使用 k2k^2k2 的暴力更新就行了。

注意数据范围，不开 long long 见祖宗。

总复杂度：O(mlog⁡m+2kk2)O(m\log m+2^kk^2)O(mlogm+2kk2)。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
inline void tmin(int& l, const int& r) {
	(l > r) && (l = r);
}
inline void tmax(int& l, const int& r) {
	(l < r) && (l = r);
}
struct edge {
	int l, r, v;
	edge(int li = 0, int ri = 0, int vi = 0) :
		l(li), v(vi), r(ri) {}
}tmp; vector<edge>e, nee, ne; vector<int>che[30];
class MSS {
	int fa[100005];
public:
	int find(const int& p) {
		return (fa[p] != p ? fa[p] = find(fa[p]) : p);
	}
	inline void set(const int& v) {
		for (int i = 1; i <= v + 1; ++i) fa[i] = i;
	}
	inline bool merge(const int& l, const int& r) {
		return (find(l) != find(r) ? fa[find(l)] = find(r), 1 : 0);
	}
	inline bool isit(const int& v) const{
		return fa[v] == v;
	}
}mstt, mss;
int n, m, k, a[100005], blk[100005], vl[30];
int ans, cnt, dep[30], fa[30], siz[30], mv[30];
inline void newmap() {
	for (int i = 1; i <= n; ++i)
		if (mstt.isit(i)) blk[i] = ++cnt;
	for (int i = 1; i <= n; ++i)
		blk[i] = blk[mstt.find(i)], vl[blk[i]] += a[i];
	for (int i = 0; i != k; ++i)
		nee[i] = edge(blk[nee[i].l], blk[nee[i].r], 0);
	for (int i = 0; i != m; ++i)
		if (mstt.merge(e[i].l, e[i].r))
			ne.emplace_back(edge(blk[e[i].l], blk[e[i].r], e[i].v));
}
inline void dfs(const int& p, const int& f) {
	dep[p] = dep[f] + 1; fa[p] = f; siz[p] = vl[p];
	for (int i = 0; i != che[p].size(); ++i)
		if (che[p][i] != f) dfs(che[p][i], p), siz[p] += siz[che[p][i]];
}
inline void cldfs() {
	memset(dep, 0, sizeof dep); memset(fa, 0, sizeof fa);
	memset(siz, 0, sizeof siz); dfs(blk[1], 0);
}
inline void chg(int l, int r, const int& v) {
	while (l != r) {
		if (dep[l] > dep[r]) swap(l, r);
		tmin(mv[r], v); r = fa[r];
	}
}
signed main() {
	ios::sync_with_stdio(0);
	cin >> n >> m >> k;
	for (int i = 1, l, r, v; i <= m; ++i)
		cin >> l >> r >> v,
		e.emplace_back(edge(l, r, v));
	mss.set(n); mstt.set(n);
	for (int i = 1, l, r; i <= k; ++i)
		cin >> l >> r, mss.merge(l, r),
		nee.emplace_back(edge(l, r, 0));
	for (int i = 1; i <= n; ++i) cin >> a[i];
	sort(e.begin(), e.end(), [&](const edge& l, const edge& r) {
		return l.v < r.v;
		});
	for (int i = 0; i != m; ++i)
		if (mss.merge(e[i].l, e[i].r))
			mstt.merge(e[i].l, e[i].r);
	newmap();
	for (int chs = 0, tmp; chs != (1ll << k); ++chs) {
		mss.set(cnt); bool ctn = 0; tmp = 0;
		for (int i = 1; i <= cnt; ++i) che[i].clear();
		for (int i = 0; i != k; ++i)
			if (chs & (1ll << i))
				if (mss.merge(nee[i].l, nee[i].r))
					che[nee[i].l].emplace_back(nee[i].r),
					che[nee[i].r].emplace_back(nee[i].l);
				else { ctn = 1; break; }
		if (ctn) continue;
		for (int i = 0; i != ne.size(); ++i)
			if (mss.merge(ne[i].l, ne[i].r))
				che[ne[i].l].emplace_back(ne[i].r),
				che[ne[i].r].emplace_back(ne[i].l);
		cldfs(); memset(mv, 0x3f, sizeof mv);
		for (int i = 0; i != ne.size(); ++i)
			chg(ne[i].l, ne[i].r, ne[i].v);
		for (int i = 0; i != k; ++i)
			if (chs & (1ll << i))
				tmp += siz[(dep[nee[i].l] > dep[nee[i].r] ? nee[i].l : nee[i].r)] *
				mv[(dep[nee[i].l] > dep[nee[i].r] ? nee[i].l : nee[i].r)];
		tmax(ans, tmp);
	}
	cout << ans << endl;
	return 0;
}
