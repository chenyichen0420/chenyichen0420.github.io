思路分析

到不了绿吧？虽然绿以下众生平等。

说白了就是问你每一次给高度在 [bi−1+1,ai][b_{i-1}+1,a_i][bi−1​+1,ai​] 之间的桥的权值加一，问你最后有多少个桥的权值不小于 kkk。

首先很自然的想到按照桥的高度排个序，然后就是二分范围，做一个区间加。

因为我们只会在最后批量加，所以我们将区间加转化为差分数组单点修改，最后做一次前缀和。

代码很短的！如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
int n, m, k, t, h[100005], v[100005], lp, ans;
signed main() {
	ios::sync_with_stdio(0);
	while (cin >> n >> m >> k) {
		memset(v, 0, sizeof v); ans = 0;
		for (int i = 1; i <= n; ++i) cin >> h[i];
		sort(h + 1, h + n + 1); lp = 1;
		for (int i = 1, a, b; i <= m; ++i)
			cin >> a >> b,
			a = upper_bound(h + 1, h + n + 1, a) - h,
			b = upper_bound(h + 1, h + n + 1, b) - h,
			v[lp]++, v[a]--, lp = b;
		for (int i = 1; i <= n; ++i)
			ans += (v[i] += v[i - 1]) >= k;
		cout << "Case " << ++t << ": " << ans << endl;
	}
}
