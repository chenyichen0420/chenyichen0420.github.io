我们举题面中的矩阵的每一个数的倍率进行分析：

[123456]=[121212]+[002244]=[121212]+2×[001122]\begin{bmatrix} 1 & 2 \\ 3 & 4 \\ 5 & 6 \end{bmatrix} = \begin{bmatrix} 1 & 2 \\ 1 & 2 \\ 1 & 2 \end{bmatrix} + \begin{bmatrix} 0 & 0 \\ 2 & 2 \\ 4 & 4 \end{bmatrix} = \begin{bmatrix} 1 & 2 \\ 1 & 2 \\ 1 & 2 \end{bmatrix} + 2\times\begin{bmatrix} 0 & 0 \\ 1 & 1 \\ 2 & 2 \end{bmatrix}​135​246​​=​111​222​​+​024​024​​=​111​222​​+2×​012​012​​

因此，我们需要能够快速求出 ∑i=x0x1∑j=y0y1(i−x0+1)×ai,j\sum_{i=x_0}^{x_1}\sum_{j=y_0}^{y_1}(i-x_0+1)\times a_{i,j}∑i=x0​x1​​∑j=y0​y1​​(i−x0​+1)×ai,j​ 和 ∑i=x0x1∑j=y0y1(j−y0)×ai,j\sum_{i=x_0}^{x_1}\sum_{j=y_0}^{y_1}(j-y_0)\times a_{i,j}∑i=x0​x1​​∑j=y0​y1​​(j−y0​)×ai,j​。

显然上述两个东西可以使用三个二维前缀和数组 Ax,y=∑i=1x∑j=1yai,j,Bx,y=∑i=1x∑j=1yi×ai,j,Cx,y=∑i=1x∑j=1yj×ai,jA_{x,y}=\sum_{i=1}^x\sum_{j=1}^ya_{i,j},B_{x,y}=\sum_{i=1}^x\sum_{j=1}^yi\times a_{i,j},C_{x,y}=\sum_{i=1}^x\sum_{j=1}^yj\times a_{i,j}Ax,y​=∑i=1x​∑j=1y​ai,j​,Bx,y​=∑i=1x​∑j=1y​i×ai,j​,Cx,y​=∑i=1x​∑j=1y​j×ai,j​ 共同维护出来。

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
struct IO {
#define mxsz (1 << 21)
	char buf[mxsz], * p1, * p2;
	char pbuf[mxsz], * pp; bool acc[128];
	IO() : p1(buf), p2(buf), pp(pbuf) {}
	~IO() { fwrite(pbuf, 1, pp - pbuf, stdout); }
	inline char gc() {
		if (p1 == p2) p2 = (p1 = buf) + fread(buf, 1, mxsz, stdin);
		return p1 == p2 ? ' ' : *p1++;
	}
	inline void setacc(const char* c) { while (*c) acc[*c++] = 1; }
	inline char getc() { char c; while (!acc[c = gc()]); return c; }
#ifndef sipt
	inline int read() {
		int r = 0; char c = gc(); while (c < '0' || c>'9') c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return r;
	}
#else
	inline int read() {
		int r = 0; char c = gc(); bool rev = 0;
		while (c < '0' || c>'9') rev |= (c == '-'), c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return rev ? ~r + 1 : r;
	}
#endif
	inline void push(const char& c) {
		if (pp - pbuf == mxsz) fwrite(pbuf, 1, mxsz, stdout), pp = pbuf;
		*pp++ = c;
	}
	inline void write(int x) {
		static char sta[22]; int top = 0;
		do sta[top++] = x % 10, x /= 10; while (x);
		while (top) push(sta[--top] ^ 48);
	}
	inline void write(int x, char opc) {
#ifdef sopt
		if (x < 0) push('-'), x = ~x + 1;
#endif
		write(x), push(opc);
	}
	inline void puts(const char* c) { while (*c) push(*c++); }
} io;
int t, n, m, a[2005][2005], s1[2005][2005], s2[2005][2005], s3[2005][2005];
inline int get(int v[2005][2005], int a, int b, int c, int d) {
	return v[c][d] - v[a][d] - v[c][b] + v[a][b];
}
signed main() {
	ios::sync_with_stdio(0);
	for (t = io.read(); t; t--) {
		n = io.read(); m = io.read();
		for (int i = 1; i <= n; ++i)
			for (int j = 1; j <= n; ++j)
				a[i][j] = io.read();
		for (int i = 1; i <= n; ++i)
			for (int j = 1; j <= n; ++j)
				s1[i][j] = a[i][j] + s1[i][j - 1] + s1[i - 1][j] - s1[i - 1][j - 1];
		for (int i = 1; i <= n; ++i)
			for (int j = 1; j <= n; ++j)
				s2[i][j] = a[i][j] * i + s2[i][j - 1] + s2[i - 1][j] - s2[i - 1][j - 1];
		for (int i = 1; i <= n; ++i)
			for (int j = 1; j <= n; ++j)
				s3[i][j] = a[i][j] * j + s3[i][j - 1] + s3[i - 1][j] - s3[i - 1][j - 1];
		/*
		1 2    1 2   0 0    1 2       0 0
		3 4 -> 1 2 + 2 2 -> 1 2 + 2 * 1 1
		5 6    1 2   4 4    1 2       2 2
		*/
		for (int i = 1, a, b, c, d; i <= m; ++i) {
			a = io.read() - 1; b = io.read() - 1; c = io.read(); d = io.read();
			int p1 = get(s3, a, b, c, d) - get(s1, a, b, c, d) * b;
			int p2 = get(s2, a, b, c, d) - get(s1, a, b, c, d) * (a + 1);
			io.write(p1 + p2 * (d - b), " \n"[i == m]);
		}
	}
}
