思路分析

首先不难发现，对于这道题来说，出现环，不论奇环偶环，因为相对位置的限制，都一定是无解的。

因此，这个东西的形态只能是一棵树。

只能是一棵树那就好办了。我们分析一下一个节点的子节点的形态及其限制。

首先任意一个节点不可能拥有超过 $2$ 个度数 $>2$ 的子节点。因为一旦分叉，上下对齐一下就会发现必须一左一右分布。

这又告诉我们事实上只有根节点才有这个权力。因为子节点必定是在根节点的左边或者右边已经选了一个位置了。同侧有两个节点是必定交叉的。

显然，将一个拥有至少两个叶子结点的点分配为根节点就能达到最优情况。

然后我们去考虑一下那些叶子节点。对于一个节点，叶子子结点的相对顺序是不重要的，也就是阶乘。

然而如果我有度数 $>2$ 的子节点，那么他们必须优先分配到最边上。特殊判断一下。注意实际上在左边和在右边是近似等价的，所以对于根节点可能还需要乘个 $2$。

最后我们交换南北两岸，也就是答案乘二，就做完了。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
constexpr int mod = 1e9 + 7;
int t, n, m, s[200005], jc[200005]; vector<int>son[200005];
inline void dfs(int p, int f, int d) {
	int gec = 0, sc = son[p].size() - (bool)(f);
	for (int sp : son[p]) if (sp != f)
		gec += (son[sp].size() > 1);
	if (gec > 2 - (bool)(f)) return void(s[p] = 0);
	else if (gec == 2) {
		s[p] = 1;
		for (int sp : son[p]) if (sp != f)
			if (son[sp].size() > 1)
				dfs(sp, p, d + 1), s[p] = s[p] * s[sp] % mod;
		s[p] = s[p] % mod * jc[sc - 2] % mod * 2 % mod;
	}
	else if (gec == 1) {
		s[p] = 1;
		for (int sp : son[p]) if (sp != f)
			if (son[sp].size() > 1)
				dfs(sp, p, d + 1), s[p] *= s[sp];
		if (sc != 1 && !d) s[p] = s[p] * jc[sc - 1] % mod * 2 % mod;
		else s[p] = s[p] * jc[sc - 1] % mod;
	}
	else s[p] = jc[sc];
}
signed main() {
	ios::sync_with_stdio(0); jc[0] = 1; cin >> t;
	for (int i = 1; i <= 2e5; ++i) jc[i] = jc[i - 1] * i % mod;
	for (int j = 1; j <= t; ++j) {
		cin >> n >> m;
		if (m >= n) {
			for (int i = 1, l, r; i <= m; ++i)
				cin >> l >> r;
			cout << "0\n";
			continue;
		}
		for (int i = 1; i <= n; ++i) son[i].clear();
		for (int i = 1, l, r; i != n; ++i)
			cin >> l >> r,
			son[l].emplace_back(r),
			son[r].emplace_back(l);
		if (n == 2) {
			cout << "2\n";
			continue;
		}
		for (int i = 1; i <= n; ++i) {
			if (son[i].size() != 1) {
				dfs(i, 0, 0);
				cout << s[i] * 2 % mod << endl;
				break;
			}
		}
	}
	return 0;
}
