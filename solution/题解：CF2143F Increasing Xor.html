思路分析

注意到，我们可以从前向后扫描，每一个数字都取他能取到的最小值显然是不劣的：他完全不影响后面的数字可以取到哪些取值。第一个数字除外，不过第一个数字总是可以取到 000 的。

我们考虑后面的每一个数字可以取到多小？假设 ap−1a_{p-1}ap−1​ 的最小值为 mmm，那么 apa_pap​ 可以取到的最小值就是当前线性基下 apa_pap​ 可以异或出来的大于 mmm 的最小值。

显然我们可以考虑把 apa_pap​ 插入线性基，把 mmm 在当前的线性基中排名给搞出来，然后再找到排名恰好高 111 的数字即可。

于是直接暴力就是 O(n2log⁡V)O(n^2\log V)O(n2logV) 的了，显然无法通过。

不过我们考虑一下将一些数字插入到不同线性基中会有什么样的性质：显然只有 log⁡V\log VlogV 个数字会插入成功，导致线性基发生变化。

也就是说在这道题中，无论区间有多长，实际上不一样的线性基也只会有 log⁡V\log VlogV 个。

能否利用这一点？我们可以考虑进行二分。具体的，我们维护出原序列的前缀线性基，然后每一次二分出线性基发生变化的位置，然后想办法直接移动过去。

欸？怎么一次性移动很多个元素啊？我们二分的东西保证了我们要尝试插入的东西不会改变线性基，于是我们获取到前一个元素的排名，这段区间的数字就会从小到大向后再分配若干个排名。

此题可解，时间复杂度 O(nlog⁡2V)O(n\log^2V)O(nlog2V)，代码如下：

CPP 
#include<bits/stdc++.h>
//#pragma GCC optimize(2)
using namespace std;
//#define int long long
constexpr bool online=0;
#define _rw_file ""
#define sipt //signed-input
#define sopt //signed-output
struct IO {
#define mxsz (1 << 21)
	char buf[mxsz], * p1, * p2;
	char pbuf[mxsz], * pp; bool acc[128];
	IO() : p1(buf), p2(buf), pp(pbuf) {}
	~IO() { fwrite(pbuf, 1, pp - pbuf, stdout); }
	inline char gc() {
		if (p1 == p2) p2 = (p1 = buf) + fread(buf, 1, mxsz, stdin);
		return p1 == p2 ? ' ' : *p1++;
	}
	inline void setacc(const char* c) { while (*c) acc[*c++] = 1; }
	inline char getc() { char c; while (!acc[c = gc()]); return c; }
#ifndef sipt
	inline int read() {
		int r = 0; char c = gc(); while (c < '0' || c>'9') c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return r;
	}
#else
	inline int read() {
		int r = 0; char c = gc(); bool rev = 0;
		while (c < '0' || c>'9') rev |= (c == '-'), c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return rev ? ~r + 1 : r;
	}
#endif
	inline void push(const char& c) {
		if (pp - pbuf == mxsz) fwrite(pbuf, 1, mxsz, stdout), pp = pbuf;
		*pp++ = c;
	}
	inline void write(int x) {
		static char sta[22]; int top = 0;
		do sta[top++] = x % 10, x /= 10; while (x);
		while (top) push(sta[--top] ^ 48);
	}
	inline void write(int x, char opc) {
#ifdef sopt
		if (x < 0) push('-'), x = ~x + 1;
#endif
		write(x), push(opc);
	}
	inline void puts(const char* c) { while (*c) push(*c++); }
} io;
int t,n,m,a[200005];
struct liner_basis{
	int v[22],t[22];
	inline void ins(int a,int c){
		for(int i=19;i>=0;i--)
			if(a>>i&1)
				if(!v[i]) return v[i]=a,void(t[i]=c);
				else{
					if(c>t[i]) swap(c,t[i]),swap(a,v[i]);
					a^=v[i];
				}
	}
	inline int rank(int a,int l){
		int cn=1,ca=0,rk=1;
		for(int i=0;i!=20;++i)
			if(t[i]>=l) cn<<=1;
		for(int i=19;i>=0;i--)
			if(t[i]<l)continue;
			else if(cn>>=1,ca>>i&1)
				if(a>>i&1) rk+=cn;
				else ca^=v[i];
			else
				if(a>>i&1) rk+=cn,ca^=v[i];
				else;
		return rk;
	}
	inline bool diff(int l,liner_basis&r){
		int c1=0,c2=0;
		for(int i=0;i!=20;++i)
			c1+=(t[i]<l),c2+=(r.t[i]<l);
		return c1!=c2;
	}
	inline int val(int a,int l){
		int cn=1,ca=0; //cerr<<a<<endl;
		for(int i=0;i!=20;++i)
			if(t[i]>=l) cn<<=1;
		if(a>=cn) return -2;
		for(int i=19;i>=0;i--)
			if(t[i]<l) continue;
			else if(cn>>=1,ca>>i&1)
				if(a>=cn) a-=cn;
				else ca^=v[i];
			else
				if(a>=cn) ca^=v[i],a-=cn;
				else;
		return ca;
	}
}lib[200005];
signed main(){
	if(online)
		freopen(_rw_file ".in","r",stdin),
		freopen(_rw_file ".out","w",stdout);
    ios::sync_with_stdio(0);
	for(t=io.read();t;t--){
		n=io.read(); m=io.read();
		for(int i=1;i<=n;++i)
			lib[i]=lib[i-1], lib[i].ins(io.read(),i);
		for(int i=1,l,r;i<=m;++i){
			l=io.read(); r=io.read();
			int cur=-1,lp,rp,mid,p=l;
			while(p<=r){
				lp=p,rp=r;
				while(lp!=rp){
					mid=lp+rp+1>>1;
					if(lib[mid].diff(l,lib[p])) rp=mid-1;
					else lp=mid;
				}
				int rk=((cur>=0)?lib[p].rank(cur,l):0);
				int rkn=rk+lp-p;
				cur=lib[p].val(rkn,l);
				//cerr<<lp<<" "<<cur<<endl;
				if(cur<-1){
					puts("NO");
					break;
				}
				p=lp+1;
			}
			if(cur>=0) puts("YES");
		}
	}
}
