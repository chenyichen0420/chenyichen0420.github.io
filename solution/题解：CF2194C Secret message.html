注意到最短的长度一定是原串长度的因数，只有 O(n)O(\sqrt n)O(n​) 个，于是问题变成检验是否存在循环节为某一指定的数的方案。

显然如果存在，那说明对于每一个 i∈[0,l)i\in[0,l)i∈[0,l) 有一个字符 c∈∩j≡i(modl){s1j,s2j,s3j}c\in\cap_{j\equiv i\pmod l}\{s1_j,s2_j,s3_j\}c∈∩j≡i(modl)​{s1j​,s2j​,s3j​}。直接合并就行了，毕竟字符串就只有 333 个。时间复杂度是 O(nn)O(n\sqrt n)O(nn​) 的。

考场上在发癫，写了个 bitset，所以是 O(nn∣Σ∣w)O(n\sqrt n\frac{|\Sigma|}{w})O(nn​w∣Σ∣​) 的（好像还更优一点）。代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
int t, n, m; string s[50005];
bitset<32>hs[50005], tmp[50005];
signed main() {
    ios::sync_with_stdio(0);
    for (cin >> t; t; t--) {
        cin >> n >> m;
        for (int i = 1; i <= m; ++i) cin >> s[i];
        for (int i = 0; i != n; ++i)
            for (int j = 1; j <= m; ++j)
                hs[i][s[j][i] - 'a'] = 1;
        for (int i = 1; i <= n; ++i)
            if (n % i == 0) {
                for (int j = 0; j != i; ++j) tmp[j].set();
                for (int j = 0; j != n; ++j) tmp[j % i] &= hs[j];
                bool cn = 1;
                for (int j = 0; j != i; ++j)
                    if (!tmp[j].count()) cn = 0;
                if (cn) {
                    // cerr << i << endl;
                    for (int j = 0; j != n; ++j)
                        for (int k = 0; k != 26; ++k)
                            if (tmp[j % i][k]) {
                                cout << char(k + 'a');
                                break;
                            }
                    cout << endl; break;
                }
            }
        for (int i = 0; i != n; ++i)
            for (int j = 1; j <= m; ++j)
                hs[i][s[j][i] - 'a'] = 0;
    }
    return 0;
}
