思路分析

考虑记录出对于每一个 i，他前面的和与 j 同余的方案数，2002×n 的时间复杂度暴力转移。

我们发现这样做非常可行。你可以在 i,j 记录一下他上一个选的数可以是哪些 p，如果这个大小在 2 及以上，那就说明找到了一组解，想办法回溯即可。

有一个小问题：在实现的时候两个数组的最后一个被选的数不一定相同，解决方法也很简单：最后再加一个 0 就行了，记得从输出中抛掉。

代码如下：

#include<bits/stdc++.h>
using namespace std;
#define int long long
int n, a[205]; bool cn[205][205];
vector<int>w[205][205], al, ar;
signed main() {
	ios::sync_with_stdio(0);
	cin >> n; cn[0][0] = 1;
	for (int i = 1; i <= n; ++i) {
		cin >> a[i];
		for (int p = 0; p != i; ++p)
			for (int j = 0; j != 200; ++j)
				if (cn[p][j])
					cn[i][(j + a[i]) % 200] = 1,
					w[i][(j + a[i]) % 200].emplace_back(p);
		for (int j = 0; j != 200; ++j) {
			if (cn[i][j] && w[i][j].size() > 1) {
				cout << "Yes\n"; int p = i, v = j;
				al.emplace_back(p); v = (v - a[p] + 200) % 200;
				p = w[p][(v + a[p]) % 200][0];
				while (p) {
					al.emplace_back(p); v = (v - a[p] + 200) % 200;
					if (!w[p][(v + a[p]) % 200].size()) break;
					p = w[p][(v + a[p]) % 200][0];
				}
				p = i, v = j;
				ar.emplace_back(p); v = (v - a[p] + 200) % 200;
				p = w[p][(v + a[p]) % 200][1];
				while (p) {
					ar.emplace_back(p); v = (v - a[p] + 200) % 200;
					if (!w[p][(v + a[p]) % 200].size()) break;
					p = w[p][(v + a[p]) % 200][0];
				}
				cout << al.size() << " ";
				while (al.size()) cout << al.back() << " ", al.pop_back();
				cout << endl;
				cout << ar.size() << " ";
				while (ar.size()) cout << ar.back() << " ", ar.pop_back();
				cout << endl;
				return 0;
			}
		}
	}
	{
		int i = n + 1;
		for (int p = 0; p != i; ++p)
			for (int j = 0; j != 200; ++j)
				if (cn[p][j])
					cn[i][(j + a[i]) % 200] = 1,
					w[i][(j + a[i]) % 200].emplace_back(p);
		for (int j = 0; j != 200; ++j) {
			if (cn[i][j] && w[i][j].size() > 1) {
				int p = i, v = j;
				v = (v - a[p] + 200) % 200;
				p = w[p][(v + a[p]) % 200][0];
				while (p) {
					al.emplace_back(p); v = (v - a[p] + 200) % 200;
					if (!w[p][(v + a[p]) % 200].size()) break;
					p = w[p][(v + a[p]) % 200][0];
				}
				p = i, v = j;
				v = (v - a[p] + 200) % 200;
				p = w[p][(v + a[p]) % 200][1];
				while (p) {
					ar.emplace_back(p); v = (v - a[p] + 200) % 200;
					if (!w[p][(v + a[p]) % 200].size()) break;
					p = w[p][(v + a[p]) % 200][0];
				}
				if (!al.size() || !ar.size()) {
					al.clear(); ar.clear();
					continue;
				}
				cout << "Yes\n";
				cout << al.size() << " ";
				while (al.size()) cout << al.back() << " ", al.pop_back();
				cout << endl;
				cout << ar.size() << " ";
				while (ar.size()) cout << ar.back() << " ", ar.pop_back();
				cout << endl;
				return 0;
			}
		}
	}
	cout << "No\n";
}
