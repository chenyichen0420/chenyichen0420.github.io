思路分析

很简单，直接枚举四种情况，看 Suneet 赢不赢就行了。单词查询复杂度 O(1)，常数 4。

代码如下：

#include<iostream>
using namespace std;
short t,a[2],b[2],ans,tmp;
inline bool judge(){
	return ((a[0]>b[0])+(a[1]>b[1]))>((a[0]<b[0])+(a[1]<b[1]));
}
int main(){
	ios::sync_with_stdio(0);
	cin>>t;
	while(t--){
		cin>>a[0]>>a[1]>>b[0]>>b[1];
		ans=judge();
		swap(a[0],a[1]); ans+=judge();
		swap(b[0],b[1]); ans+=judge();
		swap(a[0],a[1]); ans+=judge();
		putchar(ans^48); putchar('\n');
	}
	return 0;
}
