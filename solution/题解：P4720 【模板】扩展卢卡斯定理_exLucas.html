思路分析

学扩展卢卡斯定理不需要会卢卡斯定理，就像学扩展中国剩余定理不需要会中国剩余定理。

扩展卢卡斯定理不再要求 p 是质数。但是不是质数就会少非常多的性质，因此我们想办法和质数扯上一点关系。

我们考虑将 p 质因数分解，假设 P=∏piki​​。如果我们能够计算出每一个 (mn​)(modpiki​​)，就可以使用 crt 快速的合并结果，得到答案。

(mn​)=m!(n−m)!n!​=f(m!)f((n−m)!)f(n!)​×pg(n!)−g(m!)−g((n−m)!)。其中 g(v) 表示 v 中 p 这个质因数的次方数，f(v)=pg(v)v​，也就是将 v 中所有的质因数 p 全部去除掉。

g(v) 是很容易求的，我们现在需要着重考虑 f(v)。因为 p 不是质数，所以请不要像各种神奇的技巧，直接计算答案。

我们模仿 g(v) 的计算方式，逐步删去多余的 p 并计算答案。具体的，我们直接扔掉当前区间内 p 的倍数，并放到下一批继续处理。

假如说 pk=3，那么我们的计算会像这样：

1×2×3×4×5×6×7×8×9≡(1×2×4×5×7×8)×(1×2×3)(modpk)。

1×2×3≡(1×2)×(1)(modpk)。

我们会首先将倍数全部拎出来，抛去一层 p 之后就又变成了 1 开头的连续区间，递归处理即可。我们主要考虑剩下的部分。

剩下的部分我们发现仍然是呈现出很强的周期性。单组内的乘积为 ∏i=1pk−1​imodpk。这样的东西我们总共有 ⌊pkn​⌋ 组。剩余的部分直接暴力处理就行。

也就是说只要你的模数 P=∏piki​​ 满足 ∑piki​​ 不大，就可以快速的计算出你想要的结果。

思路其实不难，代码也没啥，模拟出来就行了。

代码如下：

#include<bits/stdc++.h>
using namespace std;
#define int long long
int n, m, p, d[60], c[60], cc;
inline int qpow(int a, int b, int p) {
	int ret = 1;
	for (; b; b >>= 1, a = a * a % p)
		(b & 1) && (ret = ret * a % p);
	return ret;
}
inline int fac(int n, int p, int md) {
	if (!n) return 1; int ret = 1;
	for (int i = 1; i != md; ++i)
		if (i % p) ret = ret * i % md;
	ret = qpow(ret, n / md, md);
	for (int i = n % md; i >= 1; --i)
		if (i % p) ret = ret * i % md;
	return ret * fac(n / p, p, md) % md;
}
inline int exgcd(int a, int b, int& x, int& y) {
	if (!b) return x = 1, y = 0, a;
	int v = exgcd(b, a % b, y, x);
	y -= (a / b) * x; return v;
}
inline int inv(int a, int p) {
	int x, y; exgcd(a, p, x, y);
	return (x % p + p) % p;
}
inline int caf(int v, int p) {
	int ret = 0; do ret += (v /= p); while (v); return ret;
}
inline int cvl(int n, int m, int p, int md) {
	if (n < m) return 0;
	int f1 = fac(n, p, md), f2 = fac(m, p, md), f3 = fac(n - m, p, md);
	int cnt = caf(n, p) - caf(m, p) - caf(n - m, p);
	return f1 * inv(f2, md) % md * inv(f3, md) % md * qpow(p, cnt, md) % md;
}
inline int crt() {
	int t = 1, ret = 0;
	for (int i = 0; i != cc; ++i) t *= c[i];
	for (int i = 0; i != cc; ++i)
		ret = (ret + d[i] * (t / c[i]) % t * inv(t / c[i], c[i]) % t) % t;
	return ret;
}
inline int exlucas(int n, int m, int p) {
	for (int i = 2; i * i <= p; ++i)
		if (p % i == 0) {
			int tmp = 1; while (p % i == 0) p /= i, tmp *= i;
			d[cc] = cvl(n, m, i, tmp), c[cc++] = tmp;
		}
	if (p > 1) d[cc] = cvl(n, m, p, p), c[cc++] = p;
	return crt();
}
signed main() {
	ios::sync_with_stdio(0);
	cin >> n >> m >> p;
	cout << exlucas(n, m, p) << endl;
}
