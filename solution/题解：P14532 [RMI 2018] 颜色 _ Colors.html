思路分析

思考暴力怎么做。注意到，一个点能从 aia_iai​ 变成 bib_ibi​，一个必须满足的条件是 ∃aj=bi\exist a_j=b_i∃aj​=bi​。

于是我们思考一下，什么情况下，aja_jaj​ 可以传递到 iii 来。显然，只要途径的所有点 kkk 都满足 ak≥aj≥bka_k\ge a_j\ge b_kak​≥aj​≥bk​ 即可。

必要性很明显，充分性：将每一个点在颜色传递过程中可能会变到的颜色从大到小排列，此时编号最大的连通。传递过去后次大颜色连通，以此类推。

于是这个问题就变成了对于每一条边 i,ji,ji,j，仅在 [max⁡(bi,bj),min⁡(ai,aj)][\max(b_i,b_j),\min(a_i,a_j)][max(bi​,bj​),min(ai​,aj​)] 时间段存在，此时询问在 bkb_kbk​ 时刻 kkk 是否与另若干个 al=bka_l=b_kal​=bk​ 的 lll 连通。

然后就变成了线段树分治的板子题。代码如下：

#include<bits/stdc++.h>

struct IO {
#define mxsz (1 << 21)
	char buf[mxsz], * p1, * p2;
	char pbuf[mxsz], * pp; bool acc[128];
	IO() : p1(buf), p2(buf), pp(pbuf) {}
	~IO() { fwrite(pbuf, 1, pp - pbuf, stdout); }
	inline char gc() {
		if (p1 == p2) p2 = (p1 = buf) + fread(buf, 1, mxsz, stdin);
		return p1 == p2 ? ' ' : *p1++;
	}
	inline void setacc(const char* c) { while (*c) acc[*c++] = 1; }
	inline char getc() { char c; while (!acc[c = gc()]); return c; }
#ifndef sipt
	inline int read() {
		int r = 0; char c = gc(); while (c < '0' || c>'9') c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return r;
	}
#else
	inline int read() {
		int r = 0; char c = gc(); bool rev = 0;
		while (c < '0' || c>'9') rev |= (c == '-'), c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return rev ? ~r + 1 : r;
	}
#endif
	inline void push(const char& c) {
		if (pp - pbuf == mxsz) fwrite(pbuf, 1, mxsz, stdout), pp = pbuf;
		*pp++ = c;
	}
	inline void write(int x) {
		static char sta[22]; int top = 0;
		do sta[top++] = x % 10, x /= 10; while (x);
		while (top) push(sta[--top] ^ 48);
	}
	inline void write(int x, char opc) {
#ifdef sopt
		if (x < 0) push('-'), x = ~x + 1;
#endif
		write(x), push(opc);
	}
	inline void puts(const char* c) { while (*c) push(*c++); }
} io;
int t,n,m,a[200005],b[200005],cn[200005];
vector<int>pa[200005],pb[200005];
class mersers{
	int f[200005],s[200005]; stack<int>p,sz; bool h[200005];
	inline int find(int p){return f[p]!=p?find(f[p]):p;}
public:
	inline void merge(int l,int r){
		l=find(l),r=find(r);
		if(l==r) return;
		if(s[l]<s[r]) swap(l,r);
		s[l]+=s[r],f[r]=l; p.emplace(r);
	}
	inline void save(){sz.emplace(p.size());}
	inline void roll(){
		while(p.size()!=sz.top())
			s[f[p.top()]]-=s[p.top()],
			f[p.top()]=p.top(),p.pop();
		sz.pop();
	}
	inline bool& mark(int p){return h[find(p)];}
	inline void reset(int sz){
		for(int i=1;i<=sz;++i) s[f[i]=i]=1;
	}
}mss;
struct seg_tree{
	struct node{
		int l,r; vector<pair<int,int>>v;
	}re[200005<<2];
	inline void build(int l,int r,int p){
		re[p].l=l; re[p].r=r; re[p].v.clear();
		if(l==r) return; int mid=l+r>>1;
		build(mid+1,r,p<<1|1); build(l,mid,p<<1);
	}
	inline void ins(int l,int r,const pair<int,int>&v,int p){
		if(re[p].l>=l&&re[p].r<=r) return void(re[p].v.emplace_back(v));
		if(l<=re[p<<1].r) ins(l,r,v,p<<1);
		if(r>re[p<<1].r) ins(l,r,v,p<<1|1);
	}
	inline void dfs(int p){
		mss.save();
		for(const auto&e:re[p].v) mss.merge(e.first,e.second);
		if(re[p].l==re[p].r){
			for(int c:pa[re[p].l]) mss.mark(c)=1;
			for(int c:pb[re[p].l]) cn[c]=mss.mark(c);
			for(int c:pa[re[p].l]) mss.mark(c)=0;
		}
		else dfs(p<<1),dfs(p<<1|1);
		mss.roll();
	}
}sgt;

        for(int i=1;i<=n;++i)
            pa[a[i]=io.read()].emplace_back(i);
        for(int i=1;i<=n;++i)
            pb[b[i]=io.read()].emplace_back(i);
        for(int i=1,l,r;i<=m;++i)
            if(l=io.read(),r=io.read(),max(b[l],b[r])<=min(a[l],a[r]))
				sgt.ins(max(b[l],b[r]),min(a[l],a[r]),make_pair(l,r),1);
		mss.reset(n); sgt.dfs(1); cn[0]=1;
		for(int i=1;i<=n;++i) cn[0]&=cn[i];
		cout<<cn[0]<<endl;
    }
