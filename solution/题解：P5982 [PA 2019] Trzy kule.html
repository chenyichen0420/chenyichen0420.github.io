思路分析

三个条件或起来，那肯定就是正难则反，求不满足要求的字符串数量了。

对于三个字符串的不满足要求的位置，考虑单个位置有几种可能的贡献：显然是 $8$ 种。这 $8$ 种按照是否贡献到 $1$ 上面分成两类，并将两类中互补对象配对形成 $4$ 对。

这四对的出现次数是确定的，比如分别是 $a,b,c,d$，并假设这四种当中，对 $1$ 有贡献的那一个的出现次数依次是 $x,y,z,w$。

那么最后的约束可以写成：

  1. $x,y,z,w$ 分别不大于 $a,b,c,d$。
  2. $x+y+z+w>r1$。
  3. $x+y-z-w>r2-c-d$。
  4. $x-y+z-w>r3-b-d$。

我们要计算的是 $x,y,z,w$ 的总方案数。

从中，我们显然注意到 $x$ 的贡献系数是完全一致的，所以我们将 $x$ 单独枚举，但是后面的 $y,z,w$ 怎么办？

或者说详细一点，我们枚举 $x$，那么需要求解的就是 $y+z+w>r1-x\land y-z-w>r2-c-d-x\land z-y-w>r3-b-d-x$ 的方案数之和。

进一步扩展这个思路：我们求解 $y+z>const1,y-z>const2,z-y>const3$ 的方案数，然后计算答案的时候枚举 $x,w$ 的数值。

注意到前面的 $y-z$ 和 $z-y$ 正好是相反数，因此可以少记录一维。这样是一个 $O(n^2)$ 预处理，$O(n^2)$ 查询的做法。

$const$ 的数值可以自己手推一下，不懂得就看下面的参考代码，相应位置有注释：

#include<bits/stdc++.h>
using namespace std;
#define int long long
// bool membeg;
constexpr bool online=1;
constexpr int mod=1e9+7;
int n,lim[4],r1,r2,r3,c[10005][10005];
long long ans,jc[10005],ny[10005];
char c1[10005],c2[10005],c3[10005];
inline void mad(int&l,const int r){(l+=r)>=mod&&(l-=mod);}
inline long long qpow(long long a,int b){
	long long ret=1;
	for(;b;b>>=1,a=a*a%mod)
		(b&1)&&(ret=ret*a%mod);
	return ret;
}
inline long long cvl(int n,int m){
	// cerr<<n<<" "<<m<<" "<<jc[n]*ny[m]%mod*ny[n-m]%mod<<endl;
	return jc[n]*ny[m]%mod*ny[n-m]%mod;
}
// bool memend;
signed main(){
	if(online)
		freopen("kul.in","r",stdin),
		freopen("kul.out","w",stdout);
	// memset(c,0,sizeof c);
	// cerr<<abs((long long)(&membeg)-(long long)(&memend))<<endl;
	ios::sync_with_stdio(0); cin>>n;
	for(int i=jc[0]=1;i<=n;++i) jc[i]=jc[i-1]*i%mod;
	ny[n]=qpow(jc[n],mod-2);
	for(int i=n;i>=1;i--) ny[i-1]=ny[i]*i%mod;
	/*
	那肯定就是求不满足要求的字符串数量了
	我有一些大胆的想法：能不能搞一个三元多项式？
	没有前途，每一元的次数都是 n 的，实际上没什么优化空间

	1～8 是简单的，直接 dp_{p,i,j,k} 就行了
	9~10 的特殊性质是 r<=500
	那时间复杂度肯定就是 O(r^3) 或者 O(nr^2) 的了
	感觉后者更有可能一些，思考一下：

	虽然我可能误解了最后那个神秘特殊性质的意思
	但是它确实给了我一点点启发：    
	我们可不可以把三个字符串都一样的位置直接删除掉，最后再合并进来？
	这为什么有道理：剩下的位置一定是 1/2 的选择，

	等一下，为什么我一定只剥离三个都一样的情况？
	考虑有几种可能的贡献：显然是 8 种
	这八种按照是否贡献到 1 上面分成两类
	我们将两类中互补对象配对形成四对
	这四对的出现次数是确定的，比如分别是 a,b,c,d
	假设这四种当中，对 1 有贡献的出现次数依次是 x,y,z,w
	那么最后的约束可以写成 x,y,z,w 分别不大于 a,b,c,d
	并且 x+y+z+w > r1
	并且 x+y-z-w > r2-c-d
	并且 x-y+z-w > r3-b-d
	的方案数计数

	从中显然注意到 x 的贡献系数是完全一致的
	所以前面我才会提出来独立 x 但是后面的 y,z,w 怎么办？
	本质上就是在求解：
	y+z+w > a
	y-z-w > b
	z-y-w > c
	并且 y,z,w 有一个不到 n 的上界的方案数
	直接枚举 y,z,w 就可以知道 =a,=b,=c 的方案数
	所以说再做一个前缀和不就 >a,>b,>c 了？
	显然时间复杂度就是 O(n^3) 了

	进一步扩展这个思路：我们求解
	y+z > a, y-z > b, z-y > c 的方案数
	然后计算答案的时候枚举 x,w 的数值
	注意到前面的 y-z 和 z-y 正好是相反数，因此可以少记录一维
	这样是不是一个 O(n^2) 预处理，O(n^2) 查询的做法？
	好像是对的！！！
	*/
	cin>>r1>>c1>>r2>>c2>>r3>>c3;
	for(int i=0;i!=n;++i){
		if(c1[i]==c2[i]&&c1[i]==c3[i]) lim[0]++;
		if(c1[i]==c2[i]&&c1[i]!=c3[i]) lim[1]++;
		if(c1[i]!=c2[i]&&c1[i]==c3[i]) lim[2]++;
		if(c1[i]!=c2[i]&&c1[i]!=c3[i]) lim[3]++;
	}
	// cerr<<lim[0]<<" "<<lim[1]<<" "<<lim[2]<<" "<<lim[3]<<endl;
	for(int y=0;y<=lim[1];++y)
		for(int z=0;z<=lim[2];++z)
			c[y+z][y-z+lim[2]]=cvl(lim[1],y)*cvl(lim[2],z)%mod;
			// cerr<<y+z<<" "<<y-z+lim[2]<<" "<<c[y+z][y-z+lim[2]]<<endl;
	// cerr<<endl;
	for(int i=lim[1]+lim[2];i>=0;i--)
		for(int j=lim[1]+lim[2];j>=0;j--)
			mad(c[i][j],c[i][j+1]),
			mad(c[i][j],c[i+1][j]),
			mad(c[i][j],mod-c[i+1][j+1]);
			// cerr<<i<<" "<<j<<" "<<c[i][j]<<endl;
	// cerr<<endl;
	for(int x=0;x<=lim[0];++x)
		for(int w=0;w<=lim[3];++w){
			/*
			CRITICAL STEP，VERIFY CKP，重新推导一下贡献：
			对于 s1，不满足的位置有 x+y+z+w 个
			对于 s2，不满足的位置有 x+y+(lim2-z)+(lim3-w) 个
			对于 s3，不满足的位只有 x+(lim1-y)+z+(lim3-w) 个
			于是：
			x,y,z,w 已经通过枚举范围，满足了上界的要求
			x+y+z+w > r1，因此要求 y+z > r1-x-w
			x+y+(lim2-z)+(lim3-w) > r2，因此要求 y-z > r2-lim2-lim3+w-x
			x+(lim1-y)+z+(lim3-w) > r3，因此要求 z-y > r3-lim1-lim3+w-x
			也就是 -(r3-lim1-lim3+w-x) > y-z > r2-lim2-lim3+w-x
			*/
			int lb1=max(r1-x-w+1,0ll);
			int lb2=max(r2-lim[2]-lim[3]+w-x+1,-lim[2]);
			int rb2=-(r3-lim[1]-lim[3]+w-x);
			// cerr<<x<<" "<<w<<" "<<lb1<<" "<<lb2<<" "<<rb2<<endl;
			if(rb2<=lb2) continue;
			int add=(c[lb1][lb2+lim[2]]-c[lb1][rb2+lim[2]]+mod);
			// cerr<<c[lb1][lb2+lim[2]]<<" "<<c[lb1][rb2+lim[2]]<<endl;
			add=add*cvl(lim[0],x)%mod*cvl(lim[3],w)%mod;
			ans=(ans+add)%mod;
		}
	cout<<(qpow(2,n)-ans+mod)%mod<<endl;
}
