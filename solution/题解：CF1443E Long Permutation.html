思路分析

不难注意到，总的操作次数不超过 1010。而我们知道在全排列中更改到倒数第 k 位需要 k! 次更改。

因此我们暴力维护最后的部分就行了。这部分的长度不超过 15。

具体的来说，我们不难注意到在初始序列中，我们执行 k! 次操作之后，k 位置（倒数）变成 k+1，k+1 变到 k 那里去。再执行一下就变成 k+2，k 去 k+1，k+1 去 k+2，以此类推。

也就是说执行 p×k! 次操作之后，[k,k+p−1] 和 k+p 整体换一次位置。

实现出来就行了。下面这份代码实现的比较粗糙，之能勉强卡过去。

#include<bits/stdc++.h>
using namespace std;
constexpr bool online=0;
#define int long long
#define sipt //signed-input
#define sopt //signed-output
struct IO {
#define mxsz (1 << 21)
	char buf[mxsz], * p1, * p2;
	char pbuf[mxsz], * pp; bool acc[128];
	IO() : p1(buf), p2(buf), pp(pbuf) {}
	~IO() { fwrite(pbuf, 1, pp - pbuf, stdout); }
	inline char gc() {
		if (p1 == p2) p2 = (p1 = buf) + fread(buf, 1, mxsz, stdin);
		return p1 == p2 ? ' ' : *p1++;
	}
	inline void setacc(const char* c) { while (*c) acc[*c++] = 1; }
	inline char getc() { char c; while (!acc[c = gc()]); return c; }
#ifndef sipt
	inline int read() {
		int r = 0; char c = gc(); while (c < '0' || c>'9') c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return r;
	}
#else
	inline int read() {
		int r = 0; char c = gc(); bool rev = 0;
		while (c < '0' || c>'9') rev |= (c == '-'), c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return rev ? ~r + 1 : r;
	}
#endif
	inline void push(const char& c) {
		if (pp - pbuf == mxsz) fwrite(pbuf, 1, mxsz, stdout), pp = pbuf;
		*pp++ = c;
	}
	inline void write(int x) {
		static char sta[22]; int top = 0;
		do sta[top++] = x % 10, x /= 10; while (x);
		while (top) push(sta[--top] ^ 48);
	}
	inline void write(int x, char opc) {
#ifdef sopt
		if (x < 0) push('-'), x = ~x + 1;
#endif
		write(x), push(opc);
	}
	inline void puts(const char* c) { while (*c) push(*c++); }
} io;
int n,m,a[200005],sm[200005],jc[20],nt[20],st;
signed main(){
    if(online)
        freopen("wsx.in","r",stdin),
        freopen("wsx.out","w",stdout);
    ios::sync_with_stdio(0);
    n=io.read(); m=io.read(); jc[0]=1;
    for(int i=1;i<=n;++i) a[i]=i, sm[i]=sm[i-1]+i;
    for(int i=1;i<=15;++i) jc[i]=jc[i-1]*i;
    for(int i=1,o,l,r;i<=m;++i)
        if(o=io.read(),l=io.read(),o&1) r=io.read(),io.write(sm[r]-sm[l-1],'\n');
        else{
            st+=l; l=st;
            for(int j=max(1ll,n-15);j<=n;++j) a[j]=j;
            for(int j=15;j;j--)
                while(l>=jc[j]){
                    l-=jc[j]; int sp = n-j,bp,bv=1e9;
                    for(int k=sp+1;k<=n;++k)
                        if(a[k]>a[sp]&&a[k]<bv)
                            bv=a[k],bp=k;
                    swap(a[sp],a[bp]);
                }
            for(int j=max(1ll,n-15);j<=n;++j)
                sm[j]=sm[j-1]+a[j];
        }
    return 0;
}
