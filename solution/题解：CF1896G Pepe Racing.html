思路分析

如果我只问你最大的那一个，那你怎么办？

这不唐完了吗？直接每 $n$ 个一组，分别选出最大值，然后再把第一轮的最大值和起来得到一个新的最大值就是最终的答案了。

那么，这个做法有没有扩展的希望呢？似乎是有的，我们把最大值删掉之后，把那一堆剩下部分的最大值给选出来不就行了，吗？

好像，剩的不够 $n$ 个了？确实。

那，我从别的堆里面随便薅一个出来放到自己这里是对的吗？我们发现这样确实是对的。考虑归纳，假设前 $k$ 大的都已经被删干净了，我们只需要保证第 $k+1$ 个在某一堆的最大值就行了。

可是我们发现，无论被移动到哪一堆，到底是怎么移的，总之发生移动之后会立即在被移动到的那一堆选出当前的最大值，于是第 $k+1$ 个一定处在堆顶！

没有做完，我们要求了薅的时候只能薅非最大值，那我最后 $2n-1$ 个元素选不出来啊，元素个数不够了！

不过，我们注意到性质：最后的 $2n-1$ 个元素中会被选出来的一定就是此时的堆顶的那 $n$ 个元素，于是我们直接对他们找最大值，删掉然后随便找一个顶上来就行了。最后只剩一个该删元素的时候直接把他拎出来就行。

算一下操作次数：

确定初始堆顶 $n$ 次，确定前 $n^2-2n+1$ 个元素 $2(n^2-2n+1)=2n^2-4n+2$ 次，确定后 $n$ 个元素用 $n-1$ 次，总共正好 $2n^2-2n+1$ 次。

思路还是比较自然的，瞎实现的代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
int t,n,bs[25];
set<int>v[25];
vector<int>ans;
signed main(){
    ios::sync_with_stdio(0);
	for(cin>>t;t;t--){
		cin>>n; ans.clear();
		for(int i=1;i<=n;++i) v[i].clear();
		for(int i=1;i<=n;++i)
			for(int j=1,a;j<=n;++j)
				v[i].emplace((i-1)*n+j);
		for(int i=1;i<=n;++i){
			cout<<"? ";
			for(int a:v[i]) cout<<a<<" ";
			cout<<endl;
			int a; cin>>a;
			v[i].erase(a);
			bs[i]=a;
		}
		for(int i=1;i<=n*n-n*2+1;++i){
			cout<<"? ";
			for(int j=1;j<=n;++j) cout<<bs[j]<<" ";
			cout<<endl;
			int a;
			cin>>a;
			ans.emplace_back(a);
			for(int j=1;j<=n;++j)
				if(bs[j]==a){
					bs[j]=0;
					while(v[j].size()<n)
						for(int k=1;k<=n;++k)
							if(k!=j&&v[k].size()){
								v[j].emplace(*v[k].begin());
								v[k].erase(v[k].begin());
								break;
							}
					cout<<"? ";
					for(int a:v[j]) cout<<a<<" ";
					cout<<endl;
					int b; cin>>b;
					v[j].erase(b);
					bs[j]=b;
				}
		}
		for(int i=1;i<=n;++i) v[0].emplace(bs[i]);
		for(int i=1;i!=n;++i){
			cout<<"? ";
			for(int j=1;j<=n;++j) cout<<bs[j]<<" ";
			cout<<endl;
			int a;
			cin>>a;
			v[0].erase(a);
			ans.emplace_back(a);
			for(int j=1;j<=n&&i!=n;++j)
				if(bs[j]==a){
					bs[j]=0;
					while(!v[j].size())
						for(int k=1;k<=n;++k)
							if(k!=j&&v[k].size()){
								v[j].emplace(*v[k].begin());
								v[k].erase(v[k].begin());
								break;
							}
					int b=*v[j].begin();
					v[j].erase(b);
					bs[j]=b;
				}
		}
		ans.emplace_back(*v[0].begin());
		cout<<"! "; v[0].clear();
		for(int a:ans) cout<<a<<" ";
		cout<<endl;
	}
}
