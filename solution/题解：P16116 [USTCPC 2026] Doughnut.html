思路分析

猜测：唯一一个 $-1$ 的的情况是四点环，此时 $2\times2$ 和 $1\times4$ 均可。下面不证自白，暂且当作结论使用。

特判几个情况：$k=0$ 时是 $1\times1$，输出 $2$；$k=1$ 时是 $1\times2$，输出 $3$，发现是 $2\times2$ 的结构，输出多解。

首先判断 $1\times n$ 的情况，判定标准：每一个点的度数为 $2$。直接输出 $v+1$ 即可。

其次判断 $2\times n$ 的情况，判定标准：每一点的度数为 $3$，直接输出 $2+\frac{v}{2}$ 即可。

剩余的情况满足两个维度都至少为 $3$，则每一个点的度数均为 $4$。考虑找出其 $n,m$ 的具体数值：

我们任意指定一个点为起点，并不妨设他是左上角的一个格子，但是图仍然是循环的。

对于每一个点，我们标记出他和起点的最短距离。打表就可以发现，他大致从四个角落向中间进行扩散，最后在中间形成峰值，想要理解也是不难的。

此时我们动用一下眼力，注意到如果中间的峰值为 $a$，那么他一定只会是一个 $1\times1,1\times2,2\times2$ 的小矩形。这是显然的：本来就是四角向中间进行扩散，他天生就这样。

于是，我们注意到，如果是 $1\times1$ 的，那么答案就是 $2\times a$，如果是 $1\times2$ 的，就是 $2\times a+1$，如果是 $2\times 2$ 的就是 $2\times a+2$。

这不是巧合：答案本质上就是从左上角走到右下角所需要的步数 $+1$。而上面所说的极值矩形本质上就是指定这个路径经过这些点，坐上到右下就是坐上到矩形的坐上到矩形的右下再到右下。正好体现为上述公式而已。

此题可解。考场代码如下：

CPP 
#include <iostream>
#include <algorithm>
#include <queue>
#include <cstring>
#include <set>

using namespace std;

constexpr int N = 3e5 + 5;
int T, k, d[N], maxd, l[300005], r[300005];
vector<int>son[300005];
int dis[300005]; queue<int>q;

inline void solve() {
	cin >> k;
	maxd = 0;
	set<int>s;
	for (int i = 1; i <= k; ++i) {
		cin >> l[i] >> r[i];
		d[l[i]] = d[r[i]] = 0;
		s.emplace(l[i]);
		s.emplace(r[i]);
	}
	if (k == 0) {
		cout << "2\n";
		return;
	}
	if (k == 1) {
		cout << "3\n";
		return;
	}
	for (int i = 1; i <= k; ++i) {
		d[l[i]] ++, d[r[i]]++;
		maxd = max({ maxd, d[l[i]], d[r[i]] });
	}
	if (maxd == 2) {
		if (s.size() == 4) cout << "-1\n";
		else cout << s.size() + 1 << '\n';
	}
	else if (maxd == 3) {
		cout << s.size() / 2 + 2 << '\n';
	}
	else {
		for (int i = 1;i <= s.size();++i) son[i].clear(), dis[i] = 0;
		for (int i = 1;i <= k;++i)
			son[l[i]].emplace_back(r[i]),
			son[r[i]].emplace_back(l[i]);
		q.emplace(1); dis[1] = 1;
		while (q.size()) {
			int p = q.front(); q.pop();
			for (int sp : son[p])
				if (!dis[sp])
					dis[sp] = dis[p] + 1,
					q.emplace(sp);
		}
		sort(dis + 1, dis + s.size() + 1, greater<int>());
		if (dis[1] > dis[2]) cout << dis[1] * 2 - 2 << endl;
		else if (dis[2] > dis[3]) cout << dis[1] * 2 - 1 << endl;
		else cout << dis[1] * 2 << endl;
	}
}

int main() {
	cin >> T;
	while (T--) solve();
	return 0;
}
