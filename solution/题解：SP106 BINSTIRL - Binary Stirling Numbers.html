看到了一点思路没有，考虑打表。于是，我们打出来表后，按列看，可以发现这样的图案：

于是我们不难发现一堆神奇的性质：

每两个连续的 $m$ 对应的序列是一模一样的。

对于任意的 $l=2^x$，矩阵 $(x\in[0,l-1],y\in[0,l-1]),(x\in[l,2\times l-1],y\in[0,l-1]),(x\in[0,l-1],y\in[l,2\times l-1])$ 三者一模一样，矩阵 $(x\in[l,2\times l-1],y\in[l,2\times l-1])$ 全为 $0$。

因此，我们每次可以将横纵的范围缩小 $\frac{1}{2}$，复杂度做到了 $\log$。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
int t,n,m;
inline bool solve(int l,int m,int n){
	if(l==1) return 1;
	if(m>=l/2&&n>=l/2) return 0;
	return solve(l/2,m%(l/2),n%(l/2));
}
signed main(){
	ios::sync_with_stdio(0);
	for(cin>>t;t;t--){
		cin>>n>>m; n-=m; int lt=(m+1)/2;
		int len=pow(2,ceil(log(double(lt))/log(double(2)))); n%=len;
		cout<<solve(len,lt-1,n)<<endl;
	}
}
