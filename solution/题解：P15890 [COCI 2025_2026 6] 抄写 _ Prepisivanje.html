思路分析

不是哥们，2026 年了，这种板题还能干到 #6 来？

这种互斥选择题目，一般考虑抓一种分界方式，使得“移动”一次之后点的类型改变，也就是染成二分图。然后经典结论：最大选点数为总点数减去最小割。我们同时也知道最小割就是最大匹配，然后就可以做了。

或者直接考虑网络流，源点连到左部点，右部点连到汇点。有边的左右部点之间连一个无限的边表示不可断。这就解释了上面那个经典结论，同时直接网络流一跑就是对的。

考虑染色，容易注意到，每一次操作都会导致 x+yx+yx+y 的奇偶性改变，于是按照 x+yx+yx+y 的奇偶性分类就行了。

代码如下：

CPP 
#include<iostream>
#include<vector>
#include<queue>
#include<cstring>
using namespace std;
int n, m, s, t, cnt; char c[85][85];
struct dinic {
	struct node { int p, f; }tmp;
	vector<node>e; vector<int>h[6405];
	inline void ins(int l, int r, int f) {
		tmp = { r,f };
		h[l].emplace_back(e.size());
		e.emplace_back(tmp);
		tmp = { l,0 };
		h[r].emplace_back(e.size());
		e.emplace_back(tmp);
	}
	int d[6405];
	inline bool bfs() {
		memset(d, 0x0f, sizeof d);
		d[s] = 0; queue<int>q({ s });
		while (q.size()) {
			int p = q.front(); q.pop();
			for (int i : h[p])
				if (d[e[i].p] > d[p] + 1 && e[i].f)
					d[e[i].p] = d[p] + 1,
					q.emplace(e[i].p);
		}
		return d[t] <= 1e6;
	}
	int c[6405];
	inline int dfs(int p, int f) {
		if (!f || p == t) return f;
		int ret = 0, t;
		for (int& i = c[p];i != h[p].size();++i)
			if (d[e[h[p][i]].p] == d[p] + 1 && (t = dfs(e[h[p][i]].p, min(f, e[h[p][i]].f)))) {
				f -= t; e[h[p][i]].f -= t; ret += t; e[h[p][i] ^ 1].f += t;
				if (!f) return ret;
			}
		return ret;
	}
	inline int get() {
		int ret = 0;
		while (bfs())
			memset(c, 0, sizeof c),
			ret += dfs(s, 1e9);
		return ret;
	}
}dc;
inline int get(int x, int y) {
	return (x - 1) * m + y;
}
signed main() {
	ios::sync_with_stdio(0); cin >> n >> m;
	for (int i = 1;i <= n;++i)
		for (int j = 1;j <= m;++j)
			cin >> c[i][j];
	s = n * m + 1; t = n * m + 2;
	for (int i = 1;i <= n;++i)
		for (int j = 1;j <= m;++j)
			if (c[i][j] == '0')
				if (c[i + 1][j] == '2' || c[i][j + 1] == '2' || c[i - 1][j] == '2' || c[i][j - 1] == '2')
					c[i][j] = '1';
	for (int i = 1;i <= n;++i)
		for (int j = 1;j <= m;++j)
			if ((i + j & 1) && c[i][j] == '0') {
				if (c[i + 1][j] == '0') dc.ins(get(i, j), get(i + 1, j), 1);
				if (c[i - 1][j] == '0') dc.ins(get(i, j), get(i - 1, j), 1);
				if (c[i][j + 1] == '0') dc.ins(get(i, j), get(i, j + 1), 1);
				if (c[i][j - 1] == '0') dc.ins(get(i, j), get(i, j - 1), 1);
			}
	for (int i = 1;i <= n;++i)
		for (int j = 1;j <= m;++j)
			if (c[i][j] == '0')
				if (cnt++, i + j & 1) dc.ins(s, get(i, j), 1);
				else dc.ins(get(i, j), t, 1);
			else if (c[i][j] == '2') cnt++;
	cout << cnt - dc.get() << endl;
}
