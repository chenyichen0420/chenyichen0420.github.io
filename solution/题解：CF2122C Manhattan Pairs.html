思路分析

我们思考一下对于 111 维的情况怎么做：排序后首尾配对。

也就是说我们将其按照第一维的值分成了偏大和偏小的两部分。我们画出来贡献来源的分布：

我们注意到，我们在最中间再另加一个点的话，就会发现实际上的贡献是左边的点向中间连，右边的点向中间的连。最后随意组合一下。

类似的能不能扩展到二维呢？不难发现是可以的。

实际上我们取一个点，横轴和纵轴上都是最中间的区间内的话，那么这个点左上，左下，右上，右下的点分别连接到这个点就能够将贡献取到最大。

因为这样子上部的个数与下部的个数相等，左部的个数与右部的个数相等，因此不难推出左上个数与右下的个数相等，两两配对，正好取到与中间点的距离和。右上与左下同理。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
struct pt {
	int x, id;
	inline bool operator<(const pt& r) { return x < r.x; }
}px[200005], py[200005];
int t, n, ic[200005];
vector<int>p[4];
signed main() {
	ios::sync_with_stdio(0);
	for (cin >> t; t; t--) {
		cin >> n;
		for (int i = 1; i <= n; ++i)
			cin >> px[i].x >> py[i].x,
			px[i].id = py[i].id = i;
		sort(px + 1, px + n + 1);
		sort(py + 1, py + n + 1);
		for (int i = 1; i <= n; ++i) ic[i] = 0;
		for (int i = 1; i <= n / 2; ++i)
			ic[px[i].id] |= 1,
			ic[py[i].id] |= 2;
		for (int i = 0; i != 4; ++i) p[i].clear();
		for (int i = 1; i <= n; ++i)
			p[ic[i]].emplace_back(i);
		for (int i = 0; i != p[0].size(); ++i)
			cout << p[0][i] << " " << p[3][i] << endl;
		for (int i = 0; i != p[1].size(); ++i)
			cout << p[1][i] << " " << p[2][i] << endl;
	}
	return 0;
}
