思路分析

卡卡就过了。

首先，一个很自然的转化是在前缀和相等的位置之间连一条向后的边权为 1 转移边，相邻的位置连一条向后的边权为 0 的转移边。每一次就相当于问 l−1 到 r 的最长路。

上面的只是为了叙述方便，实际上显然没必要从图论的角度去思考这道题。

不过他给了我们一个启示：我们可以考虑维护第一个左端点不小于每一个位置的 1 转移的右端点的最小编号。

于是就变成了每一次从 l−1 不停的往后跳，不跳到 r 后的点最多跳的次数。实际上暴力模拟一下就行了。时间复杂度是 O(nm) 的，考虑优化。

常见的优化思路是维护跳 2j 次之后的位置，复杂度被优化到了 O((n+m)log2​n)，实际上还是过不了，因为空间卡的非常死。

但是注意到目前的时间复杂度其实非常的优秀，可以考虑空间换时间。

注意到对于一个维护了跳 20 到 218 步的表，如果你为了空间复杂度更改一下，只维护跳了 26 到 212 的表，那么你实际需要跳的次数就是 26+6+26 次。

前一个是用 212 步暴力枚举 218 的上界，中间是 26 到 212 的正常倍增，后面是 <26 的边界。

算上求解区间需要一个 nlog2​n 的时间，总的时间复杂度是 O(nlog2​n+12n+27m)，常数比较重要就不忽略了，这样算下来其实不会超时。

代码如下：

#include<bits/stdc++.h>
using namespace std;
constexpr int mxx = 7, dnp = 6;
int n, np[400005][mxx + 1], m, l, r; pair<long long, int>c[400005];
#define sipt //signed-input
#define sopt //signed-output
struct IO {
#define mxsz (1 << 19)
	char buf[mxsz], * p1, * p2;
	char pbuf[mxsz], * pp; bool acc[128];
	IO() : p1(buf), p2(buf), pp(pbuf) {}
	~IO() { fwrite(pbuf, 1, pp - pbuf, stdout); }
	inline char gc() {
		if (p1 == p2) p2 = (p1 = buf) + fread(buf, 1, mxsz, stdin);
		return p1 == p2 ? ' ' : *p1++;
	}
	inline void setacc(const char* c) { while (*c) acc[*c++] = 1; }
	inline char getc() { char c; while (!acc[c = gc()]); return c; }
#ifndef sipt
	inline int read() {
		int r = 0; char c = gc(); while (c < '0' || c>'9') c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return r;
	}
#else
	inline int read() {
		int r = 0; char c = gc(); bool rev = 0;
		while (c < '0' || c>'9') rev |= (c == '-'), c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return rev ? ~r + 1 : r;
	}
#endif
	inline void push(const char& c) {
		if (pp - pbuf == mxsz) fwrite(pbuf, 1, mxsz, stdout), pp = pbuf;
		*pp++ = c;
	}
	inline void write(int x) {
		static char sta[22]; int top = 0;
		do sta[top++] = x % 10, x /= 10; while (x);
		while (top) push(sta[--top] ^ 48);
	}
	inline void write(int x, char opc) {
#ifdef sopt
		if (x < 0) push('-'), x = ~x + 1;
#endif
		write(x), push(opc);
	}
	inline void puts(const char* c) { while (*c) push(*c++); }
} io;
signed main() {
	ios::sync_with_stdio(0);
	memset(np, 0x3f, sizeof np); n = io.read();
	for (int i = 1; i <= n; ++i) c[i].first = c[i - 1].first + io.read(), c[i].second = i;
	sort(c, c + n + 1);
	for (int i = 1; i <= n; ++i)
		if (c[i].first == c[i - 1].first)
			np[c[i - 1].second + 1][0] = c[i].second;
	for (int i = n; i >= 1; i--) np[i][0] = min(np[i][0], np[i + 1][0]);
	for (int j = 1; j <= dnp; ++j)
		for (int i = 1; i <= n; ++i)
			if (np[i][j - 1] <= n) np[i][j] = np[np[i][j - 1] + 1][j - 1];
	for (int i = 1; i <= n; ++i) np[i][1] = np[i][dnp];
	for (int j = 2; j <= dnp; ++j)
		for (int i = 1; i <= n; ++i) np[i][j] = np[0][0];
	for (int j = 2; j <= mxx; ++j)
		for (int i = 1; i <= n; ++i)
			if (np[i][j - 1] <= n) np[i][j] = np[np[i][j - 1] + 1][j - 1];
	for (m = io.read(); m; m--) {
		int a = 0; l = io.read(); r = io.read();
		while (np[l][mxx] <= r) l = np[l][mxx] + 1, a += (1ll << mxx + dnp - 1);
		for (int i = mxx - 1; i >= 1; i--)
			if (np[l][i] <= r) l = np[l][i] + 1, a |= 1ll << i + dnp - 1;
		while (np[l][0] <= r) l = np[l][0] + 1, a++;
		io.write(a, '\n');
	}
}
