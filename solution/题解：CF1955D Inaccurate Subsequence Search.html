思路分析

典型的“滑动窗口”。每两次求值时的不同的部分只有前一次的第一个位置，应当删去；以及后一次的最后一个位置，应当加入。

这时候，我们可以用桶来解决这个问题。约定 ttt 为滑动区间的桶，tbtbtb 为整个 bbb 数组的桶。若干细节：

假设这次滑动要加入的数的下标为 rrr，要删去的为 lll。那么 t[ar]t[a_r]t[ar​] 加一。如果此时的 t[ar]≤tb[ar]t[a_r]\le tb[a_r]t[ar​]≤tb[ar​]，那么可以接着匹配到一个新的，匹配数加一。如果此时的 t[al]≤tb[al]t[a_l]\le tb[a_l]t[al​]≤tb[al​]，那么有一个已匹配的数被删除，匹配数减一。再将 t[al]t[a_l]t[al​] 减一。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
int tt, n, m, k, a[200005], b[200005], t[1000005], tb[1000005], ant, ans, lp, rp;
inline void inc(const int& v) {
	t[a[v]]++; ant += (t[a[v]] <= tb[a[v]]);
}
inline void dec(const int& v) {
	ant -= (t[a[v]] <= tb[a[v]]); t[a[v]]--;
}
int main() {
	ios::sync_with_stdio(0);
	cin >> tt;
	while (tt--) {
		cin >> n >> m >> k;
		ant = ans = 0;
		for (int i = 1; i <= n; ++i) cin >> a[i];
		for (int i = 1; i <= m; ++i) cin >> b[i], tb[b[i]]++;
		for (lp = 1, rp = 0; rp < m;) rp++, inc(rp);
		do {
			ans += ant >= k;
			dec(lp); lp++;
			rp++; inc(rp);
		} while (rp <= n);
		cout << ans << endl;
		for (int i = 1; i <= m; ++i) tb[b[i]]--;
		while (lp <= rp) dec(lp), lp++;
	}
}
