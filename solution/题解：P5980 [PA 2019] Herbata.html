思路分析

我们考虑一下把所有的东西全部拆成 $V=1$ 了之后会发生什么。

此时，初始状态变成了很多个 $l_i$ 杯 $V=1,T=a_i$ 的水。目标状态是很多个 $l_i$ 杯的 $V=1,T=b_i$ 的水。

观察到一个性质：对于两杯水，我们一定可以在温度和不变的前提下将两杯水的温度向他们的平均值变化。

现在我们考虑一下怎么样从初始状态转移到目标状态。

容易想到的方式是将拆开后的水一一对应进行转移。那么这样做为什么是不劣的呢？

我们注意到，虽然对于两杯水 $a,b$，我们可以通过“偷换”的方式使其温度变为 $b,a$，但是对于可重集本身并没有发生任何变化。

也就是说，如果我对目标排序，对初始状态也排序，那么初始状态中前半段匹配到后半段一定有一个本质相同的前半段严格匹配前半段的方案。

递推此过程即可证明一一对应是一种不劣的方案。

我们考虑合法性检验。如果一杯水变热了，说明后面有水给他降温，是谁不重要，总之我们标记我们“欠了”一些热量。

如果有一杯水变冷了，说明我们在给前面的水升温，总之我们标记我们“还了”一些热量。

此时，如果热量都还到正的了，说明这杯水温度过高，热量无法被温度更低的水吸收完。而热量又只能往热量低的地方传递，因此无解。

如果到了最后都还没有还完，说明热量根本不守恒，也是一种无解。

模拟上述过程即可。注意到拆成 $V=1$ 的之后总杯数非常多，于是我们仍然把他们视为一个大段，用两个指针快速跳转即可。代码如下：

#include<bits/stdc++.h>
using namespace std;
#define int long long
struct IO {
#define mxsz (1 << 21)
	char buf[mxsz], * p1, * p2;
	char pbuf[mxsz], * pp; bool acc[128];
	IO() : p1(buf), p2(buf), pp(pbuf) {}
	~IO() { fwrite(pbuf, 1, pp - pbuf, stdout); }
	inline char gc() {
		if (p1 == p2) p2 = (p1 = buf) + fread(buf, 1, mxsz, stdin);
		return p1 == p2 ? ' ' : *p1++;
	}
	inline void setacc(const char* c) { while (*c) acc[*c++] = 1; }
	inline char getc() { char c; while (!acc[c = gc()]); return c; }
#ifndef sipt
	inline int read() {
		int r = 0; char c = gc(); while (c < '0' || c>'9') c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return r;
	}
#else
	inline int read() {
		int r = 0; char c = gc(); bool rev = 0;
		while (c < '0' || c>'9') rev |= (c == '-'), c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return rev ? ~r + 1 : r;
	}
#endif
	inline void push(const char& c) {
		if (pp - pbuf == mxsz) fwrite(pbuf, 1, mxsz, stdout), pp = pbuf;
		*pp++ = c;
	}
	inline void write(int x) {
		static char sta[22]; int top = 0;
		do sta[top++] = x % 10, x /= 10; while (x);
		while (top) push(sta[--top] ^ 48);
	}
	inline void write(int x, char opc) {
#ifdef sopt
		if (x < 0) push('-'), x = ~x + 1;
#endif
		write(x), push(opc);
	}
	inline void puts(const char* c) { while (*c) push(*c++); }
} io;
int t,n,m,a,b,sa,sb,lm; bool cn;
pair<int,int>va[1000005],vb[1000005];
struct enumer{
    int p, c;
    inline int move(int v){
        int ret=0;
        while(v&&c+v>=va[p].second){
            ret+=(va[p].second-c)*va[p].first;
            v-=(va[p].second-c);
            ++p; c=0;
        }
        c+=v; ret+=v*va[p].first;
        return ret;
    }
}ea;
signed main(){
    ios::sync_with_stdio(0);
    for(t=io.read();t;t--){
        n=io.read(); sa=sb=lm=0; cn=1;
        for(int i=1;i<=n;++i)
            m=io.read(),a=io.read(),b=io.read(),
            sa+=m*a,sb+=m*b,
            va[i]=make_pair(a,m),
            vb[i]=make_pair(b,m);
        if(sa!=sb){
            puts("NIE");
            continue;
        }
        sort(va+1,va+n+1);
        sort(vb+1,vb+n+1);
        ea={1,0};
        for(int i=1;i<=n;++i){
            lm+=vb[i].first*vb[i].second;
            lm-=ea.move(vb[i].second);
            if(lm<0) {cn=0;break;}
        }
        puts(cn?"TAK":"NIE");
    }
    return 0;
}
