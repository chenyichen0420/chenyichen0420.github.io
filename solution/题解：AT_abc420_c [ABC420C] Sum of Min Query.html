思路分析

简单的一道题，∀i∈[1,n]\forall i\in[1,n]∀i∈[1,n]，将 min⁡(ai,bi)\min(a_i,b_i)min(ai​,bi​) 视为一个整体计算贡献。

具体的，当修改 ala_lal​ 或 blb_lbl​ 时，先将答案减去 min⁡(al,bl)\min(a_l,b_l)min(al​,bl​)，修改后再将答案加上新的 min⁡(al,bl)\min(a_l,b_l)min(al​,bl​)。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
int n, m, a[200005], b[200005], v[200005], ans; char o;
signed main() {
	ios::sync_with_stdio(0);
	cin >> n >> m;
	for (int i = 1; i <= n; ++i) cin >> a[i];
	for (int i = 1; i <= n; ++i) cin >> b[i];
	for (int i = 1; i <= n; ++i)
		ans += (v[i] = min(a[i], b[i]));
	for (int i = 1, l, r; i <= m; ++i)
		if (cin >> o >> l >> r, o == 'A')
			a[l] = r, ans -= v[l], ans += (v[l] = min(a[l], b[l])),
			cout << ans << endl;
		else
			b[l] = r, ans -= v[l], ans += (v[l] = min(a[l], b[l])),
			cout << ans << endl;
}
