思路分析

首先并查集生成出来一个生成森林，非树边放到一边备用。

接下来找出剩下多少个连通块，存到 set 中方便操作。枚举非树边，然后从 set 中任意取出一个与这条边不同集合的集合，重定向一下这条边，将两个集合合并。

代码如下：

#include<bits/stdc++.h>
using namespace std;
#define int long long
struct node {
	int l, r, id;
	node(int li = 0, int ri = 0, int ii = 0) :l(li), r(ri), id(ii) {}
};
int n, m, f[200005]; vector<node>ps, np;
vector<int>ans; set<int>hs; int pto;
inline int find(int p) {
	return p != f[p] ? f[p] = find(f[p]) : p;
}
inline bool merge(int l, int r) {
	l = find(l); r = find(r);
	if (l == r) return 1;
	f[l] = r; return 0;
}
vector<int>pp;
signed main() {
	ios::sync_with_stdio(0);
	cin >> n >> m;
	for (int i = 1, a, b; i <= m; ++i)
		cin >> a >> b, ps.emplace_back(a, b, i);
	for (int i = 1; i <= n; ++i) f[i] = i;
	for (int i = 0; i != m; ++i)
		if (merge(ps[i].l, ps[i].r))
			np.emplace_back(ps[i]);
	for (int i = 1; i <= n; ++i) hs.emplace(find(i));
	for (const node& tmp : np) {
		if (hs.size() == 1) break;
		int bt = find(tmp.l);
		for (int ps : hs) {
			if (ps != bt) {
				ans.emplace_back(tmp.id);
				ans.emplace_back(tmp.l);
				ans.emplace_back(ps);
				merge(ps, bt);
				hs.erase(ps); break;
			}
		}
	}
	cout << ans.size() / 3;
	for (int i = 0; i != ans.size(); ++i)
		if (i % 3 == 0)cout << endl << ans[i];
		else cout << " " << ans[i];
}
