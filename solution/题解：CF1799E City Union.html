思路分析

大致分为以下两步：

  1. 将框住两个连通块的最小矩形连起来；
  2. 填充凹陷。

再补充说明一下，第一步需要连起来，当且仅当两个矩形在横纵坐标对应的区间都不相交。

完全不相交的时候，我们直接在两个矩形最接近的点之间一横一竖连起来就完事了。最省填充数量。

显然这样子花费在第一步的新填充数是最少的，考虑第二步，也就是哪些是一定要填充的。

我们发现如果一个点的上面和下面都有点的话，那为了保证凸性，就不得不填上了。左右同理。同时我们也发现第一部的操作方法正好可以不在第二步浪费格子。

但是只做一次是有问题的，我们可能有三个点在 $(1,1),(3,1),(2,3)$，那么因枚举顺序不同可能无法填上 $(2,2)$。不过因为是针对连通块在进行填充，所以第一次一定会填完行或者列，只需要填两次就行了。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
struct pos { int x, y; }ep[4];
int t, n, m, f; char mp[55][55]; vector<pos>p[2];
constexpr int dx[] = { 0,0,1,-1 }, dy[] = { -1,1,0,0 };
inline void dfs(int x, int y) {
	p[f].push_back({ x,y }); mp[x][y] = '.';
	for (int i = 0; i != 4; ++i)
		if (mp[x + dx[i]][y + dy[i]] == '#')
			dfs(x + dx[i], y + dy[i]);
}
signed main() {
	ios::sync_with_stdio(0);
	for (cin >> t; t; t--) {
		cin >> n >> m; f = 0;
		memset(mp, 0, sizeof mp);
		p[0].clear(); p[1].clear();
		for (int i = 1; i <= n; ++i)
			for (int j = 1; j <= m; ++j)
				cin >> mp[i][j];
		for (int i = 1; i <= n; ++i)
			for (int j = 1; j <= m; ++j)
				if (mp[i][j] == '#')
					dfs(i, j), f++;
		ep[0] = ep[2] = { 80,80 };
		ep[1] = ep[3] = { 0,0 };
		for (const auto& v : p[0])
			ep[0].x = min(ep[0].x, v.x),
			ep[0].y = min(ep[0].y, v.y),
			ep[1].x = max(ep[1].x, v.x),
			ep[1].y = max(ep[1].y, v.y),
			mp[v.x][v.y] = '#';
		for (const auto& v : p[1])
			ep[2].x = min(ep[2].x, v.x),
			ep[2].y = min(ep[2].y, v.y),
			ep[3].x = max(ep[3].x, v.x),
			ep[3].y = max(ep[3].y, v.y),
			mp[v.x][v.y] = '#';
		if (ep[1].x < ep[2].x && ep[1].y < ep[2].y) {
			for (int i = ep[1].x + 1; i <= ep[2].x; ++i) mp[i][ep[1].y] = '#';
			for (int i = ep[1].y + 1; i != ep[2].y; ++i) mp[ep[2].x][i] = '#';
		}
		if (ep[0].x > ep[3].x && ep[1].y < ep[2].y) {
			for (int i = ep[0].x - 1; i >= ep[3].x; --i) mp[i][ep[1].y] = '#';
			for (int i = ep[1].y + 1; i != ep[2].y; ++i) mp[ep[3].x][i] = '#';
		}
		if (ep[0].x > ep[3].x && ep[0].y > ep[3].y) {
			for (int i = ep[3].x + 1; i <= ep[0].x; ++i) mp[i][ep[3].y] = '#';
			for (int i = ep[3].y + 1; i != ep[0].y; ++i) mp[ep[0].x][i] = '#';
		}
		if (ep[1].x < ep[2].x && ep[0].y > ep[3].y) {
			for (int i = ep[2].x - 1; i >= ep[1].x; --i) mp[i][ep[3].y] = '#';
			for (int i = ep[3].y + 1; i != ep[0].y; ++i) mp[ep[1].x][i] = '#';
		}
		for (int k = 1; k <= 2; ++k) {
			for (int i = 1; i <= n; ++i) {
				int up = 0, dp = 0;
				for (int j = 1; j <= m; ++j) if (mp[i][j] == '#') dp = j;
				for (int j = m; j >= 1; --j) if (mp[i][j] == '#') up = j;
				if (!up) continue;
				for (int j = up; j <= dp; ++j) mp[i][j] = '#';
			}
			for (int j = 1; j <= m; ++j) {
				int up = 0, dp = 0;
				for (int i = 1; i <= n; ++i) if (mp[i][j] == '#') dp = i;
				for (int i = n; i >= 1; i--) if (mp[i][j] == '#') up = i;
				if (!up) continue;
				for (int i = up; i <= dp; ++i) mp[i][j] = '#';
			}
		}
		for (int i = 1; i <= n; ++i, cout << endl)
			for (int j = 1; j <= m; ++j)
				cout << mp[i][j];
		cout << endl;
	}
}
