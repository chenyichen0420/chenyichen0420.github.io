就是要输出方案的最小树形图，跑一下 tarjan 版的就完了。

板子本身不再赘述，主要讲一下怎么记录路径：

在 tarjan 的伸展操作中，我们会将每一坨环给强制拆开。而拆开时的边就是我们选中的边。

所以你只需要记录一下每一条边的编号以及本来就记了的初始值，初始值非 0 再放到要输出的集合里面。

代码如下：

#include<bits/stdc++.h>
using namespace std;
#define int long long
struct IO {
#define mxsz (1 << 21)
	char buf[mxsz], * p1, * p2;
	char pbuf[mxsz], * pp;
	IO() : p1(buf), p2(buf), pp(pbuf) {}
	~IO() { fwrite(pbuf, 1, pp - pbuf, stdout); }
	inline char gc() {
		if (p1 == p2) p2 = (p1 = buf) + fread(buf, 1, mxsz, stdin);
		return p1 == p2 ? ' ' : *p1++;
	}
	inline int read() {
		int r = 0; char c = gc(); while (c < '0' || c>'9') c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return r;
	}
	inline void push(const char& c) {
		if (pp - pbuf == mxsz) fwrite(pbuf, 1, mxsz, stdout), pp = pbuf;
		*pp++ = c;
	}
	inline void write(int x) {
		static char sta[22]; int top = 0;
		do sta[top++] = x % 10, x /= 10; while (x);
		while (top) push(sta[--top] ^ 48);
	}
	inline void write(int x, char opc) {
		write(x), push(opc);
	}
} io;
struct edge { int l, r, v, orv, id; };vector<edge>son[100005];
struct mer_ser_set {
	int f[200005];
	inline int operator[](int p) {
		return f[p] ? f[p] = (*this)[f[p]] : p;
	}
}mss; //用来维护在哪坨环里
struct node {
	edge e; int tg; node* ls, * rs;
	node(const edge& v) :e(v), tg(0) { ls = rs = nullptr; }
	inline void pud() {
		if (ls) ls->tg += tg;
		if (rs) rs->tg += tg;
		e.v += tg; tg = 0;
	}
	//em, 指针版的一个好处就是调用左右子树的代码简短很多
}; using hptr = node*;
inline hptr merge(hptr l, hptr r) {
	if (!l) return r; if (!r) return l;
	if (l->e.v + l->tg > r->e.v + r->tg) swap(l, r);
	l->pud(); l->rs = merge(l->rs, r);
	swap(l->ls, l->rs); return l;
} //典中典的合并
inline edge top(node*& ep) {
	edge tmp = ep->e;
	ep = merge(ep->ls, ep->rs);
	return tmp;
} //获取堆顶并删除
hptr hp[200005]; edge lfp[200005];
int n, m, rt, lp[200005], rinxtp[200005];
inline void adde(int l, int r, int v, int id) {
	son[r].emplace_back(edge{ l,r,v,v,id });
}
bool vis[200005]; queue<hptr>q; vector<int>ap;
inline void merge() {
	for (int i = 1;i <= n;++i) {
		for (const edge& e : son[i]) q.emplace(new node(e));
		while (q.size() > 1) {
			hptr p1 = q.front(); q.pop();
			hptr p2 = q.front(); q.pop();
			q.emplace(merge(p1, p2));
		} //大小期望为 1,1,...,1,2,2,...,2,4,4,...,4,...
		hp[i] = q.front(); q.pop(); //其实是线性建堆
	}
	vis[1] = 1; //“随机”选了一个点，100% 选中 1 点
	for (int p = 1, np = 1;hp[p];vis[np = p] = 1) {
		do lfp[p] = top(hp[p]), p = mss[lfp[p].l];
		while (p == np && hp[p]);
		if (p == np) break;
		if (!vis[p]) continue;
		int cnp = ++n;
		for (p = np; p != n;p = cnp) {
			mss.f[p] = lp[p] = n;
			if (hp[p]) hp[p]->tg -= lfp[p].v;
			hp[n] = merge(hp[n], hp[p]);
			cnp = mss[lfp[p].l];
			rinxtp[cnp == n ? np : cnp] = p;
		}
	}
}
inline int expand(int rt, int ep) {
	int ret = 0;
	for (;rt != ep;rt = lp[rt]) {
		for (int sp = rinxtp[rt];sp != rt;sp = rinxtp[sp])
			if (lfp[sp].orv >= 1e16) return 1e16;
			else {
				ret += expand(lfp[sp].r, sp) + lfp[sp].orv;
				if (lfp[sp].orv) ap.emplace_back(lfp[sp].id);
			}
		if (ret >= 1e16) return 1e16;
	}
	return ret;
}
inline int dmst() {
	merge(); int ret = expand(rt, n);
	if (ret >= 1e16) ret = -1; return ret;
}
signed main() {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	ios::sync_with_stdio(0);
	n = io.read(); m = io.read(); rt = 1;
	for (int i = 1, l, r, v;i <= m;++i) {
		l = io.read(), r = io.read(), v = io.read();
		if (l != r) adde(l, r, v, i);
	}
	for (int i = 1;i != n;++i) adde(i, i + 1, 1e16, 0);
	adde(n, 1, 1e16, 0); cout << dmst() << endl;
	if (ap.size()) for (int i : ap) cout << i << " ";
}
