#include<bits/stdc++.h>
using namespace std;
#define int long long
int t, v, l[1119235], r[1119235], v2, lm; set<int>s;
signed main() {
    ios::sync_with_stdio(0);
    for (int i = 128; i <= 1e5; ++i) {
        int t = ceil(8744.0 / (i / 128));
        if (!l[t]) l[t] = i; r[t] = i;
        s.emplace(t);
    }
    for (int i : s) cerr << l[i] << " " << r[i] << endl;
    for (cin >> t; t; t--) {
        cout << "? 8744 ";
        for (int i = 1; i <= 8744; ++i) cout << "128 ";
        cout << endl; cin >> v;
        if (v) {
            assert((r[v] - l[v] + 1) * 2 + 8744 <= 2.5e4);
            cout << "? " << (r[v] - l[v] + 1) * 2 << ' ';
            for (int i = l[v]; i <= r[v]; ++i)
                cout << l[v] << " " << i - l[v] + 1 << " ";
            cout << endl; cin >> v2; lm = r[v] - l[v] + 1;
            cout << "! " << r[v] - (v2 - lm - 1) << endl;
        }
        else {
            assert(127 * 128 + 8744 <= 2.5e4);
            cout << "? " << 127 * 128 << " ";
            for (int i = 1; i <= 127 * 128; ++i) cout << "1 ";
            cout << endl; cin >> v2; bool hs = 0;
            for (int i = 1; i <= 127; ++i)
                if (ceil(127.0 * 128 / i) == v2) {
                    cout << "! " << i << endl;
                    hs = 1; break;
                }
            assert(hs);
        }
    }
}
