思路分析

你可能做过一道题叫做独木桥。这里面有一种等价思路就是这类碰撞之后从“高空”向下看会觉得两个人擦肩而过。

因此如果所有人从边界上同时出发，并且路线相交的一定碰撞，那么最后所有终点就是起始点对面的那些点。

这启示我们分批处理那些会碰撞的点。显然我们可以将 t 时间后开始移动变为先后退 t 步然后继续。

会碰到的点满足 x+y 的值相等，正确性比较显然，因为所有点的移动本质上就是将 x+y 增加 1。

因为必定会碰撞，所以说开始时在上面的就一定在上面。也就是这些人是斜向上方一一对应的。

于是分批处理一下，此题可解。代码如下：

#include<bits/stdc++.h>
using namespace std;
struct vecor {
	int x, y, id; vecor(double x = 0, double y = 0) :x(x), y(y), id(0) {}
	friend istream& operator>>(istream& l, vecor& r) { return l >> r.x >> r.y; }
	friend ostream& operator<<(ostream& l, const vecor& r) { return l << r.x << ' ' << r.y; }
	inline vecor operator+(const vecor& r) const { return vecor(x + r.x, y + r.y); }
	inline vecor operator-(const vecor& r) const { return vecor(x - r.x, y - r.y); }
	inline vecor operator*(double r) const { return vecor(x * r, y * r); }
	inline vecor operator/(double r) const { return vecor(x / r, y / r); }
	inline double operator*(const vecor& r) const { return x * r.y - y * r.x; }
	inline double operator^(const vecor& r) const { return x * r.x + y * r.y; }
	inline double abs() const { return sqrt(x * x + y * y); }
	inline double sabs() const { return x * x + y * y; }
	inline double arc() const { return atan2(y, x); }
	inline bool operator<(const vecor& r) const { return x + y < r.x + r.y; }
	inline bool operator==(const vecor& r) const { return x + y == r.x + r.y; }
	inline void zig() { swap(x, y); x = -x; }
	inline void zag() { swap(x, y); y = -y; }
	inline void resize(double s) { double t = s / abs(); x *= t; y *= t; }
}p[100005], ap[100005];
int n, x, y, o, a, t, as[100005];
signed main() {
	ios::sync_with_stdio(0);
	cin >> n >> x >> y; p[n + 1].x = 3e5;
	for (int i = 1; i <= n; ++i)
		if (cin >> o >> a >> t, o & 1)
			p[i] = vecor(a, -t), p[i].id = i;
		else p[i] = vecor(-t, a), p[i].id = i;
	sort(p + 1, p + n + 1);
	for (int l = 1, r = 0; l <= n; l = r + 1) {
		while (p[r + 1] == p[l]) r++;
		sort(p + l, p + r + 1, [&](const vecor& l, const vecor& r) {return l.x < r.x; });
		for (int i = l; i <= r; ++i)
			if (p[i].y > 0) ap[i] = vecor(x, p[i].y);
			else ap[i] = vecor(p[i].x, y);
		sort(ap + l, ap + r + 1, [&](const vecor& l, const vecor& r) {
			return l.x != r.x ? l.x<r.x : l.y>r.y; });
		for (int i = l; i <= r; ++i) as[p[i].id] = i;
	}
	for (int i = 1; i <= n; ++i) cout << ap[as[i]] << endl;
}
