思路分析

不识庐山真面目，只缘身在此山中。

我们可以向上跳 2 次 109，完全跳出这堆点。

随后我们向右移动 2 次 109，不难发现我们一定在所有点的右上方。

我们注意到一个性质：所有的点按照 x+y 的值排序之后，最大的那些点能控制得范围是他们的右上角的区间。这里的控制是指到他们的距离不比到其他任意点的距离长。

我们已经移动到了这些点的控制范围，因此我们可以知道当前点的 x+y 的数值。

类似的，我们再将它向左移动 4 次 109，就到了左上角。此时同理可得当前点的 y−x 的数值。

然后就可以解方程，这道题就做完了。代码如下：

#include<bits/stdc++.h>
using namespace std;
#define int long long
int n, x, y, ms, mc, t, v, cs, cc;
signed main() {
	ios::sync_with_stdio(0);
	for (cin >> t; t; t--) {
		cin >> n; ms = mc = -2e9;
		for (int i = 1; i <= n; ++i) {
			cin >> x >> y;
			ms = max(ms, x + y);
			mc = max(mc, y - x);
		}
		cout << "? U 1000000000" << endl; cin >> v;
		cout << "? U 1000000000" << endl; cin >> v;
		cout << "? R 1000000000" << endl; cin >> v;
		cout << "? R 1000000000" << endl; cin >> v;
		cs = v + ms;
		cout << "? L 1000000000" << endl; cin >> v;
		cout << "? L 1000000000" << endl; cin >> v;
		cout << "? L 1000000000" << endl; cin >> v;
		cout << "? L 1000000000" << endl; cin >> v;
		cc = v + mc;
		/*
		y-x = cc
		y+x+4e9 =cs
		*/
		int cy = (cc + cs - 4000000000) / 2;
		int cx = cy - cc;
		cy -= 2000000000;
		cx += 2000000000;
		assert(cx <= 1000000000 && cx >= -1000000000);
		assert(cy <= 1000000000 && cy >= -1000000000);
		cout << "! " << cx << " " << cy << endl;
	}
}
