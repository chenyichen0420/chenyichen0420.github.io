思路分析

总共只有两条直线，所以如果有三个不共线的点，那么一定有一条直线穿过其中的某两个点。

因此，我们先找到三个不共线的点，然后依次尝试让一条直线穿过其中的某两个点，并判断剩余的点在不在同一条直线上即可。

代码如下：

#include<bits/stdc++.h>
using namespace std;
#define double long double
constexpr double eps = 1e-9;
inline double sqr(double v) { return v * v; }
struct vecor {
	double x, y; vecor(double x = 0, double y = 0) :x(x), y(y) {}
	friend istream& operator>>(istream& l, vecor& r) { return l >> r.x >> r.y; }
	friend ostream& operator<<(ostream& l, const vecor& r) { return l << r.x << ' ' << r.y; }
	inline vecor operator+(const vecor& r) const { return vecor(x + r.x, y + r.y); }
	inline vecor operator-(const vecor& r) const { return vecor(x - r.x, y - r.y); }
	inline vecor operator*(double r) const { return vecor(x * r, y * r); }
	inline vecor operator/(double r) const { return vecor(x / r, y / r); }
	inline double operator*(const vecor& r) const { return x * r.y - y * r.x; }
	inline double operator^(const vecor& r) const { return x * r.x + y * r.y; }
	inline double abs() const { return sqrt(x * x + y * y); }
	inline double sabs() const { return x * x + y * y; }
	inline double arc() const { return atan2(y, x); }
	inline bool operator<(const vecor& r) const { return x != r.x ? x < r.x : y < r.y; }
	inline bool operator==(const vecor& r) const { return x == r.x && y == r.y; }
	inline void zig() { swap(x, y); x = -x; }
	inline void zag() { swap(x, y); y = -y; }
	inline void resize(double s) { double t = s / abs(); x *= t; y *= t; }
}p[100005], p1, p2, p3;
struct line {
	vecor s, e; line() :s(vecor()), e(vecor()) {};
	line(const vecor& s, const vecor& e) :s(s), e(e) {};
	inline bool online(const vecor& p) const { return abs((e - s) * (p - s)) <= eps; }
}l;
int n; bool hs; vector<vecor>s;
signed main() {
	ios::sync_with_stdio(0); cin >> n;
	if (n <= 4) return cout << "YES\n", 0;
	for (int i = 1; i <= n; ++i) cin >> p[i];
	p1 = p[1]; p2 = p[2]; l = line(p[1], p[2]);
	for (int i = 3; i <= n; ++i)
		if (!l.online(p[i])) {
			p3 = p[i]; hs = 1; break;
		}
	if (!hs) return cout << "YES\n", 0;
	for (int i = 1; i <= n; ++i)
		if (!l.online(p[i])) s.emplace_back(p[i]);
	if (s.size() <= 2) return cout << "YES\n", 0;
	l = line(s[0], s[1]); hs = 0;
	for (int i = 2; i != s.size(); ++i)
		if (!l.online(s[i])) { hs = 1; break; }
	if (!hs) return cout << "YES\n", 0;
	s.clear(); l = line(p[1], p[3]);
	for (int i = 1; i <= n; ++i)
		if (!l.online(p[i])) s.emplace_back(p[i]);
	if (s.size() <= 2) return cout << "YES\n", 0;
	l = line(s[0], s[1]); hs = 0;
	for (int i = 2; i != s.size(); ++i)
		if (!l.online(s[i])) { hs = 1; break; }
	if (!hs) return cout << "YES\n", 0;
	s.clear(); l = line(p[2], p[3]);
	for (int i = 1; i <= n; ++i)
		if (!l.online(p[i])) s.emplace_back(p[i]);
	if (s.size() <= 2) return cout << "YES\n", 0;
	l = line(s[0], s[1]); hs = 0;
	for (int i = 2; i != s.size(); ++i)
		if (!l.online(s[i])) { hs = 1; break; }
	if (!hs) return cout << "YES\n", 0;
	cout << "NO\n";
}
