思路分析

很简单的搜索题/最短路题。

我们可以直接记录出当先所在位置的横纵坐标以及当前是开状态还是关状态。

初时时是关状态，可以走 o，开状态下可以走到 x，走到开关就切换状态即可。简单标记一下用广搜解决。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
int dx[] = { 0,0,1,-1 }, dy[] = { 1,-1,0,0 };
struct node {
	int x, y, t, d;
	node(int x = 0, int y = 0, int t = 0, int d = 0) :x(x), y(y), d(d), t(t) {};
}tmp; queue<node>q; bool vis[505][505][2];
int n, m, sx, sy, ex, ey; char mp[505][505];
inline bool cn(int x, int y, int t) {
	if (x<1 || x>n || y<1 || y>m) return 0;
	if (!t && mp[x][y] == 'x') return 0;
	if (t && mp[x][y] == 'o') return 0;
	return mp[x][y] != '#';
}
signed main() {
	ios::sync_with_stdio(0);
	cin >> n >> m;
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= m; ++j)
			cin >> mp[i][j];
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= m; ++j)
			if (mp[i][j] == 'S') sx = i, sy = j;
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= m; ++j)
			if (mp[i][j] == 'G') ex = i, ey = j;
	q.emplace(sx, sy, 0, 0);
	while (q.size()) {
		tmp = q.front(); q.pop();
		if (mp[tmp.x][tmp.y] == '?') tmp.t ^= 1;
		if (vis[tmp.x][tmp.y][tmp.t]) continue;
		vis[tmp.x][tmp.y][tmp.t] = 1;
		if (tmp.x == ex && tmp.y == ey)
			return cout << tmp.d << endl, 0;
		for (int i = 0; i != 4; ++i)
			if (cn(tmp.x + dx[i], tmp.y + dy[i], tmp.t))
				q.emplace(tmp.x + dx[i], tmp.y + dy[i], tmp.t, tmp.d + 1);
	}
	cout << "-1\n";
}
