思路分析

题意有一点点复杂，其实就是让你判断一个图形是不是边数为偶数的中心对称的图形。

那就做完了。先判断是不是偶数边，然后对点求出中点并判断是否重合，重合就是，不重合就不是。

代码如下：

#include<iostream>
#include<assert.h>
#include<algorithm>
#include<cmath>
#include<map>
#include<ratio>
using namespace std;
#define int long long
#define double long double
template<typename t = double>
struct vecor {
	t x, y;
	vecor(t x = 0, t y = 0) :x(x), y(y) {}
	friend istream& operator>>(istream& l, vecor<t>& r) {
		return l >> r.x >> r.y;
	}
	friend ostream& operator<<(ostream& l, const vecor<t>& r) {
		return l << r.x << " " << r.y << " ";
	}

	inline vecor operator+(const vecor<t>& r) const {
		return vecor(x + r.x, y + r.y);
	}
	inline vecor operator-(const vecor<t>& r) const {
		return vecor(x - r.x, y - r.y);
	}
	inline t operator^(const vecor<t>& r) const {
		return x * r.x + y * r.y;
	}
	inline t operator*(const vecor<t>& r) const {
		return x * r.y - y * r.x;
	}
	inline bool operator!=(const vecor<t>& r) const {
		return x != r.x || y != r.y;
	}
	inline bool operator==(const vecor<t>& r) const{
		return x == r.x && y == r.y;
	}

	inline void resize(t sz) {
		double k = sz / sqrt(x * x + y * y);
		x *= k; y *= k;
	}
	inline void zig() { swap(x, y); x = -x; }
	inline void zag() { swap(x, y); y = -y; }

	friend bool operator<(const vecor<t>& l, const vecor<t>& r) {
		return l.x != r.x ? l.x < r.x : l.y < r.y;
	}
};

inline double sqr(double v) { return v * v; }
template<typename t>
inline double odis(const vecor<t>& l, const vecor<t>& r) {
	return sqrt(sqr(l.x - r.x) + sqr(l.y - r.y));
}
template<typename t>
inline t mdis(const vecor<t>& l, const vecor<t>& r) {
	return abs(l.x - r.x) + abs(l.y - r.y);
}
template<typename t>
inline double s(const vecor<t>& l, const vecor<t>& r, const vecor<t>& p) {
	return abs((p - l) * (p - r) / 2);
}

template<typename t>
inline bool online(const vecor<t>& l, const vecor<t>& r, const vecor<t>& p) {
	if (abs((p - l) * (p - r))) return 0;
	if (p.x < l.x && p.x < r.x) return 0;
	if (p.y < l.y && p.y < r.y) return 0;
	if (p.x > l.x && p.x > r.x) return 0;
	if (p.y > l.y && p.y > r.y) return 0;
	return 1;
}
template<typename t>
inline vecor<t>mid(const vecor<t>& l, const vecor<t>& r) {
	return vecor<t>((l.x + r.x) / 2, (l.y + r.y) / 2);
}

int n; vecor<>p[100005], mp;
#ifdef _MSC_VER
#define __gcd _Gcd
#endif
signed main() {
	ios::sync_with_stdio(0); cin >> n;
	if (n & 1) return cout << "no\n", 0;
	for (int i = 1; i <= n; ++i) cin >> p[i];
	mp = mid(p[n / 2], p[n]);
	for (int i = 1; i != n / 2; ++i)
		if (mid(p[i], p[i + n / 2]) != mp)
			return cout << "no\n", 0;
	cout << "yes\n";
}
