思路分析

我爱暴力。

一个贪心思路：我们将从第一个门进的那些人按照接受的距离从小到大排序后安排在离第一个门尽可能远的前提下离第二个门也尽可能远的位置。

这个贪心显然是正确的，因为这样缩小了第二个门所要求的更大的那部分的距离。

那怎么处理这坨东西呢？我们注意到 n×m≤104，所以直接对于每一个点暴力在矩阵里面扫一遍，取我们想要的点就完了。复杂度 O(n2×m2)，随便过，实测 0.6s 上下。

代码如下：

#include<bits/stdc++.h>
using namespace std;
int n, m, v[10005], t; bitset<10008>vis[10008];
inline int dis(int x0, int y0, int x1, int y1) {
	return abs(x0 - x1) + abs(y0 - y1);
}
signed main() {
	ios::sync_with_stdio(0); cin >> n >> m >> t;
	for (int i = 1;i <= t;++i) cin >> v[i];
	sort(v + 1, v + t + 1);
	for (int i = 1;i <= t;++i) {
		int mx, my, md; mx = my = md = 0;
		for (int j = 1;j <= n;++j)
			for (int k = 1;k <= m;++k)
				if (dis(j, k, 0, 0) > v[i]) break;
				else if (vis[j][k]) continue;
				else if (dis(j, k, 0, m + 1) > md)
					md = dis(j, k, 0, m + 1), vis[mx][my] = 0,
					mx = j, my = k, vis[j][k] = 1;
		if (!mx || !my) return puts("NO"), 0;
	}
	cin >> t;
	for (int i = 1;i <= t;++i) {
		cin >> v[i];
		int mx, my, md; mx = my = md = 0;
		for (int j = 1;j <= n;++j)
			for (int k = m;k >= 1;--k)
				if (dis(j, k, 0, m + 1) > v[i]) break;
				else if (vis[j][k]) continue;
				else if (dis(j, k, 0, m + 1) > md)
					md = dis(j, k, 0, m + 1), vis[mx][my] = 0,
					mx = j, my = k, vis[j][k] = 1;
		if (!mx || !my) return puts("NO"), 0;
	}
	puts("YES");
}
