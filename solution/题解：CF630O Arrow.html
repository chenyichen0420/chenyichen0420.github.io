思路分析

直接用点和向量暴力模拟就完了啊！

我们需要用到的：

  1. 将一个向量转一个直角
  2. 将一个向量的长度改变

第一个操作对应代码里面的 zig,zag，比较简单，就是换一下横纵坐标，然后取个反，不讲了。

第二个操作先除以本身的模，得到单位方向向量，再乘长度。

代码如下：

#include<iostream>
#include<assert.h>
#include<algorithm>
#include<cmath>
using namespace std;
#define int long long
#define double long double
template<typename t = double>
struct vecor {
	t x, y;
	vecor(t x = 0, t y = 0) :x(x), y(y) {}
	friend istream& operator>>(istream& l, vecor<t>& r) {
		return l >> r.x >> r.y;
	}
	friend ostream& operator<<(ostream& l, const vecor<t>& r) {
		return l << r.x << " " << r.y << " ";
	}

	inline vecor operator+(const vecor<t>& r) const {
		return vecor(x + r.x, y + r.y);
	}
	inline vecor operator-(const vecor<t>& r) const {
		return vecor(x - r.x, y - r.y);
	}
	inline t operator^(const vecor<t>& r) const {
		return x * r.x + y * r.y;
	}
	inline t operator*(const vecor<t>& r) const {
		return x * r.y - y * r.x;
	}
	inline bool operator==(const vecor<t>& r) const{
		return x == r.x && y == r.y;
	}

	inline void resize(t sz) {
		double k = sz / sqrt(x * x + y * y);
		x *= k; y *= k;
	}
	inline void zig() { swap(x, y); x = -x; }
	inline void zag() { swap(x, y); y = -y; }

	inline void print() {
		printf("%.12Lf %.12Lf\n", x, y);
	}
};
template<typename t>
inline t sqr(t v) { return v * v; }
template<typename t>
inline t odis(const vecor<t>& l, const vecor<t>& r) {
	return sqrt(sqr(l.x - r.x) + sqr(l.y - r.y));
}
template<typename t>
inline t mdis(const vecor<t>& l, const vecor<t>& r) {
	return abs(l.x - r.x) + abs(l.y - r.y);
}
template<typename t>
inline bool online(const vecor<t>& l, const vecor<t>& r, const vecor<t>& p) {
	if (abs((p - l) * (p - r))) return 0;
	if (p.x < l.x && p.x < r.x) return 0;
	if (p.y < l.y && p.y < r.y) return 0;
	if (p.x > l.x && p.x > r.x) return 0;
	if (p.y > l.y && p.y > r.y) return 0;
	return 1;
}
template<typename t>
inline t s(const vecor<t>& l, const vecor<t>& r, const vecor<t>& p) {
	return abs((p - l) * (p - r) / 2);
}
double a, b, c, d; vecor<>p, v, x, t;
signed main() {
	ios::sync_with_stdio(0);
	cin >> p >> x >> a >> b >> c >> d;
	v = x; v.resize(b);
	(p + v).print();
	v.zig();
	v.resize(a / 2);
	(p + v).print();
	v.resize(c / 2);
	(p + v).print();
	t = x; t.zig(), t.zig();
	t.resize(d);
	(p + v + t).print();
	v.zig(), v.zig();
	(p + v + t).print();
	(p + v).print();
	v.resize(a / 2);
	(p + v).print();
}
