思路分析

给你一只 FWT，希望你能水掉这道题。

我们容易注意到，这道题看起来非常像是三部分缝在一起了。具体来说，$(a,b)$ 是一部分，$(d,e)$ 是一部分，整体是一部分。

下面钦定 $c_s$ 表示 $s$ 在输入中出现的次数，$f_s$ 是斐波那契数列的第 $i$ 项的值。

首先考虑能不能把 $a,b$ 给合并掉。也就是我们要求出所有 $s$ 对应的 $\sum_{a|b=s,a\&b=0}f_sc_ac_b$，显然 $f_s$ 是常数项，可以暂时抛开，变成了 $\sum_{a|b=s,a\&b=0}c_ac_b$。这是子集卷积模板题。我们记这一问的答案是 $A$。

然后考虑能不能把 $d,e$ 给合并掉。也就是我们要求出所有的 $s$ 对应的 $\sum_{a\bigoplus b=s}c_ac_b$，其中 $\bigoplus$ 表示异或。这是 FWT 模板题第三问。我们记这一问的答案是 $C$。

中间的 $c$ 显然就是 $B_s=c_sf_s$。这是唐题。

最后我们把 $A,B,C$ 进行与卷积就行了。这是 FWT 模板题第二问。

此缝合题可解，代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
constexpr int mod=1e9+7,inv2=(mod+1)/2;
int n,cnt[1ll<<17],a[1ll<<17],cv[18][1ll<<17],ac[18][1ll<<17],tc[1ll<<17],fi[1000005],ans;
inline void mad(int& l,const int r){(l+=r)>=mod&&(l-=mod);}
inline void mdc(int& l,const int r){(l-=r)<0&&(l+=mod);}
namespace fwtand{
	inline void fwt(int *v){
		for(int i=1;i!=1ll<<17;i<<=1)
			for(int j=0;j!=1ll<<17;j+=i<<1)
				for(int k=0;k!=i;++k)
					mad(v[j+k],v[j+k+i]);
	}
	inline void ifwt(int *v){
		for(int i=1;i!=1ll<<17;i<<=1)
			for(int j=0;j!=1ll<<17;j+=i<<1)
				for(int k=0;k!=i;++k)
					mdc(v[j+k],v[j+k+i]);
	}
}
namespace fwtor{
	inline void fwt(int *v){
		for(int i=1;i!=1ll<<17;i<<=1)
			for(int j=0;j!=1ll<<17;j+=i<<1)
				for(int k=0;k!=i;++k)
					mad(v[j+k+i],v[j+k]);
	}
	inline void ifwt(int *v){
		for(int i=1;i!=1ll<<17;i<<=1)
			for(int j=0;j!=1ll<<17;j+=i<<1)
				for(int k=0;k!=i;++k)
					mdc(v[j+k+i],v[j+k]);
	}
}
namespace fwtxor{
	inline void fwt(int *v){
		for(int i=1;i!=1ll<<17;i<<=1)
			for(int j=0;j!=1ll<<17;j+=i<<1)
				for(int k=0;k!=i;++k){
					int a=v[j+k],b=v[j+k+i];
					v[j+k]=a+b; v[j+k+i]=a-b;
					(v[j+k]>=mod)&&(v[j+k]-=mod);
					(v[j+k+i]<0)&&(v[j+k+i]+=mod);
				}
	}
	inline void ifwt(int *v){
		for(int i=1;i!=1ll<<17;i<<=1)
			for(int j=0;j!=1ll<<17;j+=i<<1)
				for(int k=0;k!=i;++k){
					int a=v[j+k],b=v[j+k+i];
					v[j+k]=a+b; v[j+k+i]=a-b;
					(v[j+k]>=mod)&&(v[j+k]-=mod);
					(v[j+k+i]<0)&&(v[j+k+i]+=mod);
					v[j+k]=v[j+k]*inv2%mod;
					v[j+k+i]=v[j+k+i]*inv2%mod;
				}
	}
}
signed main(){
	ios::sync_with_stdio(0); cin>>n; fi[1]=1;
	for(int i=2;i<=1000000;++i)
		(fi[i]=fi[i-1]+fi[i-2])>=mod&&(fi[i]-=mod);
	for(int i=1,v;i<=n;++i) cin>>v,cnt[v]++;
	for(int i=0;i!=1ll<<17;++i)
		cv[__builtin_popcount(i)][i]=cnt[i];
	for(int i=0;i<=17;++i) fwtor::fwt(cv[i]);
	for(int i=0;i<=17;++i)
		for(int j=0;j<=i;++j)
			for(int k=0;k!=1ll<<17;++k)
				ac[i][k]=(ac[i][k]+cv[j][k]*cv[i-j][k])%mod;
	for(int i=0;i<=17;++i) fwtor::ifwt(ac[i]);
	memcpy(tc,cnt,sizeof tc);
	fwtxor::fwt(tc);
	for(int i=0;i!=1ll<<17;++i) tc[i]=tc[i]*tc[i]%mod;
	fwtxor::ifwt(tc);
	for(int i=0;i!=1ll<<17;++i)
		cnt[i]=cnt[i]*fi[i]%mod,
		tc[i]=tc[i]*fi[i]%mod;
	fwtand::fwt(cnt);
	fwtand::fwt(tc);
	for(int i=0;i!=1ll<<17;++i)
		tc[i]=tc[i]*cnt[i]%mod;
	fwtand::ifwt(tc);
	for(int i=0;i!=1ll<<17;++i)
		cnt[i]=ac[__builtin_popcount(i)][i]*fi[i]%mod;
	fwtand::fwt(cnt);
	fwtand::fwt(tc);
	for(int i=0;i!=1ll<<17;++i)
		tc[i]=tc[i]*cnt[i]%mod;
	fwtand::ifwt(tc);
	for(int i=0;i!=17;++i) ans+=tc[1ll<<i];
	cout<<ans%mod<<endl;
}
