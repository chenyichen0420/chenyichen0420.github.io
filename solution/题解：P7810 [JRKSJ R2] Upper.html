思路分析

gcd(n,m) 有一个很巧妙的性质：他一定是 n,m 都有的一个因数（废话）。

但是这样仍然不是很好处理，不过我们都知道如果我们能找出所有的质因数的话，那只要 n,m 有公共的质因数，那他们的 gcd 就不为 1，反之则一定为 1。

所以我们其实只需要考虑两个数有没有公共质因数就行了。

在 109 范围内，一个数至多只会有 9 个质因数，而本题数据范围下 去重之后最多只有 101733 个不同的质因数。

这提示我们使用动态开点线段树来维护 dp 数组。具体的，dpp​ 表示以 p 作为一个区间的结尾的最大区间数量。

然后对于每一个点 p，我们去枚举他的因数，找到有这个因数的比他更小的那些位置 lp，并将答案更新为 dplp−1​。

是时间复杂度应该是 O(nlognlogV) 的，其中枚举质因数的 logV 小于 9。

代码如下：

#include<bits/stdc++.h>
using namespace std;
struct IO {
#define mxsz (1 << 21)
	char buf[mxsz], * p1, * p2;
	char pbuf[mxsz], * pp;
	IO() : p1(buf), p2(buf), pp(pbuf) {}
	~IO() { fwrite(pbuf, 1, pp - pbuf, stdout); }
	inline char gc() {
		if (p1 == p2) p2 = (p1 = buf) + fread(buf, 1, mxsz, stdin);
		return p1 == p2 ? ' ' : *p1++;
	}
#ifndef sipt
	inline int read() {
		int r = 0; char c = gc(); while (c < '0' || c>'9') c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return r;
	}
#else
	inline int read() {
		int r = 0; char c = gc(); bool rev = 0;
		while (c < '0' || c>'9') rev |= (c == '-'), c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return rev ? ~r + 1 : r;
	}
#endif
	inline void push(const char& c) {
		if (pp - pbuf == mxsz) fwrite(pbuf, 1, mxsz, stdout), pp = pbuf;
		*pp++ = c;
	}
	inline void write(int x) {
		static char sta[22]; int top = 0;
		do sta[top++] = x % 10, x /= 10; while (x);
		while (top) push(sta[--top] ^ 48);
	}
	inline void write(int x, char opc) {
#ifdef sopt
		if (x < 0) push('-'), x = ~x + 1;
#endif
		write(x), push(opc);
	}
} io;
map<int, int>prid; int pid;
vector<int>app[100005]; int p[100005], cnp, idp[100005];
int n, a[100005], dp[100005]; vector<int>h;
inline void tmax(int& l, const int r) { (l < r) && (l = r); }
struct seg_tree {
	struct node {
		int v, ls, rs;
		node() { ls = rs = 0; v = 0xcfcfcfcf; }
	}re[102005 * 17 * 2];
	int rt[102005], cnt;
	inline void ins(int nl, int nr, int cp, int cv, int& p) {
		if (!p) p = ++cnt;
		if (nl == nr) return tmax(re[p].v, cv);
		int mid = nl + nr >> 1;
		if (cp <= mid) ins(nl, mid, cp, cv, re[p].ls);
		else ins(mid + 1, nr, cp, cv, re[p].rs);
		re[p].v = max(re[re[p].ls].v, re[re[p].rs].v);
	}
	inline int que(int nl, int nr, int qr, int p) {
		if (!p) return 0xcfcfcfcf;
		if (nr <= qr) return re[p].v;
		int ret = 0xcfcfcfcf, mid = nl + nr >> 1;
		tmax(ret, que(nl, mid, qr, re[p].ls));
		if (qr > mid) tmax(ret, que(mid + 1, nr, qr, re[p].rs));
		return ret;
	}
}sgt;
signed main() {
	ios::sync_with_stdio(0); n = io.read();
	for (int i = 2; i * i <= 1e9; ++i) {
		for (int j = 2; j * j <= i; ++j)
			if (i % j == 0) goto frd;
		p[++cnp] = i; idp[i] = cnp;
	frd:;
	}
	pid = cnp;
	for (int i = 1; i <= n; ++i) {
		int vt = a[i] = io.read(); h.emplace_back(a[i]);
		for (int j = 1; j <= cnp && p[j] * p[j] <= vt; ++j)
			if (vt % p[j] == 0) {
				app[i].emplace_back(j);
				while (vt % p[j] == 0) vt /= p[j];
			}
		if (vt != 1) {
			if (vt * 1ll * vt <= 1e9) prid[vt] = idp[vt];
			if (!prid.count(vt)) prid[vt] = ++pid;
			app[i].emplace_back(prid[vt]);
		}
	}
	sort(h.begin(), h.end()); h.erase(unique(h.begin(), h.end()), h.end());
	for (int i = 1; i <= n; ++i)
		a[i] = lower_bound(h.begin(), h.end(), a[i]) - h.begin() + 1, dp[i] = 0xcfcfcfcf;
	for (int i = 1; i <= n; ++i) {
		for (int j : app[i]) tmax(dp[i], sgt.que(0, h.size(), a[i] - 1, sgt.rt[j]) + 1);
		for (int j : app[i]) if (dp[i - 1] >= 0)sgt.ins(0, h.size(), a[i], dp[i - 1], sgt.rt[j]);
	}
	cout << max(dp[n], -1) << endl;
}
