思路分析

看到这种多串 LCP 的题目，优先想一下 SA 可不可做。

SA 的 height 数组有一个非常美妙的性质：$\operatorname{LCP}(i,j)=\min_{k=i+1}^{j}height_k$，这里的 LCP 定义为 rank 为 $i$ 的字符串和 rank 为 $j$ 的字符串的最长公共前缀的长度。

于是，$\sum_{i=1}^n\operatorname{LCP}(i,k)=\sum_{i=1}^{k-1}\min_{j=i+1}^kheight_j+\sum_{i=k+1}^n\min_{j=k+1}^iheight_j+(n-sa_k+1)$。最后一项是 $\operatorname{LCP}(k,k)$ 的贡献。

于是此题确实可解，具体来说，我们将 SA，rank，height 数组求出来，然后按照上面的式子使用单调栈将前两项的贡献算出来，最后输出的时候查询 $rank_i$ 的数据即可，时间复杂度 $n\log n$，瓶颈在于求解 SA。参考代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
int n,nsa[1000005],cnt[1000005]; char c[1000005];
int sa[1000005],rk[2000005],nrk[1000005],h[1000005];
int ansl[1000005],ansr[1000005],ant; stack<pair<int,int>>s;
signed main(){
	ios::sync_with_stdio(0); cin>>n>>c;
	memmove(c+1,c,n*sizeof(char));
	for(int i=1;i<=n;++i) rk[i]=c[i]-'a'+1;
	for(int i=1;i<n;i<<=1){
		for(int j=1;j<=n;++j) cnt[rk[j+i]]++;
		for(int j=1;j<=max(n,26ll);++j) cnt[j]+=cnt[j-1];
		for(int j=max(n,26ll);j>=1;j--) cnt[j]=cnt[j-1];
		cnt[0]=0;
		for(int j=1;j<=n;++j) nsa[++cnt[rk[j+i]]]=j;
		for(int j=0;j<=max(n,26ll);++j) cnt[j]=0;
		for(int j=1;j<=n;++j) cnt[rk[j]]++;
		for(int j=1;j<=max(n,26ll);++j) cnt[j]+=cnt[j-1];
		for(int j=max(n,26ll);j>=1;j--) cnt[j]=cnt[j-1];
		cnt[0]=0;
		for(int j=1;j<=n;++j) sa[++cnt[rk[nsa[j]]]]=nsa[j];
		for(int j=0;j<=max(n,26ll);++j) cnt[j]=0;
		for(int j=1,k=0;j<=n;++j)
			nrk[sa[j]]=(k+=(rk[sa[j]]!=rk[sa[j-1]]||rk[sa[j]+i]!=rk[sa[j-1]+i]));
		memcpy(rk+1,nrk+1,n*sizeof(int));
	}
	for(int i=1,j=0;i<=n;++i){
		if(j)j--;if(rk[i]==1)continue;
		while(c[i+j]==c[sa[rk[i]-1]+j])j++;
		h[rk[i]]=j;
	}
	s.emplace(1,0);
	for(int i=2;i<=n;++i){
		int cp=i;
		while(s.top().second>h[i])
			ant-=(s.top().second)*(cp-s.top().first),
			cp=s.top().first,s.pop();
		s.emplace(cp,h[i]), ant+=(i-cp+1)*h[i];
		ansl[i]=ant;
		// cerr<<i<<" "<<h[i]<<" "<<ansl[i]<<endl;
	}
	while(s.size()) s.pop();
	s.emplace(n,0); ant=0; //cerr<<endl;
	for(int i=n-1;i>=1;i--){
		int cp=i;
		while(s.top().second>h[i+1])
			ant-=(s.top().second)*(s.top().first-cp),
			cp=s.top().first,s.pop();
		s.emplace(cp,h[i+1]), ant+=(cp-i+1)*h[i+1];
		ansr[i]=ant;
		// cerr<<i<<" "<<h[i+1]<<" "<<ansr[i]<<endl;
	}
	for(int i=1;i<=n;++i)
		cout<<ansl[rk[i]]+ansr[rk[i]]+(n-i+1)<<'\n';
}
//私は猫です!
