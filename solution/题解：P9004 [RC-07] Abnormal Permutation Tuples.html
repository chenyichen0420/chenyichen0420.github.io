思路分析

看起来直接单从任何一个限制下手都是没有前途的，于是我们考虑打表找二者之间的联系。我们发现按照字典序排列后，逆序对的构造方式如下：

初始时有一个序列 $A_1$，仅有一个数 $0$，这就是 $n=1$ 的答案。

然后我们考虑从 $n$ 的序列 $A_n$ 递推出 $n+1$ 的序列 $A_{n+1}$。

我们令 $AB$ 表示将 $A,B$ 按顺序拼接，$A^x$ 表示将 $A$ 中每一项都加 $x$ 得到的序列。

打表容易发现，$A_{n+1}=A_nA_n^1A_n^2\dots A_n^n$。这是我的做法的基础。

接下来我们考虑怎么去求解这个问题。观察到 $A_n^x$ 他们十分的相似，仅有一些细微的区别。于是我们考虑能不能够分治求解。

不能传统的分治，那样太屎了。我们针对这道题进行设计，不难想到分成 $n$ 段，每段 $(n-1)!$ 进行求解。

我们来设计一个状态：$dp_{i,j,k,l}$，表示对于 $n=i$ 的情况，选出来的逆序对大小最大为 $j$，最小为 $k$，个数为 $l$。

我们注意到一件事情：$A_n^x$ 和 $A_n$ 的区别就是 $+x$。也就是说我们虽然只求解了 $A_n$ 对应的 $dp_n$，但是实际上对于 $A_n^x$，$dp_{n,i,j,k}^\prime=dp_{n,i-x,j-x,k}$。相当于平移了一下，完全可以不用单独求解。

于是我们考虑对于每一层，从前向后暴力合并就行了。写出来的代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
int n,m,p,dp[15][93][93][32],tmp[93][93][32],ans[15][32],sm[93][93][32];
inline void mad(int&l,const int r){(l+=r)>=p&&(l-=p);}
signed main(){
    ios::sync_with_stdio(0);
    cin>>n>>m>>p; dp[1][0][0][1]=1;
    for(int i=2;i<=n;++i){
        //总共有 n 个形似 dp[i-1] 的数组，没有显式建出来
        memcpy(dp[i],dp[i-1],sizeof tmp);
        memcpy(tmp,dp[i],sizeof tmp);
        int mx=(i-1)*(i-2)/2;
        for(int j=2;j<=i;++j){
            for(int w3=0;w3<=mx;++w3)
                for(int w2=w3;w2<=mx;++w2)
                    for(int w4=1;w4<=w2-w3+1 && w4<m;++w4)
                        if(dp[i][w2][w3][w4])
                            for(int n3=j-1;n3<w3;++n3)
                                for(int n4=1;w4+n4<=m;++n4)
									for(int n2=n3+n4-1;n2<=w3-1;++n2)
										dp[i][w2][n3][w4+n4]=(dp[i][w2][n3][w4+n4]+dp[i][w2][w3][w4]*tmp[n2-j+1][n3-j+1][n4])%p;
            for(int w3=0;w3<=(i-2)*(i-1)/2;++w3)
                for(int w2=w3;w2<=(i-2)*(i-1)/2;++w2)
                    for(int w4=1;w4<=w2-w3+1 && w4<=m;++w4)
                        mad(dp[i][w2+j-1][w3+j-1][w4],tmp[w2][w3][w4]);
            mx++;
        }//如果选择进行 log 次合并呢？不好处理。不如优化边界。
    }
    for(int i=1;i<=n;++i)
        for(int j=0;j<=i*(i-1)/2;++j)
            for(int k=0;k<=j;++k)
                for(int l=1;l<=j-k+1 && l<=m;++l)
                    mad(ans[i][l],dp[i][j][k][l]);
    for(int i=1;i<=n;++i, cout<<endl)
        for(int j=1;j<=m;++j)
            cout<<ans[i][j]<<" ";
}

我们分析一下时间复杂度，大约是 $\sum_{i=2}^n(i-1)\times\frac{i(i-1)}{2}\times\frac{i(i-1)}{2}\times m\times\frac{i(i-1)}{2}\times\frac{i(i-1)}{2}\times m=O(\sum_{i=1}^ni^9m^2)=O(n^{10}m^2)$，显然飞不过去。

我们考虑优化，注意到最里面的那一坨有很多不受其他变量影响的东西，可能可以前缀和优化！

仔细分析，因为转移到 $dp_{i,w2,n3,w4+n4}$，所以说 $n3,n4$ 的枚举是不能省略掉的。就剩一个 $n2$ 可以考虑了。

刚好，因为 $tmp$ 关于 $i$ 固定，所以说 $n2$ 的那一维可以直接省略掉。优化后的代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
int n,m,p,ans[16][32],sm[121][121][32];__int128 dp[16][121][121][32],tmp[121][121][32];
inline void mad(int&l,const int r){(l+=r)>=p&&(l-=p);}
signed main(){
    ios::sync_with_stdio(0);
    cin>>n>>m>>p; dp[1][0][0][1]=1;
    for(int i=2;i<=n;++i){
        memcpy(dp[i],dp[i-1],sizeof tmp);
        memcpy(tmp,dp[i],sizeof tmp);
        int mx=(i-1)*(i-2)/2;
        for(int n3=0;n3<=i*(i-1)/2;++n3)
            for(int n4=1;n4<m;++n4)
                for(int n2=n3;n2<=i*(i-1)/2;++n2)
                    sm[n2][n3][n4]=(n2?sm[n2-1][n3][n4]:0)+tmp[n2][n3][n4],
                    (sm[n2][n3][n4]>=p) && (sm[n2][n3][n4]-=p);
        for(int j=2;j<=i;++j){
            for(int w3=0;w3<=mx;++w3)
                for(int w2=w3;w2<=mx;++w2)
                    for(int w4=1;w4<=w2-w3+1 && w4<m;++w4)
                        if(dp[i][w2][w3][w4])
                            for(int n3=j-1;n3<w3;++n3)
                                for(int n4=1;w4+n4<=m;++n4){
                                    int l=n3+n4-j,r=w3-j,ad;
                                    if(r<l) continue;
                                    ad=sm[r][n3-j+1][n4]-(l?sm[l-1][n3-j+1][n4]:0);
                                    if(ad<0) ad+=p;
                                    dp[i][w2][n3][w4+n4]=(dp[i][w2][n3][w4+n4]+dp[i][w2][w3][w4]*ad);
                                }
            for(int w3=0;w3<=(i-2)*(i-1)/2;++w3)
                for(int w2=w3;w2<=(i-2)*(i-1)/2;++w2)
                    for(int w4=1;w4<=w2-w3+1 && w4<=m;++w4)
                        dp[i][w2+j-1][w3+j-1][w4]+=tmp[w2][w3][w4];
            mx++;
			for(int w3=0;w3<=mx;++w3)
                for(int w2=w3;w2<=mx;++w2)
                    for(int w4=1;w4<=w2-w3+1 && w4<=m;++w4)
						dp[i][w2][w3][w4]%=p;
        }
    }
    for(int i=1;i<=n;++i)
        for(int j=0;j<=i*(i-1)/2;++j)
            for(int k=0;k<=j;++k)
                for(int l=1;l<=j-k+1 && l<=m;++l)
                    ans[i][l]=(ans[i][l]+dp[i][j][k][l])%p;
    for(int i=1;i<=n;++i, cout<<endl)
        for(int j=1;j<=m;++j) cout<<ans[i][j]<<" ";
}
