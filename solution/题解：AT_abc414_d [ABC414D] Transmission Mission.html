思路分析

首先我们肯定需要将原序列排序。我们不难发现如果从 i 位置将原先的一整个区间断成两个，那么减少的和就是 ai+1​−ai​。

因此我们贪心的选取最大的 m−1 个 ai+1​−ai​ 减去即可。代码如下：

#include <bits/stdc++.h>
using namespace std;
#define int long long
int n, m, a[500005], ans; priority_queue<int>pq;
signed main(){
	ios::sync_with_stdio(0); cin >> n >> m;
	for (int i = 1; i <= n; ++i) cin >> a[i];
	sort(a + 1, a + n + 1); m--; ans = a[n] - a[1];
	for (int i = 1; i != n; ++i) pq.emplace(a[i + 1] - a[i]);
	while (m--) ans -= pq.top(), pq.pop();
	cout << ans << endl;
}
