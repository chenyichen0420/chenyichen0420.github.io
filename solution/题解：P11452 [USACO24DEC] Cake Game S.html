思路分析

给出结论：Bessie 只能吃到最小的连续的 n2+1\frac{n}{2}+12n​+1 的蛋糕。

其实很显然，不管 Bessie 从哪里开始叠蛋糕，Elsie 都可以逼着她选择最劣的那一堆。而如果 Bessie 从中间开始叠的话，那么 Elsie 无论如何都无法保证吃到 Bessie 叠的蛋糕。因此，最小的连续的 n2+1\frac{n}{2}+12n​+1 的蛋糕是对于双方而言的最优解。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
int t, n, a[500005], ans;
signed main() {
	ios::sync_with_stdio(0);
	for (cin >> t; t; t--) {
		cin >> n; ans = 1e18;
		for (int i = 1; i <= n; ++i)
			cin >> a[i], a[i] += a[i - 1];
		int ln = n / 2;
		for (int i = 1; i + ln <= n; ++i)
			ans = min(ans, a[i + ln] - a[i - 1]);
		cout << ans << " " << a[n] - ans << endl;
	}
}
