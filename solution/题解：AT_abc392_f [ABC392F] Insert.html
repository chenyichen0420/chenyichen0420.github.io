思路分析

正难则反，考虑每个数应当被插入到哪个位置。很显然，他会被插入到第 pi​ 个空位。

我们只需要快速的找出第 pi​ 个空位在哪里就行了，显然可以使用树状数组快速维护前缀和，倍增单 log 查询。

代码如下：

#include<bits/stdc++.h>
using namespace std;
int n, p[500005], cn[500005], ans[500005];
inline int lb(int y) { return y & ~y + 1; }
inline void add(int p,int v) {
	for (int i = p; i <= n; i += lb(i))
		cn[i] += v;
}
inline int get(int v) {
	int ret = 0;
	for (int i = 19; i >= 0; i--)
		if ((ret | (1ll << i)) <= n && cn[ret | (1ll << i)] < v)
			ret |= (1ll << i), v -= cn[ret | (1ll << i)];
	return ret + 1;
}
signed main() {
	ios::sync_with_stdio(0);
	cin >> n;
	for (int i = 1; i <= n; ++i)
		cin >> p[i], add(i, 1);
	for (int i = n; i >= 1; i--) {
		int ps = get(p[i]);
		ans[ps] = i;
		add(ps, -1);
	}
	for (int i = 1; i <= n; ++i)
		cout << ans[i] << " ";
}
