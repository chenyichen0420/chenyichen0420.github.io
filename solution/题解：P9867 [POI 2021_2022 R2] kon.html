思路分析

考场上卡死在快速将节点 fork 出去，并尽可能地维护单点连边的怪圈里面了。

首先，对于 Z 类型的操作，我们其实可以选择考虑一下能不能干脆直接新建一个节点用实边挂在原先的那个节点上。

既然如此，为了防止后来的对于原先的那个节点的 W 类型操作污染了其下的 Z 的数据，我们新建一个节点作为其最新状态标记，并在这个节点上面去用虚边挂新的节点。

也就是说，操作完了之后我们会形成一棵实边的辅助森林。这个森林中的每一棵树代表了从某一个节点上不停的进行 Z 操作得到的结构。其中父节点的新建时间早于子节点。

结合上虚边，我们构成了一个辅助树，其中的虚边意味着在其父节点及实边树中这个节点以下的节点都和儿子节点及实边树中这个节点以下的节点之间连了边。实际意义：

说白了就是 Z 操作不改变在辅助树中跨过虚边的部分的连通性。

于是问题变成了给你一些加点加边的操作，询问某一个点所在的实边连通块走一条虚边后到达的所有其他的实边连通块的大小之和。

这是可做的。我们考虑离线下来，然后增加一个点就是在其实边辅助树的到根的部分全部 +1，在上一个实边辅助树的连接的节点的子树部分全部 +1。删除类似。

查询的时候就是找其他的实边辅助树在这个点的贡献，以及这个实边辅助树在这个位置的贡献总和。

代码如下：

#include<bits/stdc++.h>
using namespace std;
constexpr bool online = 0;
constexpr int sz = 3e6 + 5;
int n, ptr[sz], idp, idn, idq;
constexpr int opqry = 0, opadd = 1, opdel = -1;
struct oper { int v, op; }q[sz];
vector<int>son[sz]; char op; int ov;
int fa[sz], va[sz];
inline static void lnkw(int p) {
	son[ptr[p]].emplace_back(++idn);
	q[++idq] = { ptr[p],opdel };
	ptr[p] = idn; ptr[++idp] = ++idn;
	q[++idq] = { ptr[idp],opadd };
	q[++idq] = { ptr[p],opadd };
	va[idn] = ptr[p];
}
inline static void lnkz(int p) {
	son[ptr[p]].emplace_back(++idn);
	ptr[++idp] = idn; fa[idn] = ptr[p];
	q[++idq] = { idn,opadd };
}
int rt[sz], dfl[sz], dfr[sz], idd, ans;
inline static void dfs(int p, int f) {
	rt[p] = (f ? rt[f] : p); dfl[p] = ++idd;
	for (int sp : son[p]) dfs(sp, p); dfr[p] = idd;
}
struct tree_array {
	int v[sz];
	inline void ins(int p, int t) {
		do v[p] += t; while ((p += p & ~p + 1) <= idn);
	}
	inline int que(int p) {
		int t = 0; do t += v[p]; while (p &= p - 1); return t;
	}
	inline void ins(int l, int r, int t) {
		ins(l, t); ins(r + 1, -t);
	}
	inline int que(int l, int r) {
		return que(r) - que(l - 1);
	}
}my, oth;
signed main() {
	if (online)
		freopen("ball.in", "r", stdin),
		freopen("ball.out", "w", stdout);
	ios::sync_with_stdio(0); cin >> n; ptr[1] = 1;
	q[++idq] = { 1,opadd }; idp = idn = 1; lnkw(1);
	for (int i = 1; i <= n; ++i)
		if (cin >> op >> ov, op == 'W') lnkw(ov);
		else if (op == 'Z') lnkz(ov);
		else q[++idq] = { ptr[ov],opqry };
	for (int i = 1; i <= idn; ++i) if (!rt[i]) dfs(i, 0);
	for (int i = 1; i <= idq; ++i)
		if (q[i].op == opqry) {
			ans = oth.que(dfl[q[i].v]);
			if (q[i].v = va[rt[q[i].v]]) ans += my.que(dfl[q[i].v], dfr[q[i].v]);
			cout << ans << endl;
		}
		else {
			my.ins(dfl[q[i].v], q[i].op);
			if (q[i].v = va[rt[q[i].v]])
				oth.ins(dfl[q[i].v], dfr[q[i].v], q[i].op);
		}
}
