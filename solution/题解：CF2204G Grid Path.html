思路分析

看一眼数据范围，大概就是矩阵乘法优化转移什么的，我们考虑怎么做这个东西。

较为显然的，题意就是在说每一行都可以选一个连续的区间，但是要保证第一行的左端点是 $1$，而且相邻两行的区间有交集。

考虑直接暴力模拟上述过程的话时间复杂度是 $O(nm^4)$ 的，差得多，而且看起来非常的没有前途。

我们这样考虑：不满足要求的区间就是不相交的区间，我们直接找出这样的区间对，然后把他们的贡献再给减掉就行了，在这道题的数据范围下，这样的区间对数大概就是 $2\times10^6$，看起来还是没有前途。

我们进一步考虑：我们能不能将右端点不超过每一个位置的区间的方案数之和，以及左端点不超过某一个位置的区间的方案数之和给求出来？

这看起来是非常有道理的，实际上我们需要干的就是统计出上一层所有区间的方案数之和，以及上面所说的那两类情况的和，然后直接暴力枚举区间进行转移就行了。

因为区间可以 $O(1)$ 转移，每一类数值都可以 $O(m)\times O(m)=O(m^2)$ 的转移，因此时间复杂度降到 $O(nm^2)$，并且看起来非常的有前途。

我们把总数，前后缀的数值这总共 $2m+1$ 个数字视为状态，求出转移系数，写出转移矩阵，直接转移就行了，时间复杂度 $O(m^3\log n)$，足以通过此题。参考代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
int n,m,mod,trc[301][301][301];
struct mat{
	int v[301][301];
	inline mat operator*(const mat&r)const{
		static mat ret;
		static __int128 tmp[301][301];
		memset(tmp,0,sizeof tmp);
		for(int i=0;i<=2*m;++i)
			for(int j=0;j<=2*m;++j)
				for(int k=0;k<=2*m;++k)
					tmp[i][k]+=v[i][j]*r.v[j][k];
		for(int i=0;i<=2*m;++i)
			for(int j=0;j<=2*m;++j)
				ret.v[i][j]=tmp[i][j]%mod;
		return ret;
	}
}ans,tmp;
inline void mad(int&l,const int r){(l+=r)>=mod&&(l-=mod);}
signed main(){
	ios::sync_with_stdio(0); cin>>n>>m>>mod;
	time_t st=clock();
	for(int i=1;i<=m;++i)
		for(int j=i;j<=m;++j){
			trc[i][j][1]=1;
			if(j!=m)trc[i][j][j+1]=-1;
			if(i!=1)trc[i][j][m+i-1]=-1;
		}
	for(int l=m;l>=1;l--){
		if(l!=m)memcpy(tmp.v[l],tmp.v[l+1],sizeof tmp.v[0]);
		for(int r=l;r<=m;++r)
			for(int p=0;p<=m*2;++p)
				mad(tmp.v[l][p],trc[l][r][p]);
	}
	memcpy(tmp.v[0],tmp.v[1],sizeof tmp.v[1]);
	for(int r=1;r<=m;++r){
		if(r!=1) memcpy(tmp.v[r+m],tmp.v[r+m-1],sizeof tmp.v[0]);
		for(int l=1;l<=r;++l)
			for(int p=0;p<=2*m;++p)
				mad(tmp.v[r+m][p],trc[l][r][p]);
	}
	tmp.v[0][0]=ans.v[0][1]=1;
	for(int r=1;r<=m;++r) ans.v[0][m+r]=1;
	for(int i=0;i<=2*m;++i)
		for(int j=0;j!=i;++j)
			swap(tmp.v[i][j],tmp.v[j][i]);
	//cerr<<clock()-st<<endl;
	// for(int i=1;i<=n;++i){
	// 	ans=ans*tmp;
	// 	for(int j=0;j<=m*2;++j)
	// 		cerr<<ans.v[0][j]<<" ";
	// 	cerr<<endl;
	// }
	for(;n;n>>=1,tmp=tmp*tmp) if(n&1) ans=ans*tmp;
	cout<<(ans.v[0][0]+mod)%mod<<endl;
	//cerr<<clock()-st<<endl;
}
