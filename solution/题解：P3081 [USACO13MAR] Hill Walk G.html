思路分析

大抵是有更简单的解法的，毕竟李超树是紫的。

如果模拟这个过程，那么我们会需要多次在某一个位置求高度不高于一个特定值的最高线段的编号。

如果只考虑在某一个位置求最高线段的编号，那就是一道李超树的模板题，但是加上高度限制似乎不是很好做。

不过题目中还有一个限制没有用上：“这些用于表示小山的线段都没有任何交点”。

这句话非常有用，这意味着如果在某一个位置，有一个线段比你高，那么你永远也不可能走到这个线段上面。

这其实我们直接放弃那些比你高的线段。同理，在你下面的不可能突然窜上来，因此我们只考虑这些线段。

但是这样子我们会获得一个异常曲折的上界，不好实现。

我们又发现我们可以针对新增的每一个上界统计在其正下方的线段的贡献。

具体的来说，在其正上方或者正下方的线段的左端点在 [l,r) 之间，右端点我们并不关心。只要在 r 这个位置比他低的我们都可以视为在这个边界下。

因此我们可以先将所有除了第一条线段以外的按照左端点的横坐标排序，每一次新增这个线段正下方的线段，然后找到下个线段，知道发现没有下一个线段。

代码如下：

#include<bits/stdc++.h>
using namespace std;
constexpr double ep = 1e-8;
struct line {
	double k, b;
	line(double k = 0, double b = 0) :k(k), b(b) {};
	line(double x0, double y0, double x1, double y1) {
		if (x0 == x1) k = 0, b = max(y0, y1);
		else k = (y1 - y0) / (x1 - x0), b = y0 - x0 * k;
	}
}l[100005]; int n, rt, np, ans;
inline int cmp(double l, double r) { return l + ep < r ? -1 : r + ep < l ? 1 : 0; }
inline double calc(int p, int t) { return l[t].b + l[t].k * p; }
struct val {
	int id; double v;
	val(int id, double v) :id(id), v(v) {};
	inline bool operator>(const val& r) const{
		return cmp(v, r.v) ? v > r.v : id < r.id;
	}
	friend val max(const val& v1, const val& v2, const val& v3) {
		if (v1 > v2 && v1 > v3) return v1;
		if (v2 > v1 && v2 > v3) return v2;
		return v3;
	}
};
struct seg_tree {
	struct node { int ls, rs, v; }re[100005 * 86]; int cc;
	inline void ini(int cp, int l, int r, int& p) {
		if (!p) p = ++cc; int& lp = re[p].v, mid = l + r >> 1;
		int vc = cmp(calc(mid, cp), calc(mid, lp));
		if (vc == 1 || vc == 0 && lp > cp) swap(lp, cp);
		int lc = cmp(calc(l, cp), calc(l, lp));
		int rc = cmp(calc(r, cp), calc(r, lp));
		if (lc == 1 || lc == 0 && lp > cp) ini(cp, l, mid, re[p].ls);
		if (rc == 1 || rc == 0 && lp > cp) ini(cp, mid + 1, r, re[p].rs);
	}
	inline void inf(int cl, int cr, int l, int r, int cp, int& p) {
		if (!p) p = ++cc; int mid = l + r >> 1;
		if (l >= cl && r <= cr) return ini(cp, l, r, p);
		if (cl <= mid) inf(cl, cr, l, mid, cp, re[p].ls);
		if (cr > mid) inf(cl, cr, mid + 1, r, cp, re[p].rs);
	}
	inline val que(int c, int l, int r, int p) {
		if (!p || r<c || l>c) return val(0, -1e9);
		double v = calc(c, re[p].v); int mid = l + r >> 1;
		if (l == r) return val(re[p].v, v);
		return max(val(re[p].v, v), que(c, l, mid, re[p].ls), que(c, mid + 1, r, re[p].rs));
	}
}lct;
struct node {
	int a, b, c, d;
	inline void inp() { cin >> a >> b >> c >> d; }
	inline bool operator<(const node& r) { return a < r.a; }
}in[100005]; int ad = 2;
int main() {
	ios::sync_with_stdio(0); np = 1; l[0].b = -1e9;
	cin.tie(0); cout.tie(0); cin >> n;
	for (int i = 1; i <= n; ++i) in[i].inp();
	sort(in + 2, in + n + 1);
	l[1] = line(in[1].a, in[1].b, in[1].c, in[1].d);
	while (np) {
		ans++;
		for (int a, b, c, d; ad <= n; ++ad)
			if (in[ad].a < in[np].a) continue;
			else if (in[ad].a > in[np].c) break;
			else {
				a = in[ad].a; b = in[ad].b;
				c = in[ad].c; d = in[ad].d;
				l[ad] = line(a, b, c, d);
				if (calc(in[np].c, ad) < calc(in[np].c, np))
					lct.inf(a, c - 1, 0, 1e9, ad, rt);
			}
		np = lct.que(in[np].c, 0, 1e9, rt).id;
	}
	cout << ans << endl;
}
