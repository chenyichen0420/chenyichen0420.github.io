::::success[Solution]
我们这样思考：

我们先求解全部小于/大于等于分界线的极长部分。

这似乎不好处理，但是我们直到这个问题等价于询问有几个 ai→ai+1a_i\to a_{i+1}ai​→ai+1​ 是将分界线包含在内了的，然后 +1+1+1 作为最终的答案。

这样似乎就可以做了：我们无论初始状态下，还是进行更改，都维护 ai→ai+1a_i\to a_{i+1}ai​→ai+1​ 到底覆盖了哪些分界线。然后就变成了区间加，单点查，在线 O(mlog⁡n)O(m\log n)O(mlogn)。

CPP 
#include<bits/stdc++.h>
using namespace std;
int n,m,a[200005],rt;
struct seg_tree{
    struct node{ int ls,rs,c; }re[400005<<6]; int cn;
    inline node&ls(int p){return re[re[p].ls];}
    inline node&rs(int p){return re[re[p].rs];}
    inline void pud(int p){
        if(!re[p].ls) re[p].ls=++cn;
        if(!re[p].rs) re[p].rs=++cn;
        ls(p).c+=re[p].c; rs(p).c+=re[p].c; re[p].c=0;
    }
    inline void ins(int cl,int cr,int cv,int l,int r,int& p){
        if(!p) p=++cn;
        if(l>=cl&&r<=cr) return void(re[p].c+=cv);
        pud(p); int mid=l+r>>1;
        if(cl<=mid) ins(cl,cr,cv,l,mid,re[p].ls);
        if(cr>mid) ins(cl,cr,cv,mid+1,r,re[p].rs);
    }
    inline int que(int qp,int l,int r,int p){
        if(l==r) return re[p].c;
        pud(p); int mid=l+r>>1;
        if(qp<=mid) return que(qp,l,mid,re[p].ls);
        else return que(qp,mid+1,r,re[p].rs);
    }
}sgt;
inline void chg(int l,int r,int v){
    if(l==r) return; if(l>r) swap(l,r);
    sgt.ins(l+1,r,v,1,1e9,rt);
}
signed main(){
    ios::sync_with_stdio(0); cin>>n>>m;
    for(int i=1;i<=n;++i) cin>>a[i];
    for(int i=1;i!=n;++i) chg(a[i],a[i+1],1);
    for(int i=1,o,l,r;i<=m;++i)
        if(cin>>o>>l,o&1){
            int c=sgt.que(l,1,1e9,rt)+1;
            if(a[1]<l) cout<<(c>>1)<<endl;
            else cout<<(c+1>>1)<<endl;
        }
        else{
            cin>>r;
            if(l!=n)chg(a[l],a[l+1],-1);
            if(l!=1)chg(a[l-1],a[l],-1);
            a[l]=r;
            if(l!=n)chg(a[l],a[l+1],1);
            if(l!=1)chg(a[l-1],a[l],1);
        }
}
