思路分析

线段树板子题。

我们对于每一个区间，维护左右边界的字符及其连续的数量，区间长度和区间内最长连续字符数即可。

合并的时候一般情况下直接取最左边或者最右边，当中间的字符一样的时候可以合并出一个新的区间统计到区间内最长连续字符数中。

特别注意中间可以合并的是不是直接和边界接上了。注意区分左右区间，注意细节不要打错。

代码如下：

#include <bits/stdc++.h>
using namespace std;
int n, m; char c[500005], s;
struct val {
	char lc, rc; int lv, rv, mv, ln;
	inline void set(char c) {
		lv = rv = mv = ln = 1; lc = rc = c;
	}
	inline val operator+(const val& r) {
		val ret = { 0 };
		ret.lc = lc; ret.rc = r.rc;
		ret.lv = lv; ret.rv = r.rv;
		ret.mv = max(mv, r.mv);
		ret.ln = ln + r.ln;
		if (rc == r.lc) {
			if (lv == ln) ret.lv = ln + r.lv;
			if (r.rv == r.ln) ret.rv = r.ln + rv;
			ret.mv = max(ret.mv, rv + r.lv);
		}
		return ret;
	}
};
struct seg_tree {
	struct node { int l, r; val v; }re[500005 << 2];
	inline void pup(int p) { re[p].v = re[p << 1].v + re[p << 1 | 1].v; }
	inline void build(int l, int r, int p) {
		re[p].l = l; re[p].r = r;
		if (l == r) return re[p].v.set(c[l]);
		build((l + r >> 1) + 1, r, p << 1 | 1);
		build(l, (l + r >> 1), p << 1); pup(p);
	}
	inline void ins(int cp, char cc, int p) {
		if (re[p].l == re[p].r) return re[p].v.set(cc);
		if (cp <= re[p << 1].r) ins(cp, cc, p << 1);
		else ins(cp, cc, p << 1 | 1); pup(p);
	}
	inline val que(int l, int r, int p) {
		if (re[p].l >= l && re[p].r <= r) return re[p].v;
		if (l > re[p << 1].r) return que(l, r, p << 1 | 1);
		if (r <= re[p << 1].r) return que(l, r, p << 1);
		return que(l, r, p << 1) + que(l, r, p << 1 | 1);
	}
}sgt;
signed main(){
	ios::sync_with_stdio(0);
	cin >> n >> m >> (c + 1); sgt.build(1, n, 1);
	for (int i = 1, o, l, r; i <= m; ++i)
		if (cin >> o >> l, o & 1) cin >> s, sgt.ins(l, s, 1);
		else cin >> r, cout << sgt.que(l, r, 1).mv << endl;
}
