思路分析

竟然都在写线段树，不写一点脑子都没有的平衡树，没爱了（雾）。

我们维护 FHQ Treap，每一个节点中的信息是数值，原下标。每一次的操作实质上就是将要取相反数的部分分裂出来，打上标记，然后重新合并。

因为现在值域是有交的，所以我们直接进行 FHQ 的有交分裂即可。具体来说，我们枚举到一个节点时，钦定 $l$ 的随机值不小于 $r$，然后将 $r$ 按照 $l$ 节点的 $v$ 分裂开，$l$ 的左子树和 $r$ 的左部分合并，$l$ 的右子树和 $r$ 的右部分合并。

这样操作的时间复杂度被证明是 $O(\log^2)$ 的，具体怎么证我其实不太会，不过这应该是一个比较普及的基础科技了，所以我就不严谨证明了。

大方向就是对于有序序列来说定义势能是相邻两项的差的对数，然后区间取相反数只会给势能增加 $O(\log)$，合并间隔数为 $k$ 所消耗的势能是 $k$ 的，带上本身的 $\log$ 就是 $\log^2$ 了。

参考代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
mt19937 mt(chrono::high_resolution_clock::now().time_since_epoch().count());
int n,m,a[100005],rt,tp[4]; char op;
struct fhq_treap{
	struct node{
		int v,id,ls,rs,f,mn,mx,rd;
		inline void flip(){
			swap(mn,mx); mn=-mn; mx=-mx;
			v=-v,f^=1,swap(ls,rs);
		}
	}re[100005];
	inline void pud(int p){
		if(re[p].f)
			re[re[p].ls].flip(),
			re[re[p].rs].flip(),
			re[p].f=0;
	}
	inline void pup(int p){
		re[p].mx=max({re[re[p].ls].mx,re[re[p].rs].mx,re[p].v});
		re[p].mn=min({re[re[p].ls].mn,re[re[p].rs].mn,re[p].v});
	}
	inline void split(int p,int v,int &l,int &r){
		if(!p) return void(l=r=0); pud(p);
		if(re[p].v>v) split(re[p].ls,v,l,re[r=p].ls),pup(r);
		else split(re[p].rs,v,re[l=p].rs,r),pup(l);
	}
	inline int merge(int l,int r){
		if(!l||!r)return l|r;
		pud(l); pud(r);
		if(re[l].rd>re[r].rd){
			re[l].rs=merge(re[l].rs,r);
			pup(l); return l;
		}
		else{
			re[r].ls=merge(l,re[r].ls);
			pup(r); return r;
		}
	}
	inline int intmerge(int l,int r){
		if(!l||!r) return l|r;
		if(re[l].rd<re[r].rd) swap(l,r);
		int rpl,rpr; pud(l);
		split(r,re[l].v,rpl,rpr);
		re[l].ls=intmerge(re[l].ls,rpl);
		re[l].rs=intmerge(re[l].rs,rpr);
		pup(l); return l;
	}
	inline void ins(int v,int p){
		split(rt,v,tp[1],tp[2]);
		re[p]={v,p,0,0,0,v,v,(int)mt()};
		rt=merge(merge(tp[1],p),tp[2]);
	}
	inline void flpbg(int v){
		split(rt,v,tp[1],tp[2]);
		re[tp[2]].flip();
		rt=intmerge(tp[1],tp[2]);
	}
	inline void flpsm(int v){
		split(rt,v-1,tp[1],tp[2]);
		re[tp[1]].flip();
		rt=intmerge(tp[1],tp[2]);
	}
	inline void dfs(int p){
		a[re[p].id]=re[p].v; pud(p);
		if(re[p].ls) dfs(re[p].ls);
		if(re[p].rs) dfs(re[p].rs);
	}
}fhq;
signed main(){
	ios::sync_with_stdio(0); cin>>n>>m;
	fhq.re[0].mx=1e9; fhq.re[0].mn=-1e9;
	for(int i=1,a;i<=n;++i) cin>>a,fhq.ins(a,i);
	for(int i=1,p;i<=m;++i)
		if(cin>>op>>p,op=='<') fhq.flpsm(p);
		else fhq.flpbg(p);
	fhq.dfs(rt);
	for(int i=1;i<=n;++i) cout<<a[i]<<" "; cout<<endl;
}
