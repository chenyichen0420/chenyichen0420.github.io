思路分析

平面图转对偶图板子题。之前没见过，写篇题解整理一下。

首先，我们不会出现高度非 0,1 的点。这样因为有更大的坡度，显然不优。其次贪心的思路，我们不会让 1 中间凹下去，也不会让 0 中间凸起来，因此 0,1 分别围绕左上，右下角形成连通块。于是就是一个有向图的最小割。

对于一个平面图，也就是可以做到在一个平面内边不相交的图，我们可以将相邻的面之间连上边，从而得到另外一张图。这张图在网络流中有一些美妙的性质。

事实上，对于一张无向平面图，我们先在源汇点之间连一条边，然后对于原来的每一个边，将这些边两边的面连起来，权值和原边的权值一样。像这样：

你可以把这张图理解为原图的无向版。倾斜的两条边是源汇点绕地球一圈连上的边，分成了三大块。原图有中间蓝色的 4 条边，对偶图有红色的 4 条边。

而显然一条从右上角的点到左下角的点的一条简单路径恰好对应了原图的一个割。而其中的最短路就是原图的最小割。

那么有向图我们可不可以这么干呢？可以，如果我们能够正确的指定对偶图的边的方向。

还是我们的核心思路：一条从右上角的点到左下角的点的一条简单路径恰好对应了原图的一个割。

如果我们让原图的边全部向同一个方向转一个直角的话，那么恰好就满足了我们的性质。接下来你唯一要做的就是确定到底是从右上到左下还是反着来，因为这恰好分别对应了左上分别是 0 还是 1。这个就要看你是怎么转的了。

代码如下：

#include<bits/stdc++.h>
using namespace std;
#define int long long
struct node {
	int p, v;
	node(int pi = 0, int vi = 0) :p(pi), v(vi) {};
	friend bool operator<(const node& l, const node& r) {
		return l.v > r.v;
	}
}tmp; priority_queue<node>pq;
int n, s, t; bool vis[505 * 505];
vector<node>son[505 * 505]; int d[505 * 505];
inline int id(int x, int y) {
	return (x - 1) * n + y;
}
signed main() {
	ios::sync_with_stdio(0);
	cin >> n; n += 2;
	for (int i = 2, v; i <= n; ++i)
		for (int j = 2; j != n; ++j)
			cin >> v,
			son[id(i - 1, j)].emplace_back(id(i, j), v);
	for (int i = 2, v; i != n; ++i)
		for (int j = 2; j <= n; ++j)
			cin >> v,
			son[id(i, j)].emplace_back(id(i, j - 1), v);
	for (int i = 2, v; i <= n; ++i)
		for (int j = 2; j != n; ++j)
			cin >> v,
			son[id(i, j)].emplace_back(id(i - 1, j), v);
	for (int i = 2, v; i != n; ++i)
		for (int j = 2; j <= n; ++j)
			cin >> v,
			son[id(i, j - 1)].emplace_back(id(i, j), v);
	s = n * n + 1; t = s + 1;
	for (int i = 2; i <= n; ++i)
		son[s].emplace_back(id(1, i));
	for (int j = 2; j != n; ++j)
		son[s].emplace_back(id(j, n));
	for (int i = 2; i <= n; ++i)
		son[id(i, 1)].emplace_back(t);
	for (int j = 2; j != n; ++j)
		son[id(n, j)].emplace_back(t);
	memset(d, 0x0f, sizeof d);
	pq.emplace(s, d[s] = 0);
	while (pq.size()) {
		tmp = pq.top(); pq.pop();
		if (vis[tmp.p]) continue; vis[tmp.p] = 1;
		for (const node& sp : son[tmp.p])
			if (d[sp.p] > d[tmp.p] + sp.v)
				pq.emplace(sp.p, d[sp.p] = d[tmp.p] + sp.v);
	}
	cout << d[t];
}
