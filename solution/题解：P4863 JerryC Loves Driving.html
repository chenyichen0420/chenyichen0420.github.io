思路分析

看起来想要一个线性解，那就先行吧，随便化一化式子就行：

i=1∑A​j=1∑i​⌊ji​⌋×(−1)j​=i=1∑A​j=1∑A​⌊ji​⌋×(−1)j=j=1∑A​i=1∑A​⌊ji​⌋×(−1)j=j=1∑A​(−1)ji=1∑A​⌊ji​⌋​

显然对于固定的 j，后者形如 0,0,…,0,1,1,…,1,2,2,…,2,…，可以直接 O(1) 算，于是就 O(n) 了。

代码如下：

#include<bits/stdc++.h>
using namespace std;
#define int long long
int n, m, tmp;
inline int calc(int v) {
	tmp = -v * (v + 1) / 2;
	for (int i = 2; i <= v; ++i) {
		int va = (v + 1) / i, lft = (v + 1) % i;
		tmp += (i & 1 ? -1 : 1) * (va * (va - 1) / 2 * i + lft * va);
	}
	return tmp;
}
signed main() {
	ios::sync_with_stdio(0); cin >> n >> m;
	/*
		\sum_{i=A}^B\sum_{j=1}^i (i/j)*(-1)^j
	=	\sum_{i=1}^B\sum_{j=1}^B (i/j)*(-1)^j - \sum_{i=1}^A\sum_{j=1}^A (i/j)*(-1)^j 
(ca)=	\sum_{i=1}^A\sum_{j=1}^A (i/j)*(-1)^j
	=	\sum_{j=1}^A\sum_{i=1}^A (i/j)*(-1)^j
	=	\sum_{j=1}^A(-1)^j\sum_{i=1}^A (i/j)
	*/
	cout << calc(m) - calc(n - 1) << endl;
}
