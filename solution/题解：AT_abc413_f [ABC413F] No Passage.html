思路分析

看起来跟个博弈论似的，实际上一点关系都没有，因为我们其实不难想到禁用方向那一步本质上就是强制走次短路。

那正解就已经呼之欲出了。我们以目标作为起点，倒序进行多源广搜，区别是当且仅当一个节点第二次可以被增广时我们才去增广他。因为广搜的短路优先性质，第二次可被增广的时候恰为其次短路。

然后就是两点提醒：有不可达的是直接不统计贡献；答案超过 int 的表示范围，开 long long。

代码如下：

#include <bits/stdc++.h>
using namespace std;
#define int long long
struct pos {
	int x, y;
	pos(int x = 0, int y = 0) :x(x), y(y) {};
}tmp; queue<pos>q;
int n, m, k, t[3005][3005], d[3005][3005], ans;
constexpr int dx[] = { 0,0,1,-1 }, dy[] = { 1,-1,0,0 };
signed main(){
	ios::sync_with_stdio(0);
	cin >> n >> m >> k; memset(d, -1, sizeof d);
	for (int i = 1, l, r; i <= k; ++i)
		cin >> l >> r, d[l][r] = 0, q.emplace(l, r);
	while (q.size()) {
		tmp = q.front(); q.pop();
		for (int i = 0; i != 4; ++i)
			if (d[tmp.x + dx[i]][tmp.y + dy[i]] < 0)
				if (++t[tmp.x + dx[i]][tmp.y + dy[i]] == 2)
					d[tmp.x + dx[i]][tmp.y + dy[i]] = d[tmp.x][tmp.y] + 1,
					q.emplace(tmp.x + dx[i], tmp.y + dy[i]);
	}
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= m; ++j)
			if (~d[i][j]) ans += d[i][j];
	cout << ans << endl;
}
