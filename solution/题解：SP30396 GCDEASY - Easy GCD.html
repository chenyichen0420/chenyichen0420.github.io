思路分析

建议评橙。

首先，对于一个序列，v=gcd(a1​,a2​,…,an​) 一定是不变的，我们可以预处理出来。

然后，枚举所有的 s∣v 且 s=1，其能够取到的最大值就是 ⌊sk​⌋×s。我们可以枚举其因数（别太暴力），复杂度 O(n​)。

代码如下：

#include<bits/stdc++.h>
using namespace std;
int t, n, k, a, gc, ans;
inline int gcd(const int& l, const int& r) {
	if (!r) return l;
	return gcd(r, l % r);
}
signed main() {
	ios::sync_with_stdio(0);
	cin >> t;
	while (t--) {
		cin >> n >> k; gc = ans = 0;
		while (n--) cin >> a, gc = gcd(a, gc);
		for (int i = 1; i * i <= gc; ++i)
			if (gc % i == 0) {
				if (i != 1) ans = max(ans, k / i * i);
				ans = max(ans, k / (gc / i) * (gc / i));
			}
		cout << ans << endl;
	}
	return 0;
}
