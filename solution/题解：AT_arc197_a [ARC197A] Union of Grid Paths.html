思路分析

有一个结论很重要。我们不难发现，如果没有方向强制限制，那么每走一步所形成的阵线都是一条斜线。

那么方向限制了什么呢？方向限制了你在这一步实际能到达的范围。但是不管你怎么限制，这条阵线都会是原先的那条线的一个子区间。

其实也很显然，相邻两个之间可以通过两种方向的互换得到。二者不可达当且仅当互换后在某个方向上的操作个数小于强制要求个数，或者考虑上后面的强制操作后无法抵达终点。

第二点很容易漏，但是只要考虑完整这两点，此题就可以迎刃而解了。代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
int t, n, m, cd[800005], cr[800005], ans; string s;
int dx, ux, dy, uy;
signed main() {
	ios::sync_with_stdio(0);
	for (cin >> t;t;t--) {
		cin >> n >> m >> s; ans = 1;
		for (int i = 1;i <= s.size();++i)
			cd[i] = cd[i - 1] + (s[i - 1] == 'D'),
			cr[i] = cr[i - 1] + (s[i - 1] == 'R');
		dx = dy = ux = uy = 1;
		for (int i = 1;i <= s.size();++i) {
			if (s[i - 1] == 'D') dy++, uy++;
			else if (s[i - 1] == 'R') dx++, ux++;
			else {
				if (dy + (cd[s.size()] - cd[i]) + 1 <= n) dy++;
				else dx++;
				if (ux + (cr[s.size()] - cr[i]) + 1 <= m) ux++;
				else uy++;
			}
			ans += (ux - dx + 1);
		}
		cout << ans << endl;
	}
}
