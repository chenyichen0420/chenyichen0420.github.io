思路分析

略微有点复杂，但可能比较直观。

首先我们可以通过换根动规求出以每个点为根时的距离和。考虑对 $l,r$ 的距离取最小值对答案产生的贡献。

不难想象，如果将 $l,r$ 的路径专门拎出来，这棵树会变成这个样子：

我们选择考虑以 $r$ 为根，不难发现有距离变化的就是绿框框出来的部分。

我们可以观察子树内距离的具体的变化情况，发现从右至左依次减少了 $2,4$ 的距离。显然如果这条链更长一些的话就会形如 $2,4,6,8,\dots,2k$ 的形式一直减下去。

但这只是一种情况。如果链长为奇数呢？看图分析：

这次变成了 $1,3,5,7,\dots,2k+1$ 的形式。

总之呢我们发现他总是以每一次差 $2$ 的形式增加相对变化量。也就意味着差分数组几乎全是 $2$。

因此我们可以考虑记录一个二维数组 $s_{i,j}$，表示从 $i$ 向上 $2^j$ 个点的子树大小的两倍之和。

不难发现，我们只要这样子跳过中点，就得到了变化量。特殊情况是长度为奇数的链，会多计算一些贡献。

当然了，为了实现方便，你强制让 $l$ 是深度更深的点。这样子从 $l$ 到变化的界限点一定是一路向上的链。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
vector<int>son[200005];
int n, m, td[200005], ds[200005], d[200005], sz[200005], f[200005][19], ss[200005][19];
inline void dfs(int p, int fa) {
	td[p] = d[p] = d[fa] + 1; sz[p] = 1; f[p][0] = fa;
	for (int i = 1; i <= 18; ++i) f[p][i] = f[f[p][i - 1]][i - 1];
	for (int sp : son[p]) if (sp != fa)
		dfs(sp, p), sz[p] += sz[sp], td[p] += td[sp];
}
inline void dps(int p, int fa) {
	if (fa) ds[p] = ds[fa] + n - sz[p] * 2; ss[p][0] = sz[p] * 2;
	for (int i = 1; i <= 18; ++i) ss[p][i] = ss[p][i - 1] + ss[f[p][i - 1]][i - 1];
	for (int sp : son[p]) if (sp != fa) dps(sp, p);
}
inline int lca(int l, int r) {
	for (int i = 18; i >= 0; i--)
		(d[f[l][i]] >= d[r]) && (l = f[l][i]);
	if (l == r) return l;
	for (int i = 18; i >= 0; i--)
		(f[l][i] ^ f[r][i]) && (l = f[l][i], r = f[r][i]);
	return f[l][0];
}
inline int dis(int l, int r, int lc) {
	return d[l] + d[r] - d[lc] - d[lc];
}
inline int fat(int p, int t) {
	for (int i = 18; i >= 0; i--)
		if (t >> i & 1) p = f[p][i];
	return p;
}
signed main() {
	ios::sync_with_stdio(0); cin >> n; d[0] = -1;
	for (int i = 1, l, r; i != n; ++i)
		cin >> l >> r,
		son[l].emplace_back(r),
		son[r].emplace_back(l);
	dfs(1, 0); ds[1] = td[1]; dps(1, 0); cin >> m;
	for (int i = 1, l, r; i <= m; ++i) {
		cin >> l >> r;
		if (l == r) {
			cout << ds[l] << endl;
			continue;
		}
		if (d[l] < d[r]) swap(l, r);
		int lc = lca(l, r), ol = l;
		int dt = dis(l, r, lc);
		int fp = fat(l, dt - 1 >> 1);
		int ans = ds[r];
		for (int j = 18; j >= 0; j--)
			if (d[f[l][j]] >= d[fp] - 1) {
				ans -= ss[l][j];
				l = f[l][j];
			}
		if (dt % 2) ans += ss[fp][0] / 2;
		cout << ans << endl;
	}
}
