思路分析

:::error[不嘻嘻]
呜呜呜，我再也不开 $2\times10^5$ 个 queue 和 $2\times10^5$ 个 stack 了。
:::

:::info[step1]
不难想到，如果某些绳子上有超过 $1$ 个节点，那么可能会有解当且仅当这些区间的长度是一样的，而且如果有解、则只能是这些区间的长度。

因此，我们可以只用考虑最边上的节点之间的区间长度为最终答案这一限制。

总之呢，如果我们可以快速的判断某一个区间长度是不是合法的解，那么简单的套上这个小结论就可以解决第四档部分分。
:::

:::info[step2]
对于不存在绳内区间的情况，有没有办法更好的限制可能的答案范围呢？

其实是有的：我们关注一下最后的答案。假设绳上左右剩余区间的长度分别是 $l_i,r_i$，那么答案一定满足 $r_i+l_{i+1}=k$，其中 $k$ 为你选定的区间长度。

我们将 $(r_i,l_{i+1})$ 对按照 $r_i$ 的值进行排序，不难发现 $r$ 单增，$l$ 单减。

又因为原先的所有值当中必定恰好剩余一个 $l$ 和一个 $r$，所以说最小的两个 $l$ 和最大的两个 $r$ 配成的四个对中一定会出现至少一个，否则必然无解。
:::

:::info[step3]
最后我们考虑怎么验证一个值 $k$ 是不是合法的。

注意到一个左右剩余分别为 $(l_i,r_i)$ 的绳子后面如果接有绳子 $(l_j,r_j)$，那么必定满足 $r_i+l_j=k$。

我们对此进行图论建模，不难发现其等价于有一条 $l_i\to k-r_i$ 的单向边，一条合法的拼绳方案对应了一条欧拉路径。

于是此题可解了。
:::

:::success[implement]

CPP 
#include<bits/stdc++.h>
using namespace std;
constexpr bool online=0;
int n,s[10],sc; set<int>ln; vector<int>hp;
struct rope{ int l,c; vector<int>p; }rp[100005];
unordered_map<int,int>nid; int idc,ide;
map<pair<int,int>,int>eid; vector<int>ep[200005];
vector<int>son[200005]; stack<int>awk;
int rd[200005],cd[200005],anl;
inline int getid(int v){
    if(nid.count(v)) return nid[v];
    return nid[v]=++idc;
}
inline int geted(int l,int r){
    if(eid.count({l,r})) return eid[{l,r}];
    return eid[{l,r}]=++ide;
}
inline void dfs(int p){
    while(son[p].size()){
        int sp=son[p].back();
        son[p].pop_back();
        dfs(sp);
    }
    awk.emplace(p);
}
inline void tryit(){
    int cg(0),cs(0);
    for(int i=1;i<=n;++i)
        cd[getid(rp[i].p[0])]++,
        rd[getid(anl-(rp[i].l-rp[i].p.back()))]++;
    for(int i=1;i<=idc;++i)
        if(cd[i]==rd[i]) continue;
        else if(abs(cd[i]-rd[i])!=1) goto fwd;
        else if(cd[i]>rd[i])
            if(cg) goto fwd;
            else cg=i;
        else
            if(cs) goto fwd;
            else cs=i;
    if(!cg) cg=1;
    for(int i=1;i<=n;++i)
        ep[geted(getid(rp[i].p[0]),getid(anl-(rp[i].l-rp[i].p.back())))].emplace_back(i),
        son[getid(rp[i].p[0])].emplace_back(getid(anl-(rp[i].l-rp[i].p.back())));
    dfs(cg);
    if(awk.size()!=n+1){
        while(awk.size()) awk.pop();
        for(int i=1;i<=idc;++i) son[i].clear();
        for(int i=1;i<=ide;++i) while(ep[i].size()) ep[i].pop_back();
        ide=0; eid.clear();
        goto fwd;
    }
    cout<<"Yes\n"; 
    cg=awk.top(); awk.pop();
    while(awk.size()){
        cout<<ep[geted(cg,awk.top())].back()<<" ";
        ep[geted(cg,awk.top())].pop_back();
        cg=awk.top(); awk.pop();
    }
    exit(0);
fwd:
    for(int i=1;i<=idc;++i) cd[i]=rd[i]=0;
    idc=0; nid.clear();
}
signed main(){
    if(online)
        freopen("rope.in","r",stdin),
        freopen("rope.out","w",stdout);
    ios::sync_with_stdio(0); cin>>n;
    for(int i=1;i<=n;++i){
        cin>>rp[i].c>>rp[i].l; sc+=rp[i].c;
        for(int j=1,v;j<=rp[i].c;++j)
            cin>>v,rp[i].p.emplace_back(v);
        rd[i]=rp[i].p[0]; cd[i]=rp[i].l-rp[i].p.back();
        for(int j=1;j!=rp[i].c;++j)
            ln.emplace(rp[i].p[j]-rp[i].p[j-1]);
    }
    if(ln.size()>1) return cout<<"No\n",0;
    if(ln.size()){
        memset(rd,0,sizeof rd); memset(cd,0,sizeof cd);
        anl=*ln.begin(),tryit();
    }
    else{
        sort(rd+1,rd+n+1); sort(cd+1,cd+n+1,greater<int>());
        int prob[4]={rd[1]+cd[1],rd[2]+cd[1],rd[1]+cd[2],rd[2]+cd[2]};
        memset(rd,0,sizeof rd); memset(cd,0,sizeof cd);
        for(int i=0;i!=4;++i) anl=prob[i],tryit();
    }
    cout<<"No\n";
    return 0;
}
