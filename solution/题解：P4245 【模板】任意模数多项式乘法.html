思路分析

使用 karatsuba 大法！

具体的，我们有如下优化思路：

首先避免各种边界情况，直接假设 $A,B$ 均为 $2^n$ 次的多项式，其中 $n\in\mathbb{N}^*$。

我们分裂多项式，令 $A=A_l+A_r\times x^{2^{n-1}}$，$B$ 同理。

于是展开暴力计算的式子：
$\begin{aligned} A\times B&=(A_l+A_r\times x^{2^{n-1}})\times (B_l+B_r\times x^{2^{n-1}})\\ &=A_l\times B_l+(A_l\times B_r+A_r\times B_l)\times x^{2^{n-1}}+A_r\times B_r\times x^{2^n} \end{aligned}$

假设两个 $2^0$ 的多项式的卷积计算是 $O(1)$ 的，那么时间复杂度函数 $T(n)=4\times T(\frac{n}{2})+O(n)=n^2$，显然并没有什么特别本质的优化。

接下来，我们动用一些人类智慧，注意到 $A_l\times B_r+A_r\times B_l=(A_l+B_l)\times(A_r+B_r)-A_l\times B_l-A_r\times B_r$。

反代会上面的式子，得到 $A_l\times B_l+((A_l+B_l)\times(A_r+B_r)-A_l\times B_l-A_r\times B_r)\times x^{2^{n-1}}+A_r\times B_r\times x^{2^n}$。

注意到我们只需要计算 $A_l\times B_l,(A_l+B_l)\times(A_r+B_r),A_r\times B_r$ 三次卷积了。于是时间复杂度降到 $T(n)=3\times T(\frac{n}{2})+O(n)=n^{\log_23}\approx n^{1.585}$。理论上足以通过此题。

实践的时候还是需要注意常数因子的。具体的，这个算法在长度较短的时候常数很大，因此在长度 $\le32$ 的时候可以直接使用暴力卷积。使用 __int128 减少取模次数。

另外建议使用数组，memcpy 等方式加速传值和拷贝，注意清空数组以及取模即可。参考代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
#ifdef _MSC_VER
#define __int128 long long
#endif
struct IO {
#define mxsz (1 << 21)
	char buf[mxsz], * p1, * p2;
	char pbuf[mxsz], * pp; bool acc[128];
	IO() : p1(buf), p2(buf), pp(pbuf) {}
	~IO() { fwrite(pbuf, 1, pp - pbuf, stdout); }
	inline char gc() {
		if (p1 == p2) p2 = (p1 = buf) + fread(buf, 1, mxsz, stdin);
		return p1 == p2 ? ' ' : *p1++;
	}
	inline void setacc(const char* c) { while (*c) acc[*c++] = 1; }
	inline char getc() { char c; while (!acc[c = gc()]); return c; }
#ifndef sipt
	inline int read() {
		int r = 0; char c = gc(); while (c < '0' || c>'9') c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return r;
	}
#else
	inline int read() {
		int r = 0; char c = gc(); bool rev = 0;
		while (c < '0' || c>'9') rev |= (c == '-'), c = gc();
		while (c >= '0' && c <= '9') r = r * 10 + (c ^ 48), c = gc();
		return rev ? ~r + 1 : r;
	}
#endif
	inline void push(const char& c) {
		if (pp - pbuf == mxsz) fwrite(pbuf, 1, mxsz, stdout), pp = pbuf;
		*pp++ = c;
	}
	inline void write(int x) {
		static char sta[22]; int top = 0;
		do sta[top++] = x % 10, x /= 10; while (x);
		while (top) push(sta[--top] ^ 48);
	}
	inline void write(int x, char opc) {
#ifdef sopt
		if (x < 0) push('-'), x = ~x + 1;
#endif
		write(x), push(opc);
	}
	inline void puts(const char* c) { while (*c) push(*c++); }
} io;
int n, m, p, v1[262144], v2[262144], sac[16][262144], sbd[16][262144], s[16][262144], sz;
int a[16][262144], b[16][262144], c[16][262144], d[16][262144], ans[262144];
__int128 tmp[64];
inline void mad(int& l, const int r) { (l += r) >= p && (l -= p); }
inline void split(int* v, int* a, int* b, int ln) {
	memcpy(a, v, sizeof(int) * ln);
	memcpy(b, v + ln, sizeof(int) * ln);
}
inline void calc(int* p1, int* p2, int ln, int* rs, int dep) {
	if (ln <= 32) {
		for (int i = 0; i != ln; ++i)
			for (int j = 0; j != ln; ++j)
				tmp[i + j] += p1[i] * p2[j];
		for (int i = 0; i != ln << 1; ++i)
			rs[i] = tmp[i] % p, tmp[i] = 0;
		return;
	}
	ln >>= 1;
	split(p1, a[dep], b[dep], ln);
	split(p2, c[dep], d[dep], ln);
	calc(a[dep], c[dep], ln, sac[dep], dep + 1);
	calc(b[dep], d[dep], ln, sbd[dep], dep + 1);
	for (int i = 0; i != ln; ++i)
		mad(a[dep][i], b[dep][i]),
		mad(c[dep][i], d[dep][i]);
	calc(a[dep], c[dep], ln, s[dep], dep + 1);
	ln <<= 1;
	for (int i = 0; i != ln; ++i)
		s[dep][i] = (s[dep][i] - sac[dep][i] - sbd[dep][i] + p * 2) % p;
	for (int i = 0; i != ln; ++i) rs[i] = sac[dep][i];
	for (int i = 0, j = ln >> 1; i != ln; ++i, ++j) mad(rs[j], s[dep][i]);
	for (int i = 0, j = ln; i != ln; ++i, ++j) mad(rs[j], sbd[dep][i]);
	memset(s[dep], 0, sizeof(int) * ln);
	memset(sac[dep], 0, sizeof(int) * ln);
	memset(sbd[dep], 0, sizeof(int) * ln);
}
signed main() {
	ios::sync_with_stdio(0);
	n = io.read(); m = io.read(); p = io.read();
	sz = pow(2, ceil(log2(max(n, m) + 1)));
	for (int i = 0; i <= n; ++i) v1[i] = io.read() % p;
	for (int i = 0; i <= m; ++i) v2[i] = io.read() % p;
	calc(v1, v2, sz, ans, 0);
	for (int i = 0; i <= n + m; ++i) io.write(ans[i], ' ');
}

基本上每一个点都还是可以在 1.3s 左右创过去的。不算特别卡常。
