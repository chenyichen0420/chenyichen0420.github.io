思路分析

观察性质，注意到 k≤4k\le4k≤4，于是每一种选点方案的最少路径数量一定不会超过 222。我们考虑直接统计出最少路径数为 111 的方案数。

注意到，这样的选点方案中所有的点都在一条路径上。于是我们考虑在路径的 LCA 处统计贡献，使用 dpp,idp_{p,i}dpp,i​ 表示链顶在在 ppp 位置，选择了 iii 个节点的方案数。

然后在每一个点合并信息并统计贡献就行了。注意处理 n<mn<mn<m，m=1m=1m=1，并且注意 CnmC_n^mCnm​ 是可能爆 unsigned long long 的。代码如下：

CPP 
#include<iostream>
#include<vector>
#include<queue>
#include<cstring>
#include<string>
#include<algorithm>
#include<cassert>
using namespace std;
#define int long long
constexpr int mod = 998244353;
int t, n, m, dp[200005][5];
vector<int>son[200005];
#ifdef _MSC_VER
#define __int128 long long
#endif
long long ans;
inline void dfs(int p, int f) {
	dp[p][1] = 1;
	for (int sp : son[p]) if (sp != f) {
		dfs(sp, p);
		for (int i = 1;i < m;++i)
			ans -= dp[p][i] * dp[sp][m - i],
			ans %= mod; 
		for (int i = 1;i <= m;++i)
			dp[p][i] = (dp[p][i] + dp[sp][i]) % mod;
		for (int i = 1;i < m;++i)
			dp[p][i] = (dp[p][i] + dp[sp][i - 1]) % mod;
		ans %= mod;
	}
	if(m==1) ans--;
}
inline __int128 cvl(int n, int m) {
	__int128 ret = 1;
	for (int i = 0;i != m;++i) ret = ret * (n - i);
	for (int i = 1;i <= m;++i) ret /= i;
	return ret % mod;
}
signed main() {
	ios::sync_with_stdio(0);
	for (cin >> t;t;t--) {
		cin >> n >> m;
        if(m>n){
            cout<<"0\n";
            continue;
        }
		memset(dp, 0, sizeof dp);
		for (int i = 1, l, r;i != n;++i)
			cin >> l >> r,
			son[l].emplace_back(r),
			son[r].emplace_back(l);
		dfs(1, 0);
		ans += cvl(n, m) * 2;
		cout<<((ans % mod+mod)%mod)<<endl; putchar('\n'); ans = 0;
		for (int i = 1;i <= n;++i) son[i].clear();
	}
}
