思路分析

注意到 r−lr-lr−l 并不大，显然是一个突破口。

我们考虑一下 μ\muμ 的计算中最核心的两个因素：平方因子存在性和质因子个数。

我们思考一下：平方因子和质因子都有可能达到 10910^9109 量级，非常的大，没办法直接枚举，但是我们有没有希望将平方因子的范围压缩一下呢？

我们考虑将这个数在 2×1062\times10^62×106 范围内的质因子给全部除掉，此时他剩余的质因子必定是 >2×106>2\times10^6>2×106 的。而显然，根据数据范围，这样的因子最多只有 222 个不同的。

那不就简单了吗？我们将所有的数的 2×1062\times10^62×106 以下的质因数全部除掉，记录一下有没有平方因子以及质因子个数。

随后，对于每一个仍然 >1>1>1 的数，先检验是不是完全平方数，然后使用 Miller-Rabin 检验是不是质数。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
#define int long long
constexpr int ud[] = { 2,3,5,7,11,13,17,37 };
#ifdef _MSC_VER
#define __int128 long long
#endif
inline int mul(int a, int b, int p) {
    __int128 tmp = a; return tmp * b % p;
}
inline int qpow(__int128 a, int b, int p) {
    __int128 ret = 1;
    for (; b; b >>= 1, a = a * a % p)
        (b & 1) && (ret = ret * a % p);
    return ret;
}
inline bool isprm(int n) {
    if (n < 3 || n % 2 == 0) return n == 2;//特判
    int u = n - 1, t = 0;
    while (u % 2 == 0) u /= 2, ++t;
    for (int a : ud) {
        int v = qpow(a, u, n);
        if (v == 1 || v == n - 1 || v == 0) continue;
        for (int j = 1; j <= t; j++) {
            v = mul(v, v, n);
            if (v == n - 1 && j != t) { v = 1; break; }
            if (v == 1) return 0;
        }
        if (v != 1) return 0;
    }
    return 1;
}
int l, r, v[2000005], c[2000005], p[2000005], pc, ans; bool isp[2000005];
signed main() {
    ios::sync_with_stdio(0); cin >> l >> r;
    for (int i = l; i <= r; ++i) v[i - l] = i, c[i - l] = 1;
    for (int i = 2; i <= 2e6; ++i) {
        if (!isp[i]) p[++pc] = i;
        for (int j = 1; j <= pc && i * p[j] <= 2e6; ++j)
            if (isp[i * p[j]] = 1, i% p[j] == 0)break;
    }
    for (int i = 1; i <= pc; ++i) {
        int pos = l / p[i] * p[i];
        if (pos < l) pos += p[i];
        if (pos > r) continue;
        do {
            v[pos - l] /= p[i]; c[pos - l] = -c[pos - l];
            if (v[pos - l] % p[i] == 0) c[pos - l] = 0;
            while (v[pos - l] % p[i] == 0) v[pos - l] /= p[i];
        } while ((pos += p[i]) <= r);
    }
    for (int i = l; i <= r; ++i)
        if (c[i - l] && v[i - l] > 1) {
            int v1 = sqrt(v[i - l]);
            if (v1 * v1 == v[i - l]) c[i - l] = 0;
            else if (isprm(v[i - l])) c[i - l] = -c[i - l];
        }
    for (int i = l; i <= r; ++i) ans += c[i - l];
    cout << ans << endl;
    return 0;
}
/*
如果我将 10^6 及以下的因子全部除尽了，那么剩下的会是什么？
他最多只能再被质数除两次了。
也就是说，我们首先将其开根，检验是不是质数的平方
随后检查他是不是质数。如果是那就 -\mu(x)，否则不动。
快速检验一个大数是不是质数，使用 Miller-rabin。
*/
