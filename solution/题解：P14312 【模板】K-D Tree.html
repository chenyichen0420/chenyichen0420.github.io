思路分析

K-D Tree 是一种用来相对高效的处理高维对点信息维护的数据结构，一般来说支持在线动态加点加权值等操作。

我们首先注意到这个数据结构是需要支持动态加点的，并且能够处理高维点信息。我们思考一个合理的数据结构形式。

容易想到使用树形结构。并且为了能够动态扩容，我们可能会需要类似于二叉搜索树那样的组织形式。又因为要求高效，所以我们还是希望这个二叉搜索树能够尽可能地平衡。

但是，棘手的地方在于我们需要处理高维点信息，这意味着一般的二叉搜索树或者平衡树并不能直接套到这个地方来。不过这不妨碍我们往这个方向去思考。

注意到核心矛盾是平衡的结构与良好的高维性质。于是我们思考按照维度进行划分。一种非常自然但是有一些问题的想法是按照某一维的值作为主键值构建类似线段树的结构。

问题就在于这一维的值有可能大量的重复，例如在二维情况下点分布呈 "L" 形。

我们再次思考：既然抓住一维不放并不是很正确，那我让每一维轮着来呢？

他会形成这样的结构（图片来自 oiwiki）：

可见其实确实非常的平衡。虽然会有很多的点，但是树高能够被控制在 $\log n+O(1)$ 级别。

那么查询呢？不难想到我们可以对每一个点维护一个“支配区间”，这个“区间”就是他分管的最大范围。然后我们就可以快速地统计一个区间内的所有点的点权之和了。

至于区间修改，也是几乎一样的，打好 tag，及时下传就行了。现在思考下一步：加点。

按照原先的构建方式确实可以对静态的点集构造出相对平衡的树，但是只要我一直往一个小区域加点，你的树就会越来越不平衡，就假掉了。

你可能会想到类似替罪羊树的不平衡重构。这样做确实能够在可以接受的时间复杂度内将树的高度维持在 $O(\log n)$ 级别。但是很可惜，这是不正确的。

当树高为 $\log n+O(1)$ 时，查询的时间复杂度是 $O(n^{1-\frac{1}{k}})$ 的，其中 $k$ 为维度数，但是当树高退化到 $O(\log n)$ 之后，这个复杂度也是不正确的，详见 oi-wiki 上面的证明与紧随其后的说明。

我们需要能够快速正确维护新点的方式。幸运的是，我们有二进制分组大法。

我们考虑维护若干棵 $2$ 的自然数次幂的 K-D Tree。当插入一个新的点的时候，我们将其视为一个大小为 $1$ 的 K-D Tree，并不断合并相同大小的子树。

显然这个过程是类似于二进制 $+1$ 进位的过程。容易分析出这样做的时间复杂度是 $O(n\log^2n)$ 的，因为 K-D Tree 的构建是 $O(n\log n)$ 的。

这样子做查询是不是又带一个 $\log$ 呢？事实上，我们粗略的分析时间复杂度为 $O(\sum_{i\ge0}(\frac{n}{2^i})^{1-\frac{1}{k}})=O(n^{1-\frac{1}{k}})$。所以其实时间复杂度没变。

讲完了，剩下的就是写屎山然后成功调疯了。代码如下：

#include<bits/stdc++.h>
using namespace std;
#define int long long
inline void tmin(int& l, const int r) { (l > r) && (l = r); }
inline void tmax(int& l, const int r) { (l < r) && (l = r); }
using ini = initializer_list<int>;
template<const int wd>
struct pos {
	int p[wd];
	pos<wd>() { memset(p, 0, sizeof p); }
	pos<wd>(ini pv) {
		memcpy(p, pv.begin(), pv.size() << 2);
	}
	inline bool operator<(const pos<wd>& r) const{
		for (int i = 0;i != wd;++i)
			if (p[i] != r.p[i]) return p[i] < r.p[i];
		return 1;
	}
	inline bool operator==(const pos<wd>& r) const{
		for (int i = 0;i != wd;++i)
			if (p[i] != r.p[i]) return 0;
		return 1;
	}
};
template<const int wd>
inline bool ninter(const pos<wd>& alu, const pos<wd>& ard, const pos<wd>& blu, const pos<wd>& brd) {
	for (int i = 0;i != wd;++i)
		if (blu.p[i] > ard.p[i] || alu.p[i] > brd.p[i]) return 1;
	return 0;
}
template<const int wd>
inline bool finter(const pos<wd>& alu, const pos<wd>& ard, const pos<wd>& blu, const pos<wd>& brd) {
	for (int i = 0;i != wd;++i)
		if (blu.p[i] < alu.p[i] || brd.p[i] > ard.p[i]) return 0;
	return 1;
}
template<const int wd,const int szs>
class kdtree {
	struct node {
		pos<wd> np, lu, rd; int ls, rs, v, sm, f, sz, ad;
		node() :ls(0), rs(0), v(0), sm(0), f(0), sz(0), ad(0) {};
		inline void add(int t){sm+=t*sz; ad+=t; v+=t;}
	}re[szs];
	int rt[20]; stack<int>ps;
	vector<pair<pos<wd>, int>>mt;
	inline void ret(int p) { ps.emplace(p); }
	inline int get() {
		int p = ps.top(); ps.pop();
		memset(&re[p], 0, sizeof(re[p]));
		return p;
	}
	inline void pup(int p) {
		re[p].sm = re[re[p].ls].sm + re[re[p].rs].sm + re[p].v;
		re[p].sz = re[re[p].ls].sz + re[re[p].rs].sz + 1;
		re[p].lu = re[p].rd = re[p].np;
		for (int i = 0;i != wd;++i) {
			if (re[p].ls)
				tmin(re[p].lu.p[i], re[re[p].ls].lu.p[i]),
				tmax(re[p].rd.p[i], re[re[p].ls].rd.p[i]);
			if (re[p].rs)
				tmin(re[p].lu.p[i], re[re[p].rs].lu.p[i]),
				tmax(re[p].rd.p[i], re[re[p].rs].rd.p[i]);
		}
	}
	inline void pud(int p){
		int&t=re[p].ad; if(!t) return;
		if(re[p].ls) re[re[p].ls].add(t);
		if(re[p].rs) re[re[p].rs].add(t);
		t=0;
	}
	inline void getall(int p) {
		mt.emplace_back(re[p].np, re[p].v);
		ret(p); pud(p);
		if (re[p].ls) getall(re[p].ls);
		if (re[p].rs) getall(re[p].rs);
	}
	inline int build(int l, int r, int cf) {
		if (l > r) return 0; 
		int p = get();
		if (l == r) {
			re[p].np = mt[l].first;
			re[p].v = mt[l].second;
			re[p].f = cf;
			pup(p); 
			return p;
		}
		int md = l + r >> 1;
		nth_element(mt.begin() + l, mt.begin() + md, mt.begin() + r + 1,
			[&](const pair<pos<wd>, int>& l, const pair<pos<wd>, int>& r) {
				return l.first.p[cf] < r.first.p[cf];
			});
		re[p].np = mt[md].first;
		re[p].v = mt[md].second;
		re[p].f = cf; cf = (cf + 1) % wd;
		re[p].ls = build(l, md - 1, cf);
		re[p].rs = build(md + 1, r, cf);
		pup(p);
		return p;
	}
	inline int que(const pos<wd>& lu, const pos<wd>& rd, int p) {
		if (!p || ninter(lu, rd, re[p].lu, re[p].rd)) return 0;
		if (finter(lu, rd, re[p].lu, re[p].rd)) return re[p].sm;
		int ret = 0; pud(p);
		if (finter(lu, rd, re[p].np, re[p].np)) ret += re[p].v;
		ret += que(lu, rd, re[p].ls); 
		ret += que(lu, rd, re[p].rs);
		return ret;
	}
	inline void add(const pos<wd>& lu, const pos<wd>& rd, int v,int p) {
		if (!p || ninter(lu, rd, re[p].lu, re[p].rd)) return;
		if (finter(lu, rd, re[p].lu, re[p].rd)) return re[p].add(v);
		if (finter(lu, rd, re[p].np, re[p].np)) re[p].v+=v;
		pud(p);
		add(lu, rd, v, re[p].ls); 
		add(lu, rd, v, re[p].rs);
		pup(p);
	}
public:
	inline void init() {
		memset(rt, 0, sizeof rt);
		for (int i = szs;i;i--) ret(i);
	}
	inline void ins(const pos<wd>& p, int v) {
		int lp = 0; mt.emplace_back(p, v);
		while (rt[lp]) getall(rt[lp]), rt[lp] = 0, lp++;
		sort(mt.begin(), mt.end());
		rt[lp] = build(0, mt.size()-1, 0); mt.clear();
	}
	inline void add(const pos<wd>&pa,const pos<wd>&pb,int v){
		for (int i = 0;i != 20;++i) add(pa, pb, v, rt[i]);
	}
	inline int que(const pos<wd>&pa, const pos<wd>&pb) {
		int ret = 0;
		for (int i = 0;i != 20;++i)
			ret += que(pa, pb, rt[i]);
		return ret;
	}
};
