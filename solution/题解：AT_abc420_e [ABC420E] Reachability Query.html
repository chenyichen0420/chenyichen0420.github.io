思路分析

带权并查集的很板的一道题目。

我们考虑记录出连通块中黑点的个数。显然连通块中有黑点与可以到达黑点是等价的。

修改点的时候就将对应的连通块的黑点的计数增减一下。

合并的时候将前者的父节点指向后者，然后将后者的黑点计数加上前者的即可。

代码如下：

#include<bits/stdc++.h>
using namespace std;
#define int long long
int n, m, f[200005], c[200005], v[200005];
inline int find(int p) {
	return f[p] != p ? f[p] = find(f[p]) : p;
}
inline void merge(int l, int r) {
	l = find(l); r = find(r);
	if (l == r) return;
	f[l] = r; c[r] += c[l];
}
signed main() {
	ios::sync_with_stdio(0);
	cin >> n >> m;
	for (int i = 1; i <= n; ++i) f[i] = i;
	for (int i = 1, o, l, r; i <= m; ++i)
		if (cin >> o >> l, o == 1) cin >> r, merge(l, r);
		else if (o & 1) puts(c[find(l)] ? "Yes" : "No");
		else v[l] ^= 1, c[find(l)] += v[l] * 2 - 1;
}
