思路分析

首先不难想到在 $n$ 足够大的时候，直接取 $\frac{b_i}{a_i-b_i}$ 最大的会最优，毕竟 $a_i-b_i$ 是折损的数量，同等折损数量下拿回来的越多就越赚，没有毛病！

但是，根据我们之前在做各种题目的经验来看，如果就这样取，到一定程度之后反而有可能不优。为什么呢？有可能一个在前面来说相对来说比较赚的决策会因为在最后占据了过多的空间导致有的没有被利用起来，损失惨重。

但是我们不难观察到题面当中一个较为异常的地方就在于它的 $A,B\le300$，甚至支持 $V^3$ 级别的运算，多半有用。再仔细思考一下，豁然开朗：

当 $n>V^2(V=300)$ 的时候，根据数学直觉（好吧其实是我不会证），$n$ 一定可以被 $[1,300]$ 中的数字以不同的形式表示出来，也就是说不管你这个选多选少，我都有办法通过调整别的数字来保证总占用不变。

但是当 $n\le V^2$ 就有可能了。这部分我们直接动态规划预处理出来就行了。

但是也有一些问题，这样做是 $O(V^2m)$ 的，确实卡不过去，怎么办？

我们发现无论如何，相同的 $A_i$ 对应的只有最大的那个 $B_i$ 是有用的，也就是有用的转移只有 $V$ 个。从而时间复杂度是 $O(V^3)$ 的。算上前边暴力处理大数值范围，总时间复杂度差不多是 $O(V^3+m\log m)$ 的，能过。

代码如下：

CPP 
#include <bits/stdc++.h>
using namespace std;
#define int long long
struct node {
	int a, b;
	inline bool operator<(const node& r) {
		if ((a - b) * r.b != (r.a - r.b) * b)
			return (a - b) * r.b < (r.a - r.b) * b;
		else return a < r.a;
	}
}v[200005];
struct bigint {
	int v[256], w;
	bigint(int t = 0) {
		memset(v, 0, sizeof v); if (!t)w = 1;
		while (t) v[++w] = t % 10, t /= 10;
	}
	inline void operator+=(const int& r) {
		v[1] += r;
		for (int i = 1; i != w; ++i)
			v[i + 1] += v[i] / 10, v[i] %= 10;
		while (v[w] > 9)
			v[w + 1] += v[w] / 10,
			v[w] %= 10, w++;
	}
	inline void opt() {
		for (int i = w; i >= 1; i--) cout << v[i];
	}
}ans;
int n, m, dp[90005], bt[305];
signed main(){
	ios::sync_with_stdio(0); cin >> n >> m; ans = bigint(n);
	for (int i = 1; i <= m; ++i) cin >> v[i].a >> v[i].b;
	sort(v + 1, v + m + 1);
	for (int i = 1; i <= m; ++i)
		if (n >= 90000) {
			ans += (n - 90000) / (v[i].a - v[i].b) * v[i].b;
			n -= (n - 90000) / (v[i].a - v[i].b) * (v[i].a - v[i].b);
			while (n >= 90000) n -= v[i].a - v[i].b, ans += v[i].b;
		}
	for (int i = 1; i <= m; ++i)
		bt[v[i].a] = max(bt[v[i].a], v[i].b);
	for (int i = 1; i <= 90000; ++i)
		for (int j = 1; j <= 300; ++j)
			if (i >= j) dp[i] = max(dp[i], dp[i - j + bt[j]] + bt[j]);
	ans += dp[n];
	ans.opt();
}
