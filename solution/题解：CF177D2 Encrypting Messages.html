解法

我们其实很容易得到对于数列 aaa 中下标为 iii 的数应该加的那些数在 bbb 中的下标：从 max⁡(p−n+m,1)\max(p-n+m,1)max(p−n+m,1) 到 min⁡(p,m)\min(p,m)min(p,m)。要推的话也很简单。

对于最后一个位置，如果 iii 大于等于 mmm，那么在第 i−m+1i-m+1i−m+1 次操作时会加上最后一个数 bmb_mbm​，否则最后也只能在第 111 次操作时加上 bpb_pbp​。第一个位置以此类推，留给读者自行思考。

这样，每一次 aia_iai​ 都先加上 ∑max⁡(p−n+m,1)≤j≤min⁡(p,m)bj\sum_{\max(p-n+m,1)\le j\le\min(p,m)}b_j∑max(p−n+m,1)≤j≤min(p,m)​bj​，然后取一次模输出就行了。

代码如下：

CPP 
#include<bits/stdc++.h>
using namespace std;
int n, m, c, a[100005], b[100005], qh[100005];
inline int gtfs(int p) {
	if (p + m - 1 <= n) return 1;
	return p - n + m;
}
inline int gtlt(int p) {
	if (p >= m) return m;
	return p;
}
signed main() {
	ios::sync_with_stdio(0);
	cin >> n >> m >> c;
	for (int i = 1; i <= n; ++i) cin >> a[i];
	for (int i = 1; i <= m; ++i) cin >> b[i], qh[i] = qh[i - 1] + b[i];
	for (int i = 1; i <= n; ++i)
		a[i] += qh[gtlt(i)] - qh[gtfs(i) - 1],
		cout << a[i] % c << " ";
	return 0;
}
